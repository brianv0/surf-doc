184  -- Mux incoming data
185  process ( vc0FrameTxValid, vc0FrameTxSOF, vc0FrameTxEOF, vc0FrameTxEOFE, vc0FrameTxData, vc0RemAlmostFull,
186  vc1FrameTxValid, vc1FrameTxSOF, vc1FrameTxEOF, vc1FrameTxEOFE, vc1FrameTxData, Vc1RemAlmostFull,

205  muxRemAlmostFull <= vc1RemAlmostFull;
206  muxSerial <= vc1Serial;
207  when "10" =>
208  muxFrameTxValid <= vc2FrameTxValid;
209  muxFrameTxSOF <= vc2FrameTxSOF;
210  muxFrameTxEOF <= vc2FrameTxEOF;
211  muxFrameTxEOFE <= vc2FrameTxEOFE;
212  muxFrameTxData <= vc2FrameTxData;
213  muxRemAlmostFull <= vc2RemAlmostFull;
214  muxSerial <= vc2Serial;
215  when others =>
216  muxFrameTxValid <= vc3FrameTxValid;
217  muxFrameTxSOF <= vc3FrameTxSOF;
218  muxFrameTxEOF <= vc3FrameTxEOF;
219  muxFrameTxEOFE <= vc3FrameTxEOFE;
220  muxFrameTxData <= vc3FrameTxData;
221  muxRemAlmostFull <= vc3RemAlmostFull;
222  muxSerial <= vc3Serial;
223  end case;
224  end process;
225 
226 
227  -- Choose data for SOF & EOF Positions
228  GEN_DATA: for i in 0 to (TX_LANE_CNT_G-1) generate
229 
230  -- SOF, vc number and serial number
231  socWord(i*16+15 downto i*16+14) <= schTxDataVc;
232  socWord(i*16+13 downto i*16+8) <= muxSerial;
233  socWord(i*16+7 downto i*16) <= (others=>'0');
234 
235  -- EOF, buffer status
236  eocWord(i*16+15 downto i*16+12) <= intOverflow;
237  eocWord(i*16+11 downto i*16+8) <= vc3LocAlmostFull & vc2LocAlmostFull & vc1LocAlmostFull & vc0LocAlmostFull;
238  eocWord(i*16+7 downto i*16) <= (others=>'0');
239  end generate;
240 
241 
242  -- Simple state machine to control transmission of data frames
243  process ( pgpTxClk ) begin
244  if rising_edge(pgpTxClk) then
245  if pgpTxClkRst = '1' then
246  curState <= ST_IDLE_C after TPD_G;
247  cellCnt <= (others=>'0') after TPD_G;
248  int0FrameTxReady <= '0' after TPD_G;
249  int1FrameTxReady <= '0' after TPD_G;
250  int2FrameTxReady <= '0' after TPD_G;
251  int3FrameTxReady <= '0' after TPD_G;
252  intTimeout <= '0' after TPD_G;
253  schTxSOF <= '0' after TPD_G;
254  schTxEOF <= '0' after TPD_G;
255  schTxAck <= '0' after TPD_G;
256  vc0Serial <= (others=>'0') after TPD_G;
257  vc1Serial <= (others=>'0') after TPD_G;
258  vc2Serial <= (others=>'0') after TPD_G;
259  vc3Serial <= (others=>'0') after TPD_G;
260  curTypeLast <= (others=>'0') after TPD_G;
261  intOverflow <= (others=>'0') after TPD_G;
262  elsif pgpTxClkEn = '1' then
263  -- State control
264  if pgpTxLinkReady = '0' then
265  curState <= ST_IDLE_C after TPD_G;
266  else
267  curState <= nxtState after TPD_G;
268  end if;
269 
270  -- Payload Counter
271  if cellCntRst = '1' then
272  cellCnt <= (others=>'1') after TPD_G;
273  elsif cellCnt /= 0 then
274  cellCnt <= cellCnt - 1 after TPD_G;
275  end if;
276 
277  -- Outgoing ready signal
278  case schTxDataVc is
279  when "00" =>
280  int0FrameTxReady <= nxtFrameTxReady after TPD_G;
281  int1FrameTxReady <= '0' after TPD_G;
282  int2FrameTxReady <= '0' after TPD_G;
283  int3FrameTxReady <= '0' after TPD_G;
284  when "01" =>
285  int0FrameTxReady <= '0' after TPD_G;
286  int1FrameTxReady <= nxtFrameTxReady after TPD_G;
287  int2FrameTxReady <= '0' after TPD_G;
288  int3FrameTxReady <= '0' after TPD_G;
289  when "10" =>
290  int0FrameTxReady <= '0' after TPD_G;
291  int1FrameTxReady <= '0' after TPD_G;
292  int2FrameTxReady <= nxtFrameTxReady after TPD_G;
293  int3FrameTxReady <= '0' after TPD_G;
294  when others =>
295  int0FrameTxReady <= '0' after TPD_G;
296  int1FrameTxReady <= '0' after TPD_G;
297  int2FrameTxReady <= '0' after TPD_G;
298  int3FrameTxReady <= nxtFrameTxReady after TPD_G;
299  end case;
300 
301  -- Register timeout request
302  if schTxReq = '1' then
303  intTimeout <= schTxTimeout after TPD_G;
304  end if;
305 
306  -- Update Last Type
307  curTypeLast <= nxtTypeLast after TPD_G;
308 
309  -- VC Serial Numbers
310  if pgpTxLinkReady = '0' then
311  vc0Serial <= (others=>'0') after TPD_G;
312  vc1Serial <= (others=>'0') after TPD_G;
313  vc2Serial <= (others=>'0') after TPD_G;
314  vc3Serial <= (others=>'0') after TPD_G;
315  elsif serialCntEn = '1' then
316  case schTxDataVc is
317  when "00" => vc0Serial <= vc0Serial + 1 after TPD_G;
318  when "01" => vc1Serial <= vc1Serial + 1 after TPD_G;
319  when "10" => vc2Serial <= vc2Serial + 1 after TPD_G;
320  when others => vc3Serial <= vc3Serial + 1 after TPD_G;
321  end case;
322  end if;
323 
324  -- Scheduler Signals
325  schTxSOF <= nxtTxSOF after TPD_G;
326  schTxEOF <= nxtTxEOF after TPD_G;
327  schTxAck <= nxtTxAck after TPD_G;
328 
329  -- Overflow Latch Until Send
330  if vc0LocOverflow = '1' then
331  intOverflow(0) <= '1' after TPD_G;
332  elsif curState = ST_EMPTY_C or curState = ST_EOC_C then
333  intOverflow(0) <= '0' after TPD_G;
334  end if;
335 
336  if vc1LocOverflow = '1' then
337  intOverflow(1) <= '1' after TPD_G;
338  elsif curState = ST_EMPTY_C or curState = ST_EOC_C then
339  intOverflow(1) <= '0' after TPD_G;
340  end if;
341 
342  if vc2LocOverflow = '1' then
343  intOverflow(2) <= '1' after TPD_G;
344  elsif curState = ST_EMPTY_C or curState = ST_EOC_C then
345  intOverflow(2) <= '0' after TPD_G;
346  end if;
347 
348  if vc3LocOverflow = '1' then
349  intOverflow(3) <= '1' after TPD_G;
350  elsif curState = ST_EMPTY_C or curState = ST_EOC_C then
351  intOverflow(3) <= '0' after TPD_G;
352  end if;
353  end if;
354  end if;
355  end process;
356 
357 
358  -- Drive TX Ready
359  vc0FrameTxReady <= int0FrameTxReady;
360  vc1FrameTxReady <= int1FrameTxReady;
361  vc2FrameTxReady <= int2FrameTxReady;
362  vc3FrameTxReady <= int3FrameTxReady;
363 
364 
365  -- Async state control
366  process ( curState, schTxIdle, schTxReq, intTimeout, cellCnt, eocWord, socWord, curTypeLast,
367  muxFrameTxValid, muxFrameTxSOF, muxFrameTxEOF, muxFrameTxEOFE, muxFrameTxData,
368  muxRemAlmostFull ) begin
369  case curState is
370 
371  -- Idle
372  when ST_IDLE_C =>
373  cellCntRst <= '1';
374  nxtFrameTxReady <= '0';
375  nxtType <= TX_DATA_C;
376  nxtData <= (others=>'0');
377  nxtTxSOF <= '0';
378  nxtTxEOF <= '0';
379  nxtTxAck <= '0';
380  serialCntEn <= '0';
381  nxtTypeLast <= (others=>'0');
382 
383  -- Idle request
384  if schTxIdle = '1' then
385  nxtState <= ST_EMPTY_C;
386 
387  -- Cell transmit request
388  elsif schTxReq = '1' then
389  nxtState <= ST_SOC_C;
390  else
391  nxtState <= curState;
392  end if;
393 
394  -- Send empty cell
395  when ST_EMPTY_C =>
396  cellCntRst <= '1';
397  nxtFrameTxReady <= '0';
398  nxtType <= TX_EOC_C;
399  nxtTxSOF <= '0';
400  nxtTxEOF <= '0';
401  nxtTxAck <= '1';
402  serialCntEn <= '0';
403  nxtData <= eocWord;
404  nxtTypeLast <= (others=>'0');
405 
406  -- Go back to idle
407  nxtState <= ST_IDLE_C;
408 
409  -- Send first charactor of cell, assert ready
410  when ST_SOC_C =>
411  cellCntRst <= '1';
412  nxtFrameTxReady <= not intTimeout;
413  nxtTxEOF <= '0';
414  nxtTxAck <= '0';
415  serialCntEn <= '0';
416  nxtData <= socWord;
417  nxtTypeLast <= (others=>'0');
418 
419  -- Determine type
420  if intTimeout = '1' then
421  nxtType <= TX_SOC_C;
422  nxtTxSOF <= '0';
423  elsif muxFrameTxSOF = '1' then
424  nxtType <= TX_SOF_C;
425  nxtTxSOF <= '1';
426  else
427  nxtType <= TX_SOC_C;
428  nxtTxSOF <= '0';
429  end if;
430 
431  -- Move on to normal data
432  nxtState <= ST_DATA_C;
433 
434  -- Send data
435  when ST_DATA_C =>
436  cellCntRst <= '0';
437  nxtTxEOF <= '0';
438  nxtTxSOF <= '0';
439  nxtTxAck <= '0';
440  serialCntEn <= '0';
441  nxtData <= muxFrameTxData;
442 
443  -- Timeout frame, force EOFE
444  if intTimeout = '1' then
445  nxtType <= TX_DATA_C;
446  nxtTypeLast <= TX_EOFE_C;
447  nxtState <= ST_CRCA_C;
448  nxtFrameTxReady <= '0';
449 
450  -- Valid is de-asserted
451  elsif muxFrameTxValid = '0' then
452  nxtTypeLast <= TX_EOC_C;
453  nxtFrameTxReady <= '0';
454  nxtType <= TX_CRCA_C;
455 
456  -- One or two CRC words?
457  if TX_LANE_CNT_G = 1 then
458  nxtState <= ST_CRCB_C;
459  else
460  nxtState <= ST_EOC_C;
461  end if;
462  else
463  nxtType <= TX_DATA_C;
464 
465  -- EOFE is asserted
466  if muxFrameTxEOFE = '1' then
467  nxtTypeLast <= TX_EOFE_C;
468  nxtState <= ST_CRCA_C;
469  nxtFrameTxReady <= '0';
470 
471  -- EOF is asserted
472  elsif muxFrameTxEOF = '1' then
473  nxtTypeLast <= TX_EOF_C;
474  nxtState <= ST_CRCA_C;
475  nxtFrameTxReady <= '0';
476 
477  -- Pause is asserted
478  elsif muxRemAlmostFull = '1' then
479  nxtTypeLast <= TX_EOC_C;
480  nxtState <= ST_CRCA_C;
481  nxtFrameTxReady <= '0';
482 
483  -- Cell size reached
484  elsif cellCnt = 0 then
485  nxtTypeLast <= TX_EOC_C;
486  nxtState <= ST_CRCA_C;
487  nxtFrameTxReady <= '0';
488 
489  -- Keep sending cell data
490  else
491  nxtTypeLast <= curTypeLast;
492  nxtState <= curState;
493  nxtFrameTxReady <= '1';
494  nxtType <= TX_DATA_C;
495  end if;
496  end if;
497 
498  -- Send CRC A
499  when ST_CRCA_C =>
500  cellCntRst <= '1';
501  nxtTxEOF <= '0';
502  nxtTxSOF <= '0';
503  nxtTxAck <= '0';
504  serialCntEn <= '0';
505  nxtData <= (others=>'0');
506  nxtType <= TX_CRCA_C;
507  nxtTypeLast <= curTypeLast;
508  nxtFrameTxReady <= '0';
509 
510  -- One or two CRC words?
511  if TX_LANE_CNT_G = 1 then
512  nxtState <= ST_CRCB_C;
513  else
514  nxtState <= ST_EOC_C;
515  end if;
516 
517  -- Send CRC B
518  when ST_CRCB_C =>
519  cellCntRst <= '1';
520  nxtTxEOF <= '0';
521  nxtTxSOF <= '0';
522  nxtTxAck <= '0';
523  serialCntEn <= '0';
524  nxtData <= (others=>'0');
525  nxtType <= TX_CRCB_C;
526  nxtTypeLast <= curTypeLast;
527  nxtFrameTxReady <= '0';
528  nxtState <= ST_EOC_C;
529 
530  -- Send End of Cell
531  when ST_EOC_C =>
532  cellCntRst <= '1';
533  nxtTxSOF <= '0';
534  nxtTxAck <= '1';
535  serialCntEn <= '1';
536  nxtData <= eocWord;
537  nxtType <= curTypeLast;
538  nxtTypeLast <= curTypeLast;
539  nxtFrameTxReady <= '0';
540  nxtState <= ST_IDLE_C;
541 
542  -- EOF?
543  if curTypeLast /= TX_EOC_C then
544  nxtTxEOF <= '1';
545  else
546  nxtTxEOF <= '0';
547  end if;
548 
549  -- Default State
550  when others =>
551  cellCntRst <= '0';
552  nxtTxEOF <= '0';
553  nxtTxSOF <= '0';
554  nxtTxAck <= '0';
555  serialCntEn <= '0';
556  nxtData <= (others=>'0');
557  nxtType <= (others=>'0');
558  nxtTypeLast <= (others=>'0');
559  nxtFrameTxReady <= '0';
560  nxtState <= ST_IDLE_C;
561  end case;
562  end process;
563 
564 
565  -- Delay chain to allow CRC data to catch up.
566  process ( pgpTxClk ) begin
567  if rising_edge(pgpTxClk) then

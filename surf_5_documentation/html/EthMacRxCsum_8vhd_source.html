133  v.fragDet := r.fragDet(EMAC_CSUM_PIPELINE_C downto 0) & r.fragDet(0);
134  v.eofeDet := r.eofeDet(EMAC_CSUM_PIPELINE_C downto 0) & r.eofeDet(0);
135  v.ipv4Det := r.ipv4Det(EMAC_CSUM_PIPELINE_C downto 0) & r.ipv4Det(0);
136  v.udpDet := r.udpDet(EMAC_CSUM_PIPELINE_C downto 0) & r.udpDet(0);
137  v.tcpDet := r.tcpDet(EMAC_CSUM_PIPELINE_C downto 0) & r.tcpDet(0);

139  v.protLen := r.protLen(EMAC_CSUM_PIPELINE_C downto 0) & r.protLen(0);
140  v.protCsum := r.protCsum(EMAC_CSUM_PIPELINE_C downto 0) & r.protCsum(0);
141 
142  -- Reset the flags
143  v.tKeep := (others => '0');
144  v.mAxisMaster.tValid := '0';
145  v.mAxisMaster.tLast := '0';
146 
147  -- Check for tLast in pipeline
148  if (v.mAxisMasters(EMAC_CSUM_PIPELINE_C+1).tLast = '1') then
149  -- Check if IPv4 is detected and being checked
150  if (r.ipv4Det(EMAC_CSUM_PIPELINE_C+1) = '1') and (ipCsumEn = '1') then
151  -- Forward the result of checksum calculation
152  axiStreamSetUserBit(EMAC_AXIS_CONFIG_C, v.mAxisMasters(EMAC_CSUM_PIPELINE_C+1), EMAC_IPERR_BIT_C, not(r.valid(0)));
153  axiStreamSetUserBit(EMAC_AXIS_CONFIG_C, v.mAxisMasters(EMAC_CSUM_PIPELINE_C+1), EMAC_EOFE_BIT_C, not(r.valid(0)));
154  end if;
155  -- Check if UDP is detected and being checked
156  if (r.ipv4Det(EMAC_CSUM_PIPELINE_C+1) = '1') and (r.udpDet(EMAC_CSUM_PIPELINE_C+1) = '1') and (udpCsumEn = '1') and (r.fragDet(EMAC_CSUM_PIPELINE_C+1) = '0') then
157  -- Forward the result of checksum calculation
158  axiStreamSetUserBit(EMAC_AXIS_CONFIG_C, v.mAxisMasters(EMAC_CSUM_PIPELINE_C+1), EMAC_UDPERR_BIT_C, not(r.valid(1)));
159  axiStreamSetUserBit(EMAC_AXIS_CONFIG_C, v.mAxisMasters(EMAC_CSUM_PIPELINE_C+1), EMAC_EOFE_BIT_C, not(r.valid(1)));
160  -- Check for mismatch in IPv4 length with UDP length
161  if (r.ipv4Len(EMAC_CSUM_PIPELINE_C+1) /= (r.protLen(EMAC_CSUM_PIPELINE_C+1) + 20)) then
162  -- Set the error flags
163  axiStreamSetUserBit(EMAC_AXIS_CONFIG_C, v.mAxisMasters(EMAC_CSUM_PIPELINE_C+1), EMAC_UDPERR_BIT_C, '1');
164  axiStreamSetUserBit(EMAC_AXIS_CONFIG_C, v.mAxisMasters(EMAC_CSUM_PIPELINE_C+1), EMAC_EOFE_BIT_C, '1');
165  end if;
166  end if;
167  -- Check if TCP is detected and being checked
168  if (r.ipv4Det(EMAC_CSUM_PIPELINE_C+1) = '1') and (r.tcpDet(EMAC_CSUM_PIPELINE_C+1) = '1') and (tcpCsumEn = '1') and (r.fragDet(EMAC_CSUM_PIPELINE_C+1) = '0') then
169  -- Forward the result of checksum calculation
170  axiStreamSetUserBit(EMAC_AXIS_CONFIG_C, v.mAxisMasters(EMAC_CSUM_PIPELINE_C+1), EMAC_TCPERR_BIT_C, not(r.valid(1)));
171  axiStreamSetUserBit(EMAC_AXIS_CONFIG_C, v.mAxisMasters(EMAC_CSUM_PIPELINE_C+1), EMAC_EOFE_BIT_C, not(r.valid(1)));
172  end if;
173  end if;

175  -- State Machine
176  case r.state is
177  ----------------------------------------------------------------------
178  when IDLE_S =>
179  -- Reset the flags and counter
180  v.fragDet(0) := '0';
181  v.eofeDet(0) := '0';
182  v.ipv4Det(0) := '0';
183  v.udpDet(0) := '0';
184  v.tcpDet(0) := '0';
185  v.tcpFlag := '0';
186  -- Check for valid data
187  if (sAxisMaster.tValid = '1') then
188  -- Move the data
189  v.mAxisMaster := sAxisMaster;
190  -- Check for no EOF
191  if (sAxisMaster.tLast = '0') then
192  -- Check if NON-VLAN
193  if (VLAN_G = false) then
194  -- Check for EtherType = IPV4 = 0x0800
195  if (sAxisMaster.tData(111 downto 96) = IPV4_TYPE_C) then
196  -- Set the flag
197  v.ipv4Det(0) := '1';
198  end if;
199  -- Fill in the IPv4 header checksum
200  v.ipv4Hdr(0) := sAxisMaster.tData(119 downto 112); -- IPVersion + Header length
201  v.ipv4Hdr(1) := sAxisMaster.tData(127 downto 120); -- DSCP and ECN
202  end if;
203  -- Next state
204  v.state := IPV4_HDR0_S;
205  end if;
206  end if;
207  ----------------------------------------------------------------------
208  when IPV4_HDR0_S =>
209  -- Check for valid data
210  if (sAxisMaster.tValid = '1') then
211  -- Move the data
212  v.mAxisMaster := sAxisMaster;
213  -- Check for EOF
214  if (sAxisMaster.tLast = '1') then
215  -- Set the error flag if IPv4 is detected and being checked
216  axiStreamSetUserBit(EMAC_AXIS_CONFIG_C, v.mAxisMaster, EMAC_IPERR_BIT_C, r.ipv4Det(0));
217  -- Next state
218  v.state := IDLE_S;
219  else
220  -- Check if NON-VLAN
221  if (VLAN_G = false) then
222  -- Fill in the IPv4 header checksum
223  v.ipv4Hdr(2) := sAxisMaster.tData(7 downto 0); -- IPV4_Length(15 downto 8)
224  v.ipv4Hdr(3) := sAxisMaster.tData(15 downto 8); -- IPV4_Length(7 downto 0)
225  v.ipv4Hdr(4) := sAxisMaster.tData(23 downto 16); -- IPV4_ID(15 downto 8)
226  v.ipv4Hdr(5) := sAxisMaster.tData(31 downto 24); -- IPV4_ID(7 downto 0)
227  v.ipv4Hdr(6) := sAxisMaster.tData(39 downto 32); -- Flags(2 downto 0) and Fragment Offsets(12 downto 8)
228  v.ipv4Hdr(7) := sAxisMaster.tData(47 downto 40); -- Fragment Offsets(7 downto 0)
229  v.ipv4Hdr(8) := sAxisMaster.tData(55 downto 48); -- Time-To-Live
230  v.ipv4Hdr(9) := sAxisMaster.tData(63 downto 56); -- Protocol
231  v.ipv4Hdr(10) := sAxisMaster.tData(71 downto 64); -- IPV4_Checksum(15 downto 8)
232  v.ipv4Hdr(11) := sAxisMaster.tData(79 downto 72); -- IPV4_Checksum(7 downto 0)
233  v.ipv4Hdr(12) := sAxisMaster.tData(87 downto 80); -- Source IP Address
234  v.ipv4Hdr(13) := sAxisMaster.tData(95 downto 88); -- Source IP Address
235  v.ipv4Hdr(14) := sAxisMaster.tData(103 downto 96); -- Source IP Address
236  v.ipv4Hdr(15) := sAxisMaster.tData(111 downto 104); -- Source IP Address
237  v.ipv4Hdr(16) := sAxisMaster.tData(119 downto 112); -- Destination IP Address
238  v.ipv4Hdr(17) := sAxisMaster.tData(127 downto 120); -- Destination IP Address
239  -- Fill in the TCP/UDP checksum
240  v.tData(63 downto 0) := sAxisMaster.tData(127 downto 80) & sAxisMaster.tData(63 downto 56) & x"00";
241  v.tKeep(7 downto 0) := (others => '1');
242  else
243  -- Check for EtherType = IPV4 = 0x0800
244  if (sAxisMaster.tData(15 downto 0) = IPV4_TYPE_C) then
245  -- Set the flag
246  v.ipv4Det(0) := '1';
247  end if;
248  -- Fill in the IPv4 header checksum
249  v.ipv4Hdr(0) := sAxisMaster.tData(23 downto 16); -- IPVersion + Header length
250  v.ipv4Hdr(1) := sAxisMaster.tData(31 downto 24); -- DSCP and ECN
251  v.ipv4Hdr(2) := sAxisMaster.tData(39 downto 32); -- IPV4_Length(15 downto 8)
252  v.ipv4Hdr(3) := sAxisMaster.tData(47 downto 40); -- IPV4_Length(7 downto 0)
253  v.ipv4Hdr(4) := sAxisMaster.tData(55 downto 48); -- IPV4_ID(15 downto 8)
254  v.ipv4Hdr(5) := sAxisMaster.tData(63 downto 56); -- IPV4_ID(7 downto 0)
255  v.ipv4Hdr(6) := sAxisMaster.tData(71 downto 64); -- Flags(2 downto 0) and Fragment Offsets(12 downto 8)
256  v.ipv4Hdr(7) := sAxisMaster.tData(79 downto 72); -- Fragment Offsets(7 downto 0)
257  v.ipv4Hdr(8) := sAxisMaster.tData(87 downto 80); -- Time-To-Live
258  v.ipv4Hdr(9) := sAxisMaster.tData(95 downto 88); -- Protocol
259  v.ipv4Hdr(10) := sAxisMaster.tData(103 downto 96); -- IPV4_Checksum(15 downto 8)
260  v.ipv4Hdr(11) := sAxisMaster.tData(111 downto 104); -- IPV4_Checksum(7 downto 0)
261  v.ipv4Hdr(12) := sAxisMaster.tData(119 downto 112); -- Source IP Address
262  v.ipv4Hdr(13) := sAxisMaster.tData(127 downto 120); -- Source IP Address
263  -- Fill in the TCP/UDP checksum
264  v.tData(31 downto 0) := sAxisMaster.tData(127 downto 119) & sAxisMaster.tData(95 downto 88) & x"00";
265  v.tKeep(3 downto 0) := (others => '1');
266  end if;
267  -- Latch the IPv4 length value
268  v.ipv4Len(0)(15 downto 8) := v.ipv4Hdr(2);
269  v.ipv4Len(0)(7 downto 0) := v.ipv4Hdr(3);
270  -- Check for UDP protocol
271  if (v.ipv4Hdr(9) = UDP_C) then
272  v.udpDet(0) := '1';
273  end if;
274  -- Check for TCP protocol
275  if (v.ipv4Hdr(9) = TCP_C) then
276  v.tcpDet(0) := '1';
277  end if;
278  -- Check for fragmentation
279  if (v.ipv4Hdr(6)(7) = '1') or (v.ipv4Hdr(6)(4 downto 0) /= 0) or (v.ipv4Hdr(7) /= 0) then
280  -- Set the flags
281  v.fragDet(0) := '1';
282  v.fragSof := '1';
283  end if;
284  -- Next state
285  v.state := IPV4_HDR1_S;
286  end if;
287  end if;
288  ----------------------------------------------------------------------
289  when IPV4_HDR1_S =>
290  -- Check for valid data
291  if (sAxisMaster.tValid = '1') then
292  -- Move the data
293  v.mAxisMaster := sAxisMaster;
294  -- Fill in the TCP/UDP checksum
295  v.tKeep := sAxisMaster.tKeep;
296  v.tData := sAxisMaster.tData;
297  -- Check if NON-VLAN
298  if (VLAN_G = false) then
299  -- Fill in the IPv4 header checksum
300  v.ipv4Hdr(18) := sAxisMaster.tData(7 downto 0); -- Destination IP Address
301  v.ipv4Hdr(19) := sAxisMaster.tData(15 downto 8); -- Destination IP Address
302  -- Check for UDP data with inbound checksum
303  if (r.ipv4Det(0) = '1') and (r.udpDet(0) = '1') then
304  -- Mask off inbound UDP checksum
305  v.tData := sAxisMaster.tData(127 downto 80) & x"0000" & sAxisMaster.tData(63 downto 0);
306  -- Latch the inbound UDP checksum
307  v.protCsum(0)(15 downto 8) := sAxisMaster.tData(71 downto 64);
308  v.protCsum(0)(7 downto 0) := sAxisMaster.tData(79 downto 72);
309  -- Latch the inbound UDP length
310  v.protLen(0)(15 downto 8) := sAxisMaster.tData(55 downto 48);
311  v.protLen(0)(7 downto 0) := sAxisMaster.tData(63 downto 56);
312  end if;
313  -- Track the number of bytes (include IPv4 header offset from previous state)
314  v.byteCnt := getTKeep(sAxisMaster.tKeep) + 18;
315  else
316  -- Fill in the IPv4 header checksum
317  v.ipv4Hdr(14) := sAxisMaster.tData(7 downto 0); -- Source IP Address
318  v.ipv4Hdr(15) := sAxisMaster.tData(15 downto 8); -- Source IP Address
319  v.ipv4Hdr(16) := sAxisMaster.tData(23 downto 16); -- Destination IP Address
320  v.ipv4Hdr(17) := sAxisMaster.tData(31 downto 24); -- Destination IP Address
321  v.ipv4Hdr(18) := sAxisMaster.tData(39 downto 32); -- Destination IP Address
322  v.ipv4Hdr(19) := sAxisMaster.tData(47 downto 40); -- Destination IP Address
323  -- Check for UDP data with inbound checksum
324  if (r.ipv4Det(0) = '1') and (r.udpDet(0) = '1') then
325  -- Mask off inbound UDP checksum
326  v.tData := sAxisMaster.tData(127 downto 112) & x"0000" & sAxisMaster.tData(95 downto 0);
327  -- Latch the inbound UDP checksum
328  v.protCsum(0)(15 downto 8) := sAxisMaster.tData(103 downto 96);
329  v.protCsum(0)(7 downto 0) := sAxisMaster.tData(111 downto 104);
330  -- Latch the inbound UDP length
331  v.protLen(0)(15 downto 8) := sAxisMaster.tData(87 downto 80);
332  v.protLen(0)(7 downto 0) := sAxisMaster.tData(95 downto 88);
333  end if;
334  -- Track the number of bytes (include IPv4 header offset from previous state)
335  v.byteCnt := getTKeep(sAxisMaster.tKeep) + 14;
336  end if;
337  -- Check for EOF
338  if (sAxisMaster.tLast = '1') then
339  -- Next state
340  v.state := IDLE_S;
341  else
342  -- Next state
343  v.state := MOVE_S;
344  end if;
345  end if;
346  ----------------------------------------------------------------------
347  when MOVE_S =>
348  -- Check for valid data
349  if (sAxisMaster.tValid = '1') then
350  -- Move the data
351  v.mAxisMaster := sAxisMaster;
352  -- Fill in the TCP/UDP checksum
353  v.tData := sAxisMaster.tData;
354  v.tKeep := sAxisMaster.tKeep;
355  -- Check for TCP data with inbound checksum
356  if (r.ipv4Det(0) = '1') and (r.tcpDet(0) = '1') and (r.tcpFlag = '0') then
357  -- Set the flag
358  v.tcpFlag := '1';
359  -- Calculate TCP length from IPv4 length
360  v.protLen(0) := r.ipv4Len(0) - 20;
361  -- Check if NON-VLAN
362  if (VLAN_G = false) then
363  -- Mask off inbound TCP checksum
364  v.tData := sAxisMaster.tData(127 downto 32) & x"0000" & sAxisMaster.tData(15 downto 0);
365  -- Latch the inbound TCP checksum
366  v.protCsum(0)(15 downto 8) := sAxisMaster.tData(23 downto 16);
367  v.protCsum(0)(7 downto 0) := sAxisMaster.tData(31 downto 24);
368  else
369  -- Mask off inbound TCP checksum
370  v.tData := sAxisMaster.tData(127 downto 64) & x"0000" & sAxisMaster.tData(47 downto 0);
371  -- Latch the inbound TCP checksum
372  v.protCsum(0)(15 downto 8) := sAxisMaster.tData(55 downto 48);
373  v.protCsum(0)(7 downto 0) := sAxisMaster.tData(63 downto 56);
374  end if;
375  end if;
376  -- Track the number of bytes
377  v.byteCnt := r.byteCnt + getTKeep(sAxisMaster.tKeep);
378  -- Check for EOF
379  if (sAxisMaster.tLast = '1') or (v.byteCnt > MAX_FRAME_SIZE_C) then
380  -- Check for overflow condition
381  if (sAxisMaster.tLast = '0') then
382  -- Force EOF
383  v.mAxisMaster.tLast := '1';
384  -- Set the error flag
385  axiStreamSetUserBit(EMAC_AXIS_CONFIG_C, v.mAxisMaster, EMAC_EOFE_BIT_C, '1');
386  -- Next state
387  v.state := BLOWOFF_S;
388  else
389  -- Next state
390  v.state := IDLE_S;
391  end if;
392  end if;
393  end if;
394  ----------------------------------------------------------------------
395  when BLOWOFF_S =>
396  -- Check for a valid EOF
397  if (sAxisMaster.tValid = '1') and (sAxisMaster.tLast = '1') then
398  -- Next State
399  v.state := IDLE_S;
400  end if;
401  ----------------------------------------------------------------------
402  end case;
403 
404  -- Check for first TUSER on the output AXIS stream
405  if (axiStreamGetUserBit(EMAC_AXIS_CONFIG_C, v.mAxisMasters(EMAC_CSUM_PIPELINE_C+1), EMAC_SOF_BIT_C, 0) = '1') then
406  -- Set the fragmentation flag
407  axiStreamSetUserBit(EMAC_AXIS_CONFIG_C, v.mAxisMasters(EMAC_CSUM_PIPELINE_C+1), EMAC_FRAG_BIT_C, r.fragSof, 0);
408  -- Reset the flag
409  v.fragSof := '0';
410  end if;
411 
412  -- Reset
413  if (ethRst = '1') then
414  v := REG_INIT_C;
415  end if;
416 
417  -- Register the variable for next clock cycle
418  rin <= v;
419 
420  -- Outputs
421  mAxisMaster <= r.mAxisMasters(EMAC_CSUM_PIPELINE_C+1);
422  dbg <= dummy;
423 
424  end process comb;
425

429  r <= rin after TPD_G;
430  end if;
431  end process seq;

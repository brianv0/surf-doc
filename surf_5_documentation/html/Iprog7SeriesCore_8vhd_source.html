48  function selectMapBitSwapping (input : slv) return slv is
49  variable i : integer;
50  variable j : integer;
51  variable output : slv(0 to 31);
52  begin
53  for i in 0 to 3 loop
54  for j in 0 to 7 loop
55  output((8*i)+j) := input((8*i)+(7-j));
56  end loop;
57  end loop;
58  return output;
59  end function selectMapBitSwapping;

63  type RegType is record

112  comb : process (icapClkRst, icapGrant, icapReload, icapReloadAddr, r) is
113  variable v : RegType;
114  begin
115  v := r;
116 
117  case (r.state) is
118  when IDLE_S =>
119  v.csl := '1';
120  v.rnw := '1';
121  v.cnt := (others => '0');
122  if (icapReload = '1') then
123  if (icapGrant = '1') then
124  v.state := PROG_S;
125  else
126  v.state := REQ_S;
127  end if;
128  v.req := '1';
129  end if;
130 
131  when REQ_S =>
132  -- Wait to be granted access to the ICAP
133  if (icapGrant = '1') then
134  v.state := PROG_S;
135  end if;
136 
137  when PROG_S =>
138  v.csl := '0';
139  v.rnw := '0';
140  v.cnt := r.cnt + 1;
141  case (r.cnt) is
142  when X"0" =>
143  --Sync Word
144  v.configData := selectMapBitSwapping(X"AA995566");
145  when X"1" =>
146  --Type 1 NO OP
147  v.configData := selectMapBitSwapping(X"20000000");
148  when X"2" =>
149  --Type 1 Write 1 Words to WBSTAR
150  v.configData := selectMapBitSwapping(X"30020001");
151  when X"3" =>
152  --Warm Boot Start Address (Load the Desired Address)
153  v.configData := selectMapBitSwapping(bitReverse(icapReloadAddr));
154  when X"4" =>
155  --Type 1 Write 1 Words to CMD
156  v.configData := selectMapBitSwapping(X"30008001");
157  when X"5" =>
158  --IPROG Command
159  v.configData := selectMapBitSwapping(X"0000000F");
160  when X"6" =>
161  --Type 1 NO OP
162  v.configData := selectMapBitSwapping(X"20000000");
163  v.state := IDLE_S;
164  when others => null;
165  end case;
166 
167  when others => null;
168  end case;
169 
170  if (icapClkRst = '1') then
171  v := REG_INIT_C;
172  end if;
173 
174  rin <= v;
175 
176  icapRnw <= r.rnw;
177  icapCsl <= r.csl;
178  icapI <= r.configData;
179  icapReq <= r.req;
180 
181  end process comb;
182 
183  seq : process (icapClk) is
184  begin
185  if (rising_edge(icapClk)) then
186  r <= rin after TPD_G;
187  end if;
188  end process seq;
189 
190 end architecture rtl;

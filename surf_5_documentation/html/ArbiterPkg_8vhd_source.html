39  -- Priority encoder with index p having top priority
40  -- followed by p-1, p-2, etc., rolling over to the highest index when 0 reached
41  -- Returns highest priority index with value '1', encoded as unsigned slv.
42  function priorityEncode (v : slv; p : integer) return slv is
43  variable bestReq : integer;
44  variable rotatedV : unsigned(v'range);
45  variable ret : unsigned(bitSize(v'length-1)-1 downto 0) := (others => '0');
46  variable bestReqU : unsigned(ret'length downto 0);
47  begin
48 -- print("priorityEncode(" & str(v) & ", " & str(p) & ")");
49  -- Rotate input by n to give n top priority
50  rotatedV := rotate_right(unsigned(v), p);
51 
52  -- Find lowest index with value of '1'
53  bestReq := 0;
54  for i in v'range loop
55  if (rotatedV(i) = '1') then
56  bestReq := i;
57  end if;
58  end loop;
59 
60  -- Convert integer to unsigned
61  bestReqU := to_unsigned(bestReq, bestReqU'length);
62 
63  -- Add p to rotated select and mod by length to undo the rotation
64  ret := resize((bestReqU+p) mod v'length, ret'length);
65 
66  return slv(ret);
67  end function priorityEncode;

70  procedure arbitrate (
71  req : in slv;
72  lastSelected : in slv;
73  nextSelected : inout slv;
74  valid : inout sl;
75  ack : out slv) is
76  begin
77 -- print("arbitrate(" & str(req'length));-- & ", " & str(lastSelected'length) & ", " str(nextSelected'length) & ", " & ")");
78  valid := uOr(req);
79  if (valid = '1') then
80  nextSelected := priorityEncode(req, to_integer(unsigned(lastSelected)+1) mod req'length);
81  ack := decode(slv(nextSelected))(req'range);
82  else
83  nextSelected := lastSelected;
84  ack(ack'range) := (others => '0');
85  end if;
86  end procedure arbitrate;

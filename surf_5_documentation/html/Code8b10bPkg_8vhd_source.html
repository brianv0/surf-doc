64  procedure encode8b10b (
65  dataIn : in slv(7 downto 0);
66  dataKIn : in sl;
67  dispIn : in sl;
68  dataOut : out slv(9 downto 0);
69  dispOut : out sl)
70  is
71  variable ai, bi, ci, di, ei, fi, gi, hi, ki : sl;
72  variable aeqb, ceqd : sl;
73  variable l22, l40, l04, l13, l31 : sl;
74  variable ao, bo, co, do, eo, io, fo, go, ho, jo : sl;
75  variable pd1s6, nd1s6, ndos6, pdos6 : sl;
76  variable alt6, alt7 : sl;
77  variable nd1s4, pd1s4, ndos4, pdos4 : sl;
78  variable illegalk : sl;
79  variable compls6, disp6, compls4 : sl;
80  begin
81 
82  ai := dataIn(0);
83  bi := dataIn(1);
84  ci := dataIn(2);
85  di := dataIn(3);
86  ei := dataIn(4);
87  fi := dataIn(5);
88  gi := dataIn(6);
89  hi := dataIn(7);
90  ki := dataKIn;
91 
92  aeqb := (ai and bi) or (not ai and not bi);
93  ceqd := (ci and di) or (not ci and not di);
94  l22 := (ai and bi and not ci and not di) or
95  (ci and di and not ai and not bi) or
96  (not aeqb and not ceqd);
97  l40 := ai and bi and ci and di;
98  l04 := not ai and not bi and not ci and not di;
99  l13 := (not aeqb and not ci and not di) or
100  (not ceqd and not ai and not bi);
101  l31 := (not aeqb and ci and di) or
102  (not ceqd and ai and bi);
103 
104  -- The 5B/6B encoding
105 
106  ao := ai;
107  bo := (bi and not l40) or l04;
108  co := l04 or ci or (ei and di and not ci and not bi and not ai);
109  do := di and not (ai and bi and ci);
110  eo := (ei or l13) and not (ei and di and not ci and not bi and not ai);
111  io := (l22 and not ei) or
112  (ei and not di and not ci and not (ai and bi)) or -- D16, D17, D18
113  (ei and l40) or
114  (ki and ei and di and ci and not bi and not ai) or -- K.28
115  (ei and not di and ci and not bi and not ai);
116 
117  -- pds16 indicates cases where d-1 is assumed + to get our encoded value
118  pd1s6 := (ei and di and not ci and not bi and not ai) or (not ei and not l22 and not l31);
119  -- nds16 indicates cases where d-1 is assumed - to get our encoded value
120  nd1s6 := ki or (ei and not l22 and not l13) or (not ei and not di and ci and bi and ai);
121 
122  -- ndos6 is pds16 cases where d-1 is + yields - disp out - all of them
123  ndos6 := pd1s6;
124  -- pdos6 is nds16 cases where d-1 is - yields + disp out - all but one
125  pdos6 := ki or (ei and not l22 and not l13);
126 
127 
128  -- some Dx.7 and all Kx.7 cases result in run length of 5 case unless
129  -- an alternate coding is used (referred to as Dx.A7, normal is Dx.P7)
130  -- specifically, D11, D13, D14, D17, D18, D19.
131  if (dispIn = '1') then
132  alt6 := (not ei and di and l31);
133  else
134  alt6 := (ei and not di and l13);
135  end if;
136  alt7 := fi and gi and hi and (ki or alt6);
137 
138  fo := fi and not alt7;
139  go := gi or (not fi and not gi and not hi);
140  ho := hi;
141  jo := (not hi and (gi xor fi)) or alt7;
142 
143  -- nd1s4 is cases where d-1 is assumed - to get our encoded value
144  nd1s4 := fi and gi;
145  -- pd1s4 is cases where d-1 is assumed + to get our encoded value
146  pd1s4 := (not fi and not gi) or (ki and ((fi and not gi) or (not fi and gi)));
147 
148  -- ndos4 is pd1s4 cases where d-1 is + yields - disp out - just some
149  ndos4 := (not fi and not gi);
150  -- pdos4 is nd1s4 cases where d-1 is - yields + disp out
151  pdos4 := fi and gi and hi;
152 
153  -- only legal K codes are K28.0- > .7, K23/27/29/30.7
154  -- K28.0- > 7 is ei = di = ci = 1, bi = ai = 0
155  -- K23 is 10111
156  -- K27 is 11011
157  -- K29 is 11101
158  -- K30 is 11110 - so K23/27/29/30 are ei and l31
159  illegalk := ki and
160  (ai or bi or not ci or not di or not ei) and -- not K28.0- > 7
161  (not fi or not gi or not hi or not ei or not l31); -- not K23/27/29/30.7
162 
163  -- now determine whether to do the complementing
164  -- complement if prev disp is - and pd1s6 is set, or + and nd1s6 is set
165  compls6 := (pd1s6 and not dispin) or (nd1s6 and dispin);
166 
167  -- disparity out of 5b6b is disp in with pdso6 and ndso6
168  -- pds16 indicates cases where d-1 is assumed + to get our encoded value
169  -- ndos6 is cases where d-1 is + yields - disp out
170  -- nds16 indicates cases where d-1 is assumed - to get our encoded value
171  -- pdos6 is cases where d-1 is - yields + disp out
172  -- disp toggles in all ndis16 cases, and all but that 1 nds16 case
173 
174  disp6 := dispin xor (ndos6 or pdos6);
175 
176  compls4 := (pd1s4 and not disp6) or (nd1s4 and disp6);
177  dispOut := disp6 xor (ndos4 or pdos4);
178 
179  dataOut := (jo xor compls4) & (ho xor compls4) &
180  (go xor compls4) & (fo xor compls4) &
181  (io xor compls6) & (eo xor compls6) &
182  (do xor compls6) & (co xor compls6) &
183  (bo xor compls6) & (ao xor compls6);
184 
185  end procedure encode8b10b;

201  variable p22aceeqi, p22ancneeqi : sl;
202  variable p13en, anbnenin, abei, cdei, cndnenin : sl;
203  variable p22enin, p22ei, p31dnenin, p31e : sl;
204  variable compa, compb, compc, compd, compe : sl;
205  variable ao, bo, co, do, eo, ko, fo, go, ho : sl;
206  variable feqg, heqj, fghj22, fghjp13, fghjp31 : sl;
207  variable alt7, k28, k28p : sl;
208  variable disp6p, disp6n, disp4p, disp4n : sl;

210 
211  ai := dataIn(0);
212  bi := dataIn(1);

214  di := dataIn(3);
215  ei := dataIn(4);
216  ii := dataIn(5);
217  fi := dataIn(6);
218  gi := dataIn(7);
219  hi := dataIn(8);
220  ji := dataIn(9);

223  ceqd := (ci and di) or (not ci and not di);
224  p22 := (ai and bi and not ci and not di) or
225  (ci and di and not ai and not bi) or
226  (not aeqb and not ceqd);
227  p13 := (not aeqb and not ci and not di) or
228  (not ceqd and not ai and not bi);

231

233  p04 := not ai and not bi and not ci and not di;
234 
235  disp6a := p31 or (p22 and dispin); -- pos disp if p22 and was pos, or p31.
236  disp6a2 := p31 and dispin; -- disp is ++ after 4 bits
237  disp6a0 := p13 and not dispin; -- -- disp after 4 bits
238 
239  disp6b := (((ei and ii and not disp6a0) or (disp6a and (ei or ii)) or disp6a2 or
240  (ei and ii and di)) and (ei or ii or di)) ;
241 
242  -- The 5B/6B decoding special cases where ABCDE not = abcde
243 
244  p22bceeqi := p22 and bi and ci and (ei xnor ii);
245  p22bncneeqi := p22 and not bi and not ci and (ei xnor ii);
246  p13in := p13 and not ii;
247  p31i := p31 and ii;
248  p13dei := p13 and di and ei and ii;
249  p22aceeqi := p22 and ai and ci and (ei xnor ii);
250  p22ancneeqi := p22 and not ai and not ci and (ei xnor ii);
251  p13en := p13 and not ei;
252  anbnenin := not ai and not bi and not ei and not ii;
253  abei := ai and bi and ei and ii;
254  cdei := ci and di and ei and ii;
255  cndnenin := not ci and not di and not ei and not ii;
256 
257  -- non-zero disparity cases:
258  p22enin := p22 and not ei and not ii;
259  p22ei := p22 and ei and ii;
260  -- p13in := p12 and not ii ;
261  -- p31i := p31 and ii ;
262  p31dnenin := p31 and not di and not ei and not ii;
263  -- p13dei := p13 and di and ei and ii ;
264  p31e := p31 and ei;
265 
266  compa := p22bncneeqi or p31i or p13dei or p22ancneeqi or
267  p13en or abei or cndnenin;
268  compb := p22bceeqi or p31i or p13dei or p22aceeqi or
269  p13en or abei or cndnenin;
270  compc := p22bceeqi or p31i or p13dei or p22ancneeqi or
271  p13en or anbnenin or cndnenin;
272  compd := p22bncneeqi or p31i or p13dei or p22aceeqi or
273  p13en or abei or cndnenin;
274  compe := p22bncneeqi or p13in or p13dei or p22ancneeqi or
275  p13en or anbnenin or cndnenin;
276 
277  ao := ai xor compa;
278  bo := bi xor compb;
279  co := ci xor compc;
280  do := di xor compd;
281  eo := ei xor compe;
282 
283  feqg := (fi and gi) or (not fi and not gi);
284  heqj := (hi and ji) or (not hi and not ji);
285  fghj22 := (fi and gi and not hi and not ji) or
286  (not fi and not gi and hi and ji) or
287  (not feqg and not heqj);
288  fghjp13 := (not feqg and not hi and not ji) or
289  (not heqj and not fi and not gi);
290  fghjp31 := ((not feqg) and hi and ji) or
291  (not heqj and fi and gi);
292 
293  dispout := (fghjp31 or (disp6b and fghj22) or (hi and ji)) and (hi or ji);
294 
295  ko := ((ci and di and ei and ii) or (not ci and not di and not ei and not ii) or
296  (p13 and not ei and ii and gi and hi and ji) or
297  (p31 and ei and not ii and not gi and not hi and not ji)) ;
298 
299  alt7 := (fi and not gi and not hi and -- 1000 cases, where disp6b is 1
300  ((dispin and ci and di and not ei and not ii) or ko or
301  (dispin and not ci and di and not ei and not ii))) or
302  (not fi and gi and hi and -- 0111 cases, where disp6b is 0
303  ((not dispin and not ci and not di and ei and ii) or ko or
304  (not dispin and ci and not di and ei and ii))) ;
305 
306  k28 := (ci and di and ei and ii) or not (ci or di or ei or ii);
307  -- k28 with positive disp into fghi - .1, .2, .5, and .6 special cases
308  k28p := not (ci or di or ei or ii);
309  fo := (ji and not fi and (hi or not gi or k28p)) or
310  (fi and not ji and (not hi or gi or not k28p)) or
311  (k28p and gi and hi) or
312  (not k28p and not gi and not hi);
313  go := (ji and not fi and (hi or not gi or not k28p)) or
314  (fi and not ji and (not hi or gi or k28p)) or
315  (not k28p and gi and hi) or
316  (k28p and not gi and not hi);
317  ho := ((ji xor hi) and not ((not fi and gi and not hi and ji and not k28p) or (not fi and gi and hi and not ji and k28p) or
318  (fi and not gi and not hi and ji and not k28p) or (fi and not gi and hi and not ji and k28p))) or
319  (not fi and gi and hi and ji) or (fi and not gi and not hi and not ji);
320 
321  disp6p := (p31 and (ei or ii)) or (p22 and ei and ii);
322  disp6n := (p13 and not (ei and ii)) or (p22 and not ei and not ii);
323  disp4p := fghjp31;
324  disp4n := fghjp13;
325 
326  codeErr := p40 or p04 or (fi and gi and hi and ji) or (not fi and not gi and not hi and not ji) or
327  (p13 and not ei and not ii) or (p31 and ei and ii) or
328  (ei and ii and fi and gi and hi) or (not ei and not ii and not fi and not gi and not hi) or
329  (ei and not ii and gi and hi and ji) or (not ei and ii and not gi and not hi and not ji) or
330  (not p31 and ei and not ii and not gi and not hi and not ji) or
331  (not p13 and not ei and ii and gi and hi and ji) or
332  (((ei and ii and not gi and not hi and not ji) or
333  (not ei and not ii and gi and hi and ji)) and
334  not ((ci and di and ei) or (not ci and not di and not ei))) or
335  (disp6p and disp4p) or (disp6n and disp4n) or
336  (ai and bi and ci and not ei and not ii and ((not fi and not gi) or fghjp13)) or
337  (not ai and not bi and not ci and ei and ii and ((fi and gi) or fghjp31)) or
338  (fi and gi and not hi and not ji and disp6p) or
339  (not fi and not gi and hi and ji and disp6n) or
340  (ci and di and ei and ii and not fi and not gi and not hi) or
341  (not ci and not di and not ei and not ii and fi and gi and hi);
342 
343  dataOut := ho & go & fo & eo & do & co & bo & ao;

350  (dispin and not disp6n and disp4p) or
351  (not dispin and not disp6p and not fi and not gi) or
352  (not dispin and not ai and not bi and not ci) or
353  (not dispin and not disp6p and disp4n) or
354  (disp6p and disp4p) or (disp6n and disp4n)) ;

356  end procedure decode8b10b;

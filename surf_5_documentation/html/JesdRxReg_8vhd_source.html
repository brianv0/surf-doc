154  comb : process (axiRst_i, axilReadMaster, axilWriteMaster, r, s_RdAddr,
155  s_WrAddr, s_statusRxArr, s_statusCnt, s_rawData) is
156  variable v : RegType;
157  variable axilStatus : AxiLiteStatusType;
158  variable axilWriteResp : slv(1 downto 0);
159  variable axilReadResp : slv(1 downto 0);
160  begin
161  -- Latch the current value
162  v := r;
163 
164  ----------------------------------------------------------------------------------------------
165  -- Axi-Lite interface
166  ----------------------------------------------------------------------------------------------
167  axiSlaveWaitTxn(axilWriteMaster, axilReadMaster, v.axilWriteSlave, v.axilReadSlave, axilStatus);
168 
169  if (axilStatus.writeEnable = '1') then
170  axilWriteResp := ite(axilWriteMaster.awaddr(1 downto 0) = "00", AXI_RESP_OK_C, AXI_ERROR_RESP_G);
171  case (s_WrAddr) is
172  when 16#00# => -- ADDR (0x00)
173  v.enableRx := axilWriteMaster.wdata(L_G-1 downto 0);
174  when 16#01# => -- ADDR (0x04)
175  v.sysrefDlyRx := axilWriteMaster.wdata(SYSRF_DLY_WIDTH_C-1 downto 0);
176  when 16#02# => -- ADDR (0x08)
177  v.rxPolarity := axilWriteMaster.wdata(L_G-1 downto 0);
178  when 16#04# => -- ADDR (0x10)
179  v.commonCtrl := axilWriteMaster.wdata(5 downto 0);
180  when 16#05# => -- ADDR (0x14)
181  v.linkErrMask := axilWriteMaster.wdata(5 downto 0);
182  when 16#06# => -- ADDR (0x18)
183  v.invertData := axilWriteMaster.wdata(L_G-1 downto 0);
184  when 16#20# to 16#2F# =>
185  for I in (L_G-1) downto 0 loop
186  if (axilWriteMaster.awaddr(5 downto 2) = I) then
187  v.testTXItf(I) := axilWriteMaster.wdata(15 downto 0);
188  end if;
189  end loop;
190  when 16#30# to 16#3F# =>
191  for I in (L_G-1) downto 0 loop
192  if (axilWriteMaster.awaddr(5 downto 2) = I) then
193  v.testSigThr(I) := axilWriteMaster.wdata(31 downto 0);
194  end if;
195  end loop;
196  when others =>
197  axilWriteResp := AXI_ERROR_RESP_G;
198  end case;
199  axiSlaveWriteResponse(v.axilWriteSlave);
200  end if;
201 
202  if (axilStatus.readEnable = '1') then
203  axilReadResp := ite(axilReadMaster.araddr(1 downto 0) = "00", AXI_RESP_OK_C, AXI_ERROR_RESP_G);
204  v.axilReadSlave.rdata := (others => '0');
205  case (s_RdAddr) is
206  when 16#00# => -- ADDR (0x0)
207  v.axilReadSlave.rdata(L_G-1 downto 0) := r.enableRx;
208  when 16#01# => -- ADDR (0x04)
209  v.axilReadSlave.rdata(SYSRF_DLY_WIDTH_C-1 downto 0) := r.sysrefDlyRx;
210  when 16#02# => -- ADDR (0x08)
211  v.axilReadSlave.rdata(L_G-1 downto 0) := r.rxPolarity;
212  when 16#04# => -- ADDR (0x10)
213  v.axilReadSlave.rdata(5 downto 0) := r.commonCtrl;
214  when 16#05# => -- ADDR (0x14)
215  v.axilReadSlave.rdata(5 downto 0) := r.linkErrMask;
216  when 16#06# => -- ADDR (0x18)
217  v.axilReadSlave.rdata(L_G-1 downto 0) := r.invertData;
218  when 16#10# to 16#1F# =>
219  for I in (L_G-1) downto 0 loop
220  if (axilReadMaster.araddr(5 downto 2) = I) then
221  v.axilReadSlave.rdata(RX_STAT_WIDTH_C-1 downto 0) := s_statusRxArr(I);
222  end if;
223  end loop;
224  when 16#20# to 16#2F# =>
225  for I in (L_G-1) downto 0 loop
226  if (axilReadMaster.araddr(5 downto 2) = I) then
227  v.axilReadSlave.rdata(15 downto 0) := r.testTXItf(I);
228  end if;
229  end loop;
230  when 16#30# to 16#3F# =>
231  for I in (L_G-1) downto 0 loop
232  if (axilReadMaster.araddr(5 downto 2) = I) then
233  v.axilReadSlave.rdata(31 downto 0) := r.testSigThr(I);
234  end if;
235  end loop;
236  when 16#40# to 16#4F# =>
237  for I in (L_G-1) downto 0 loop
238  if (axilReadMaster.araddr(5 downto 2) = I) then
239  for J in 31 downto 0 loop
240  v.axilReadSlave.rdata(J) := s_statusCnt(I, J);
241  end loop;
242  end if;
243  end loop;
244  when 16#50# to 16#5F# =>
245  for I in (L_G-1) downto 0 loop
246  if (axilReadMaster.araddr(5 downto 2) = I) then
247  v.axilReadSlave.rdata := s_rawData(I);
248  end if;
249  end loop;
250  when others =>
251  axilReadResp := AXI_ERROR_RESP_G;
252  end case;
253  axiSlaveReadResponse(v.axilReadSlave);
254  end if;
255 
256  -- Reset
257  if (axiRst_i = '1') then
258  v := REG_INIT_C;
259  end if;
260 
261  -- Register the variable for next clock cycle
262  rin <= v;
263 
264  -- Outputs
265  axilReadSlave <= r.axilReadSlave;
266  axilWriteSlave <= r.axilWriteSlave;
267  rxPolarity <= r.rxPolarity;
268 
269  end process comb;

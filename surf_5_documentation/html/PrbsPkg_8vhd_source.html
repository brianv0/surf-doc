40  function getPrbs1xTap (input : slv; tap0 : natural) return slv is
41  variable retVar : slv(input'left downto 0) := (others => '0');
42  begin
43 
44  --check for a valid tap location
45  assert (tap0 <= input'left) report "PrbsPkg: getPrbs1xTap's tap0 input is out of range" severity failure;
46 
47  -- shift register
48  for i in (input'left - 1) downto 0 loop
49  retVar(i) := input(i+1);
50  end loop;
51 
52  -- calculate the "xor'd" feedback
53  retVar(input'left) := input(0) xor input(tap0);
54 
55  --return the result
56  return retVar;
57 
58  end function;

135 
136  -- shift register
137  for i in 0 to (input'left-1) loop
138  retVar(i) := input(i+1);
139  end loop;
140  retVar(retVar'left) := '0';
141 
142  for i in 0 to (input'left) loop
143  if (i = tap0 or i = tap1 or i = tap2 or i = tap3) then
144  retVar(i) := retVar(i) xor input(0);
145  end if;
146  end loop;

154  variable retVar : slv(input'left downto 0) := (others => '0');

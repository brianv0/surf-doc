64  comb : process (axisMaster, axisRst, axisSlave, r) is
65  variable v : RegType;
66  variable i : natural;
67  begin
68  -- Latch the current value
69  v := r;
70 
71  -- Check if ready to move data
72  if (axisMaster.tValid = '1') and (axisSlave.tReady = '1') then
73  -- State Machine
74  case r.state is
75  ----------------------------------------------------------------------
76  when IDLE_S =>
77  -- Check for SOF
78  if ssiGetUserSof(AXI_CONFIG_G, axisMaster) = '1' then
79  -- Move the data
80  v.copy := axisMaster;
81  -- Reset the counter
82  v.cnt := (others => '0');
83  -- Check for non-EOF
84  if axisMaster.tLast = '0' then
85  -- Next state
86  v.state := MOVE_S;
87  end if;
88  end if;
89  ----------------------------------------------------------------------
90  when MOVE_S =>
91  -- Move the data
92  v.copy := axisMaster;
93  -- Increment the counter
94  v.cnt := r.cnt + 1;
95  if axisMaster.tLast = '1' then
96  -- Next state
97  v.state := IDLE_S;
98  end if;
99  ----------------------------------------------------------------------
100  end case;
101  end if;
102 
103  -- Reset
104  if (axisRst = '1') then
105  v := REG_INIT_C;
106  end if;
107 
108  -- Register the variable for next clock cycle
109  rin <= v;
110 
111  end process comb;
112 
113  seq : process (axisClk) is

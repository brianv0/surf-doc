103  inEn <= AdcSpiIn.RnW after TPD_G;
104  sdi <= AdcSpiIn.din(conv_integer(23-pntr)) after TPD_G;
105  elsif pntr > 3 then
106  sdi <= AdcSpiIn.addr(conv_integer(15-pntr)) after TPD_G;
107  elsif pntr = 0 then
108  sdi <= AdcSpiIn.RnW after TPD_G;
109  else
110  sdi <= '0' after TPD_G;
111  end if;
112  cnt <= cnt + 1 after TPD_G;
113  -- Min. 20 ns wait
114  if cnt = MAX_CNT_C then
115  cnt <= 0 after TPD_G;
116  if pntr > 15 then
117  r.dout(conv_integer(23-pntr)) <= sdo after TPD_G;
118  end if;
119  state <= SCK_HIGH_S after TPD_G;
120  end if;
121  ----------------------------------------------------------------------
122  when SCK_HIGH_S =>
123  sck <= '1' after TPD_G;
124  cnt <= cnt + 1 after TPD_G;
125  -- Min. 20 ns wait
126  if cnt = MAX_CNT_C then
127  cnt <= 0 after TPD_G;
128  pntr <= pntr + 1 after TPD_G;
129  if pntr = 23 then
130  cs <= '1' after TPD_G;
131  pntr <= (others => '0') after TPD_G;
132  inEn <= '0' after TPD_G;
133  r.ack <= '1' after TPD_G;
134  state <= HANDSHAKE_S after TPD_G;
135  else
136  state <= SCK_LOW_S after TPD_G;
137  end if;
138  end if;
139  ----------------------------------------------------------------------
140  when HANDSHAKE_S =>
141  r.ack <= '1' after TPD_G;
142  if AdcSpiIn.req = '0' then
143  r.ack <= '0' after TPD_G;
144  sck <= '0' after TPD_G;
145  state <= IDLE_S after TPD_G;
146  end if;
147  ----------------------------------------------------------------------
148  end case;
149  end if;
150  end if;
151  end process;
152 
153 end rtl;

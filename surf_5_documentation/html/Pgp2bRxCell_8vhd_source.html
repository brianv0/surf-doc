374  vc1Serial <= (others=>'0') after TPD_G;

405  if compData(13 downto 8) = vc1Serial then
406  serErr <= '0' after TPD_G;

420  end if;

424  when others =>
425  if compData(13 downto 8) = vc3Serial then
426  serErr <= '0' after TPD_G;
427  else
428  vc3Serial <= compData(13 downto 8) after TPD_G;

441  end case;
442  end if;
443  end if;
444  end if;
445  end if;
446  end process;

452  if pgpRxClkRst = '1' then
453  crcNotZero <= '0' after TPD_G;
454  linkDownCnt <= (others=>'0') after TPD_G;
455  inCellEn <= '0' after TPD_G;
456  inCellSerErr <= '0' after TPD_G;
457  inCellSOF <= '0' after TPD_G;
458  inCellEOC <= '0' after TPD_G;
459  inCellEOF <= '0' after TPD_G;

463  abortVc <= (others=>'0') after TPD_G;

472 
473  -- CRC Error
474  if crcRxOut = 0 then
475  crcNotZero <= '0' after TPD_G;
476  else
477  crcNotZero <= '1' after TPD_G;
478  end if;
479 
480  -- Link down counter
481  if pgpRxLinkReady = '1' then
482  linkDownCnt <= (others=>'0') after TPD_G;
483  elsif linkDownCnt(4) = '0' then
484  linkDownCnt <= linkDownCnt + 1 after TPD_G;
485  end if;
486 
487  -- Count size of each cell received
488  if cellRxPause = '0' then
489  if inCellEn = '1' then
490  inCellCnt <= inCellCnt - 1 after TPD_G;
491  else
492  inCellCnt <= (others=>'1') after TPD_G;
493  end if;
494  end if;
495 
496  -- Link is down. Terminate transmission for any active VCs
497  if pgpRxLinkReady = '0' then
498 
499  -- Enabled every 4 clocks to ensure proper spacing between generated EOFs
500  if linkDownCnt(1 downto 0) = "11" then
501 
502  -- VC is active
503  if vcInFrame(conv_integer(linkDownCnt(3 downto 2))) = '1' then
504  abortEn <= '1' after TPD_G;
505  vcInFrame(conv_integer(linkDownCnt(3 downto 2))) <= '0' after TPD_G;
506  else
507  abortEn <= '0' after TPD_G;
508  end if;
509  else
510  abortEn <= '0' after TPD_G;
511  end if;
512 
513  -- VC for abort
514  abortVc <= linkDownCnt(3 downto 2) after TPD_G;
515 
516  -- Clear cell control signals
517  inCellEn <= '0' after TPD_G;
518  inCellSerErr <= '0' after TPD_G;
519  inCellSOF <= '0' after TPD_G;
520  inCellEOC <= '0' after TPD_G;
521  inCellEOF <= '0' after TPD_G;
522  inCellEOFE <= '0' after TPD_G;
523  intCellError <= '0' after TPD_G;
524 
525  -- Link is ready
526  else
527 
528  -- Clear abort flags
529  abortVc <= (others=>'0') after TPD_G;
530  abortEn <= '0' after TPD_G;
531 
532  -- Link flush set
533  if pgpRxFlush = '1' then
534  vcInFrame <= (others=>'0') after TPD_G;
535 
536  -- Pipeline enable
537  elsif cellRxPause = '0' then
538 
539  -- SOC Received
540  if detSOC = '1' then
541 
542  -- Do we output data and mark in frame?
543  -- Yes if SOF is set
544  -- Yes if already in frame
545  if nxtCellEn = '1' then
546  inCellEn <= '1' after TPD_G;
547  vcInFrame(conv_integer(currVc)) <= '1' after TPD_G;
548  end if;
549 
550  -- Do we mark output as SOF?
551  -- Yes if SOF is seen and we are not already in frame
552  if detSOF = '1' and vcInFrame(conv_integer(currVc)) = '0' then
553  inCellSOF <= '1' after TPD_G;
554  end if;
555 
556  -- Do we mark serial error flag?
557  -- Yes if SOF is set and we are already in frame
558  -- Yes if serial number error and we are in frame
559  if vcInFrame(conv_integer(currVc)) = '1' and (detSOF = '1' or serErr = '1') then
560  inCellSerErr <= '1' after TPD_G;
561  end if;
562 
563  -- Mark out of cell after EOC
564  elsif inCellEOC = '1' then
565  inCellEn <= '0' after TPD_G;
566  inCellSerErr <= '0' after TPD_G;
567  inCellSOF <= '0' after TPD_G;
568 
569  -- Clear frame state if EOF
570  if inCellEOF = '1' then
571  vcInFrame(conv_integer(currVc)) <= '0' after TPD_G;
572  end if;
573 
574  -- Clear SOF
575  else
576  inCellSOF <= '0' after TPD_G;
577  end if;
578 
579  -- End of cell, check for short cell case
580  if detEOC = '1' and (inCellEn = '1' or nxtCellEn = '1') then
581  inCellEOC <= '1' after TPD_G;
582  intCellError <= inCellSerErr or crcNotZero after TPD_G;
583 
584  -- Cell is too short
585  if detEOF = '0' and inCellCnt /= 1 and EN_SHORT_CELLS_G = 0 then
586  inCellEOF <= '1' after TPD_G;
587  inCellEOFE <= '1' after TPD_G;
588  intCellError <= '1' after TPD_G;
589  else
590  inCellEOF <= detEOF or inCellSerErr or crcNotZero after TPD_G;
591  inCellEOFE <= detEOFE or inCellSerErr or crcNotZero after TPD_G;
592  intCellError <= inCellSerErr or crcNotZero after TPD_G;
593  end if;
594 
595  -- Cell might be too long
596  elsif inCellEn = '1' and inCellCnt = 0 and inCellEOC = '0' then
597  inCellEOC <= '1' after TPD_G;
598  inCellEOF <= '1' after TPD_G;
599  inCellEOFE <= '1' after TPD_G;
600  intCellError <= '1' after TPD_G;
601  else
602  inCellEOC <= '0' after TPD_G;
603  inCellEOF <= '0' after TPD_G;
604  inCellEOFE <= '0' after TPD_G;
605  intCellError <= '0' after TPD_G;
606  end if;
607  end if;
608  end if;
609  end if;
610  end if;
611  end process;
612 
613 
614  -- Do we output data and mark in frame?
615  -- Yes if SOF is set
616  -- Yes if already in frame
617  nxtCellEn <= '1' when (detSOF = '1' or vcInFrame(conv_integer(currVc)) = '1') else '0';
618 
619 
620  -- Data Output
621  process ( pgpRxClk ) begin
622  if rising_edge(pgpRxClk) then
623  if pgpRxClkRst = '1' then
624  vcFrameRxData <= (others=>'0') after TPD_G;
625  vcFrameRxSOF <= '0' after TPD_G;
626  vcFrameRxEOF <= '0' after TPD_G;
627  vcFrameRxEOFE <= '0' after TPD_G;
628  vc0FrameRxValid <= '0' after TPD_G;
629  vc1FrameRxValid <= '0' after TPD_G;
630  vc2FrameRxValid <= '0' after TPD_G;
631  vc3FrameRxValid <= '0' after TPD_G;
632  elsif pgpRxClkEn = '1' then
633  -- Data abort is enabled
634  if abortEn = '1' then
635  case abortVc is
636  when "00" =>
637  vc0FrameRxValid <= '1' after TPD_G;
638  vc1FrameRxValid <= '0' after TPD_G;
639  vc2FrameRxValid <= '0' after TPD_G;
640  vc3FrameRxValid <= '0' after TPD_G;
641  when "01" =>
642  vc0FrameRxValid <= '0' after TPD_G;
643  vc1FrameRxValid <= '1' after TPD_G;
644  vc2FrameRxValid <= '0' after TPD_G;
645  vc3FrameRxValid <= '0' after TPD_G;
646  when "10" =>
647  vc0FrameRxValid <= '0' after TPD_G;
648  vc1FrameRxValid <= '0' after TPD_G;
649  vc2FrameRxValid <= '1' after TPD_G;
650  vc3FrameRxValid <= '0' after TPD_G;
651  when others =>
652  vc0FrameRxValid <= '0' after TPD_G;
653  vc1FrameRxValid <= '0' after TPD_G;
654  vc2FrameRxValid <= '0' after TPD_G;
655  vc3FrameRxValid <= '1' after TPD_G;
656  end case;
657 
658  -- Abort output
659  vcFrameRxSOF <= '0' after TPD_G;
660  vcFrameRxEOF <= '1' after TPD_G;
661  vcFrameRxEOFE <= '1' after TPD_G;
662 
663  -- Pipeline is enabled
664  elsif cellRxPause = '0' and inCellEn = '1' then
665  case currVc is
666  when "00" =>
667  vc0FrameRxValid <= '1' after TPD_G;
668  vc1FrameRxValid <= '0' after TPD_G;
669  vc2FrameRxValid <= '0' after TPD_G;
670  vc3FrameRxValid <= '0' after TPD_G;
671  when "01" =>
672  vc0FrameRxValid <= '0' after TPD_G;
673  vc1FrameRxValid <= '1' after TPD_G;
674  vc2FrameRxValid <= '0' after TPD_G;
675  vc3FrameRxValid <= '0' after TPD_G;
676  when "10" =>
677  vc0FrameRxValid <= '0' after TPD_G;
678  vc1FrameRxValid <= '0' after TPD_G;
679  vc2FrameRxValid <= '1' after TPD_G;
680  vc3FrameRxValid <= '0' after TPD_G;
681  when others =>
682  vc0FrameRxValid <= '0' after TPD_G;
683  vc1FrameRxValid <= '0' after TPD_G;
684  vc2FrameRxValid <= '0' after TPD_G;
685  vc3FrameRxValid <= '1' after TPD_G;
686  end case;
687 
688  -- Data output
689  vcFrameRxData <= outData after TPD_G;
690  vcFrameRxSOF <= inCellSOF after TPD_G;
691  vcFrameRxEOF <= inCellEOF after TPD_G;
692  vcFrameRxEOFE <= inCellEOFE after TPD_G;
693 
694  -- Paused or no data
695  else
696  vc0FrameRxValid <= '0' after TPD_G;
697  vc1FrameRxValid <= '0' after TPD_G;
698  vc2FrameRxValid <= '0' after TPD_G;
699  vc3FrameRxValid <= '0' after TPD_G;
700  end if;
701  end if;
702  end if;
703  end process;
704 
705 
706  -- Update buffer status on successfull cell reception
707  process ( pgpRxClk ) begin
708  if rising_edge(pgpRxClk) then
709  if pgpRxClkRst = '1' then
710  vc0RemAlmostFull <= '1' after TPD_G;
711  vc0RemOverflow <= '0' after TPD_G;
712  vc1RemAlmostFull <= '1' after TPD_G;
713  vc1RemOverflow <= '0' after TPD_G;
714  vc2RemAlmostFull <= '1' after TPD_G;
715  vc2RemOverflow <= '0' after TPD_G;
716  vc3RemAlmostFull <= '1' after TPD_G;
717  vc3RemOverflow <= '0' after TPD_G;
718  elsif pgpRxClkEn = '1' then
719  -- Link is not ready, force buffer states to bad
720  if pgpRxLinkReady = '0' then
721  vc0RemAlmostFull <= '1' after TPD_G;
722  vc0RemOverflow <= '0' after TPD_G;
723  vc1RemAlmostFull <= '1' after TPD_G;
724  vc1RemOverflow <= '0' after TPD_G;
725  vc2RemAlmostFull <= '1' after TPD_G;
726  vc2RemOverflow <= '0' after TPD_G;
727  vc3RemAlmostFull <= '1' after TPD_G;
728  vc3RemOverflow <= '0' after TPD_G;
729 
730  -- Update buffer status
731  elsif cellRxEOC = '1' then
732  vc0RemAlmostFull <= cellRxData(8) after TPD_G;
733  vc0RemOverflow <= cellRxData(12) after TPD_G;
734  vc1RemAlmostFull <= cellRxData(9) after TPD_G;
735  vc1RemOverflow <= cellRxData(13) after TPD_G;
736  vc2RemAlmostFull <= cellRxData(10) after TPD_G;
737  vc2RemOverflow <= cellRxData(14) after TPD_G;
738  vc3RemAlmostFull <= cellRxData(11) after TPD_G;
739  vc3RemOverflow <= cellRxData(15) after TPD_G;
740  end if;
741  end if;
742  end if;
743  end process;
744 
745 end Pgp2bRxCell;

97 architecture rtl of ConnFSM is
98  --
99  constant SAMPLES_PER_TIME_C : integer := integer(TIMEOUT_UNIT_G * CLK_FREQUENCY_G);
100  --
101  type StateType is (
102  CLOSED_S,
103  SEND_SYN_S,
104  WAIT_SYN_S,
105  LISTEN_S,
106  SEND_ACK_S,
107  SEND_SYN_ACK_S,
108  WAIT_ACK_S,
109  SEND_RST_S,
110  OPEN_S
111  );
112 
113  type RegType is record
114  connActive : sl;
115  closed : sl;
116  sndSyn : sl;
117  sndAck : sl;
118  sndRst : sl;
119  txAckF : sl;
120  peerTout : sl;
121  paramReject : sl;
122 
123  rssiParam : RssiParamType;
124 
125  --
126  txBufferSize : integer range 1 to 2 ** (SEGMENT_ADDR_SIZE_G);
127  txWindowSize : integer range 1 to 2 ** (WINDOW_ADDR_SIZE_G);
128  --
129  timeoutCntr : integer range 0 to RETRANS_TOUT_G * SAMPLES_PER_TIME_C;
130  resendCntr : integer range 0 to MAX_RETRANS_CNT_G+1;
131 
132  ---
133  state : StateType;
134 
135  end record RegType;
136 
137  constant REG_INIT_C : RegType := (
138  connActive => '0',
139  closed => '1',
140  sndSyn => '0',
141  sndAck => '0',
142  sndRst => '0',
143  txAckF => '0',
144  peerTout => '0',
145  paramReject => '0',
146 
147  rssiParam => RSSI_PARAM_INIT_C,
148 
149  timeoutCntr => 0,
150  resendCntr => 0,
151  --
152  txBufferSize=> 1,
153  txWindowSize=> 1,
154 
155  ---
156  state => CLOSED_S
157 
158  );
159 
160  signal r : RegType := REG_INIT_C;
161  signal rin : RegType;
162 
163 
164 
165 begin
166 
167  comb : process (r, rst_i, connRq_i, rxRssiParam_i, synHeadSt_i, rxFlags_i,
168  ackHeadSt_i, rstHeadSt_i, appRssiParam_i, rxValid_i, closeRq_i) is
169  variable v : RegType;
170  begin
171  v := r;
172 
173  -- /////////////////////////////////////////////////////////
174  ------------------------------------------------------------
175  -- Connection FSM
176  -- Synchronization and parameter negotiation
177  ------------------------------------------------------------
178  -- /////////////////////////////////////////////////////////
179 
180  case r.state is
181  ----------------------------------------------------------------------
182  when CLOSED_S =>
183 
184  -- Initialize parameters
185  v := REG_INIT_C;
186 
187  -- Next state condition
188  -- Initiated by high level App and depends on whether it is a
189  -- server or client.
190  if (connRq_i = '1' and SERVER_G = true) then
191  v.state := LISTEN_S; -- Server Passive open
192  elsif (connRq_i = '1' and SERVER_G = false) then
193  v.state := SEND_SYN_S;-- Client Active open
194  end if;
195  ----------------------------------------------------------------------
196  -- Client
197  -- Actively open connection
198  -- Propose parameters
199  -- Accept parameters from server or close connection
200  ----------------------------------------------------------------------
201  when SEND_SYN_S =>
202 
203  v.connActive := '0';
204  v.closed := '0';
205  v.sndSyn := '1';
206  v.sndAck := '0';
207  v.sndRst := '0';
208  v.txAckF := '0';
209  v.peerTout := '0';
210  v.paramReject := '0';
211  v.timeoutCntr := 0;
212 
213  -- Send the Client proposed parameters
214  v.rssiParam := appRssiParam_i;
215 
216  if (synHeadSt_i = '1') then
217  v.state := WAIT_SYN_S;
218  end if;
219 
220  ----------------------------------------------------------------------
221  when WAIT_SYN_S =>
222 
223  v.connActive := '0';
224  v.sndSyn := '0';
225  v.sndAck := '0';
226  v.sndRst := '0';
227  v.txAckF := '0';
228  v.timeoutCntr := r.timeoutCntr + 1;
229  --
230  if (rxValid_i = '1' and rxFlags_i.syn = '1' and rxFlags_i.ack = '1') then
231  -- Check parameters
232  if (
233  rxRssiParam_i.version = appRssiParam_i.version and -- Version match
234  rxRssiParam_i.chksumEn = appRssiParam_i.chksumEn and -- Checksum match
235  rxRssiParam_i.timeoutUnit= appRssiParam_i.timeoutUnit -- Timeout unit match
236  ) then
237  -- Accept the parameters from the server
238  v.rssiParam := rxRssiParam_i;
239  v.txBufferSize := conv_integer(rxRssiParam_i.maxSegSize(15 downto 3)); -- Divide by 8
240  v.txWindowSize := conv_integer(rxRssiParam_i.maxOutsSeg);
241 
242  -- Check the sizes and saturate to MAX if bigger
243  if(rxRssiParam_i.maxOutsSeg > (2**WINDOW_ADDR_SIZE_G)) then -- Number of segments in a window
244  v.rssiParam.maxOutsSeg := toSlv((2**WINDOW_ADDR_SIZE_G), v.rssiParam.maxOutsSeg'length);
245  v.txWindowSize := (2**WINDOW_ADDR_SIZE_G);
246  end if;
247 
248  if(rxRssiParam_i.maxSegSize > (2**SEGMENT_ADDR_SIZE_G)*8) then -- Number of bytes in a segment
249  v.rssiParam.maxSegSize := toSlv((2**SEGMENT_ADDR_SIZE_G)*8, v.rssiParam.maxSegSize'length);
250  v.txBufferSize := (2**SEGMENT_ADDR_SIZE_G); -- Divide by 8
251  end if;
252  --
253  v.state := SEND_ACK_S;
254  else
255  -- Reject parameters and reset the connection
256  v.paramReject := '1';
257  v.rssiParam := r.rssiParam;
258  --
259  v.state := SEND_RST_S;
260  end if;
261  elsif (rxValid_i = '1' and rxFlags_i.rst = '1') then
262  v.state := CLOSED_S;
263  elsif (r.timeoutCntr = RETRANS_TOUT_G * SAMPLES_PER_TIME_C) then
264  if (r.resendCntr >= MAX_RETRANS_CNT_G) then
265  v.peerTout := '1';
266  v.state := CLOSED_S;
267  else
268  v.closed := '1'; -- Signal closed to reset seqN (Resend SYN with the initial seqN)
269  v.resendCntr := r.resendCntr + 1;
270  v.state := SEND_SYN_S;
271  end if;
272  end if;
273  ----------------------------------------------------------------------
274  when SEND_ACK_S =>
275  --
276  v.connActive := '0';
277  v.sndSyn := '0';
278  v.sndAck := '1';
279  v.sndRst := '0';
280  v.txAckF := '1';
281  v.timeoutCntr := 0;
282  v.resendCntr := 0;
283  v.paramReject := '0';
284  --
285  v.rssiParam := r.rssiParam;
286 
287  --
288  if (ackHeadSt_i = '1') then
289  v.state := OPEN_S;
290  end if;
291 
292  ----------------------------------------------------------------------
293  -- Server
294  -- Passively open connection. Go to listen state and wait for SYN.
295  -- Check clients parameters:
296  -- If valid accept parameters,
297  -- If not valid propose new parameters.
298  ----------------------------------------------------------------------
299  when LISTEN_S =>
300  --
301  v.connActive := '0';
302  v.closed := '0';
303  v.sndSyn := '0';
304  v.sndAck := '0';
305  v.sndRst := '0';
306  v.txAckF := '0';
307  v.peerTout := '0';
308  v.timeoutCntr := 0;
309  --
310  if (rxValid_i = '1' and rxFlags_i.syn = '1') then
311  -- Accept the parameters from the client
312  v.rssiParam := rxRssiParam_i;
313  v.txWindowSize := conv_integer(rxRssiParam_i.maxOutsSeg);
314  v.txBufferSize := conv_integer(rxRssiParam_i.maxSegSize(15 downto 3)); -- Divide by 8
315  -- Set the receiver side information to be sent
316  v.rssiParam.maxOutsSeg := appRssiParam_i.maxOutsSeg;
317  v.rssiParam.maxSegSize := appRssiParam_i.maxSegSize;
318 
319  -- Check the sizes and saturate to MAX if bigger
320  if(rxRssiParam_i.maxOutsSeg > (2**WINDOW_ADDR_SIZE_G)) then -- Number of segments in a window
321  v.txWindowSize := (2**WINDOW_ADDR_SIZE_G);
322  end if;
323 
324  if(rxRssiParam_i.maxSegSize > (2**SEGMENT_ADDR_SIZE_G)*8) then -- Number of bytes in a segment
325  v.txBufferSize := (2**SEGMENT_ADDR_SIZE_G); -- Divide by 8
326  end if;
327 
328  --
329  v.paramReject := '0';
330 
331  -- Check parameters that have to match
332  if (
333  rxRssiParam_i.version /= appRssiParam_i.version or -- Version equality
334  rxRssiParam_i.chksumEn /= appRssiParam_i.chksumEn or -- Checksum match
335  rxRssiParam_i.timeoutUnit/= appRssiParam_i.timeoutUnit -- Timeout unit match
336  ) then
337 
338  -- Propose different parameters (overwrite)
339  v.rssiParam.version := appRssiParam_i.version;
340  v.rssiParam.timeoutUnit := appRssiParam_i.timeoutUnit;
341  v.rssiParam.chksumEn := appRssiParam_i.chksumEn;
342  --
343  v.paramReject := '1';
344  end if;
345  -- Go to ACK state
346  v.state := SEND_SYN_ACK_S;
347  end if;
348  ---------------------------------------------------------------------
349  when SEND_SYN_ACK_S =>
350 
351  v.connActive := '0';
352  v.closed := '0';
353  v.sndSyn := '1';
354  v.sndAck := '0';
355  v.sndRst := '0';
356  v.txAckF := '1';
357  v.paramReject := '0';
358  v.timeoutCntr := 0;
359 
360  -- Send the Server parameters
361  v.rssiParam := r.rssiParam;
362 
363  if (synHeadSt_i = '1') then
364  v.state := WAIT_ACK_S;
365  end if;
366 
367  when WAIT_ACK_S =>
368 
369  v.connActive := '0';
370  v.sndSyn := '0';
371  v.sndAck := '0';
372  v.sndRst := '0';
373  v.txAckF := '0';
374  v.paramReject := '0';
375  --
376  v.timeoutCntr := r.timeoutCntr+1;
377 
378  --
379  v.rssiParam := r.rssiParam;
380 
381  if (rxValid_i = '1' and rxFlags_i.ack = '1') then
382  v.state := OPEN_S;
383  elsif (rxValid_i = '1' and rxFlags_i.rst = '1') then
384  v.state := CLOSED_S;
385  elsif (r.timeoutCntr = RETRANS_TOUT_G * SAMPLES_PER_TIME_C) then
386  if (r.resendCntr >= MAX_RETRANS_CNT_G) then
387  v.peerTout := '1';
388  v.state := CLOSED_S;
389  else
390  v.closed := '1'; -- Signal closed to reset seqN (Resend SYN with the initial seqN)
391  v.resendCntr := r.resendCntr + 1;
392  v.state := SEND_SYN_ACK_S;
393  end if;
394  end if;
395 
396  ----------------------------------------------------------------------
397  -- Open connection
398  --
399  --
400  ----------------------------------------------------------------------
401  when OPEN_S =>
402  --
403  v.connActive := '1';
404  v.sndSyn := '0';
405  v.sndAck := '0';
406  v.sndRst := '0';
407  v.txAckF := '1';
408  v.paramReject := '0';
409  --
410  v.timeoutCntr := 0;
411  v.resendCntr := 0;
412  --
413  v.rssiParam := r.rssiParam;
414 
415  --
416  if (rxValid_i = '1' and rxFlags_i.rst = '1') then
417  v.state := SEND_RST_S;
418  elsif (closeRq_i = '1') then
419  v.state := SEND_RST_S;
420  end if;
421 
422  ----------------------------------------------------------------------
423  -- Reset the connection
424  -- - Send Rst segment
425  -- - Go to Closed
426  ----------------------------------------------------------------------
427  when SEND_RST_S =>
428  --
429  v.connActive := r.connActive;
430  v.sndSyn := '0';
431  v.sndAck := '0';
432  v.sndRst := '1';
433  v.txAckF := '0';
434  v.paramReject := '0';
435  --
436  v.rssiParam := r.rssiParam;
437 
438  --
439  if (rstHeadSt_i = '1') then
440  v.state := CLOSED_S;
441  end if;
442 
443  ----------------------------------------------------------------------
444  when others =>
445  v := REG_INIT_C;
446  ----------------------------------------------------------------------
447  end case;
448 
449  if (rst_i = '1') then
450  v := REG_INIT_C;
451  end if;
452 
453  rin <= v;
454  -----------------------------------------------------------
455  end process comb;
456 
457  seq : process (clk_i) is
458  begin
459  if (rising_edge(clk_i)) then
460  r <= rin after TPD_G;
461  end if;
462  end process seq;
463 
464  ------------------------------------------------------------------------------
465  -- Output assignment
466  rssiParam_o <= r.rssiParam;
467  connActive_o <= r.connActive;
468  sndSyn_o <= r.sndSyn;
469  sndAck_o <= r.sndAck;
470  sndRst_o <= r.sndRst;
471  txAckF_o <= r.txAckF;
472 
473  -- Parameters for receiver (have to be correctly set by the app)
474  rxBufferSize_o <= conv_integer(appRssiParam_i.maxSegSize(15 downto 3)); -- Divide by 8
475  rxWindowSize_o <= conv_integer(appRssiParam_i.maxOutsSeg);
476  -- Parameters for transmitter are received by the peer and checked by FSM
477  txBufferSize_o <= r.txBufferSize;
478  txWindowSize_o <= r.txWindowSize;
479  closed_o <= r.closed;
480  --
481  peerTout_o <= r.peerTout;
482  paramReject_o <= r.paramReject;
483  ---------------------------------------------------------------------
484 end architecture rtl;

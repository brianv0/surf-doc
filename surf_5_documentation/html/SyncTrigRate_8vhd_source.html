131  comb : process (r, refRst, trigCntSync) is
132  variable v : RegType;
133  begin
134  -- Latch the current value
135  v := r;
136 
137  -- Reset strobing signals
138  v.updated := '0';
139  v.updateStat := '0';
140 
141  -- Check for timeout
142  if r.timer = TIMEOUT_C then
143  -- Reset the timer
144  v.timer := 0;
145  -- Update the rate measurement
146  v.updated := '1';
147  v.rate := trigCntSync - r.trigCntDly;
148  -- Keep a delayed copy of trigCntSync
149  v.trigCntDly := trigCntSync;
150  else
151  -- Increment the timer
152  v.timer := r.timer + 1;
153  end if;
154 
155  -- Check for update previous clock cycle
156  if (r.updated = '1') then
157  -- Set the flag
158  v.updateStat := '1';
159  -- Check if first time after reset
160  if (r.armed = '0') then
161  -- Set the flag
162  v.armed := '1';
163  -- Pass the current values to the statistics measurements
164  v.rateMax := r.rate;
165  v.rateMin := r.rate;
166  else
167  -- Compare for max. value
168  if (r.rate > r.rateMax) then
169  -- Update the statistics measurement
170  v.rateMax := r.rate;
171  end if;
172  -- Compare for min. value
173  if (r.rate < r.rateMin) then
174  -- Update the statistics measurement
175  v.rateMin := r.rate;
176  end if;
177  end if;
178  end if;
179 
180  -- Reset
181  if refRst = '1' then
182  -- Re-arm and reset statistics measurements only
183  v.armed := '0';
184  v.rateMax := r.rate;
185  v.rateMin := r.rate;
186  end if;
187 
188  -- Register the variable for next clock cycle
189  rin <= v;
190 
191  end process comb;

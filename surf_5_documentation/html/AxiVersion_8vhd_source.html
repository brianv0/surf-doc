71  constant RELOAD_COUNT_C : integer := integer(AUTO_RELOAD_TIME_G / CLK_PERIOD_G);
72  constant TIMEOUT_1HZ_C : natural := (getTimeRatio(1.0, CLK_PERIOD_G) -1);

78  type RegType is record

167  comb : process (axiReadMaster, axiRst, axiWriteMaster, dnaValue, fdValue,
168  fpgaEnReload, r, userValues) is
169  variable v : RegType;
170  variable axilEp : AxiLiteEndpointType;
171  begin
172  -- Latch the current value
173  v := r;
174 
175  -- Reset strobes
176  v.masterReset := '0';
177 
178  ---------------------------------
179  -- First Stage Boot Loader (FSBL)
180  ---------------------------------
181  -- Check if timer enabled
182  if fpgaEnReload = '1' then
183  v.counter := v.counter + 1;
184  end if;
185 
186  -- Check for reload condition
187  if AUTO_RELOAD_EN_G and (r.counter = RELOAD_COUNT_C) and (fpgaEnReload = '1') and (r.haltReload = '0') then
188  v.fpgaReload := '1';
189  end if;
190 
191  ------------------------
192  -- AXI-Lite Transactions
193  ------------------------
194 
195  -- Determine the transaction type
196  axiSlaveWaitTxn(axilEp, axiWriteMaster, axiReadMaster, v.axiWriteSlave, v.axiReadSlave);
197 
198  axiSlaveRegisterR(axilEp, x"000", 0, BUILD_INFO_C.fwVersion);
199  axiSlaveRegister(axilEp, x"004", 0, v.scratchPad);
200  axiSlaveRegisterR(axilEp, x"008", 0, r.upTimeCnt);
201 
202  axiSlaveRegister(axilEp, x"100", 0, v.haltReload);
203  axiSlaveRegister(axilEp, x"104", 0, v.fpgaReload);
204  axiSlaveRegister(axilEp, x"108", 0, v.fpgaReloadAddr);
205  axiSlaveRegister(axilEp, x"10C", 0, v.masterReset);
206 
207  axiSlaveRegisterR(axilEp, x"300", 0, fdValue);
208  axiSlaveRegisterR(axilEp, x"400", userValues);
209  axiSlaveRegisterR(axilEp, x"500", 0, DEVICE_ID_G);
210 
211  -- axiSlaveRegisterR(axilEp, x"600", 0, BUILD_INFO_C.gitHash);-- axiSlaveRegisterR() Broken, only first 32-bit show up in software
212  axiSlaveRegisterR(axilEp, x"600", 0, BUILD_INFO_C.gitHash(31 downto 0));
213  axiSlaveRegisterR(axilEp, x"604", 0, BUILD_INFO_C.gitHash(63 downto 32));
214  axiSlaveRegisterR(axilEp, x"608", 0, BUILD_INFO_C.gitHash(95 downto 64));
215  axiSlaveRegisterR(axilEp, x"60C", 0, BUILD_INFO_C.gitHash(127 downto 96));
216  axiSlaveRegisterR(axilEp, x"610", 0, BUILD_INFO_C.gitHash(159 downto 128));
217 
218  axiSlaveRegisterR(axilEp, x"700", 0, dnaValue);
219  axiSlaveRegisterR(axilEp, x"800", BUILD_STRING_ROM_C);
220 
221  axiSlaveDefault(axilEp, v.axiWriteSlave, v.axiReadSlave, AXI_ERROR_RESP_G);
222 
223  ---------------------------------
224  -- Uptime counter
225  ---------------------------------
226  if r.timer = TIMEOUT_1HZ_C then
227  v.timer := 0;
228  v.upTimeCnt := r.upTimeCnt + 1;
229  else
230  v.timer := r.timer + 1;
231  end if;
232 
233  --------
234  -- Reset
235  --------
236  if (axiRst = '1') then
237  v := REG_INIT_C;
238  end if;
239 
240  -- Register the variable for next clock cycle
241  rin <= v;
242 
243  -- Outputs
244  axiReadSlave <= r.axiReadSlave;
245  axiWriteSlave <= r.axiWriteSlave;
246  fpgaReload <= r.fpgaReload;
247  fpgaReloadAddr <= r.fpgaReloadAddr;
248  masterRstDet <= v.masterReset;
249  upTimeCnt <= r.upTimeCnt;
250 
251  end process comb;
252 
253  seq : process (axiClk) is
254  begin
255  if (rising_edge(axiClk)) then
256  r <= rin after TPD_G;
257  end if;
258  end process seq;
259 
260  asyncRst <= axiRst or masterRstDet;
261

265  port map (
266  clk => axiClk,
267  asyncRst => asyncRst,
268  syncRst => masterReset);

270 end architecture rtl;

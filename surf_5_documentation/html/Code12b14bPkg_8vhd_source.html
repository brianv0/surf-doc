292  -- Determine the disparity of a vector
293  function getDisparity (vec : slv) return BlockDisparityType is
294  variable ones : integer;
295  variable zeros : integer;
296  variable disparity : BlockDisparityType;
297  begin
298  zeros := 0;
299  ones := 0;
300  for i in vec'range loop
301  if (vec(i) = '0') then
302  zeros := zeros + 1;
303  end if;
304  end loop;
305 
306  ones := vec'length-zeros;
307  disparity := ones-zeros;
308 
309  return disparity;
310 
311  end function getDisparity;

313  function toSlv (d : BlockDisparityType) return slv is

331  function toBlockDisparityType (d : slv(1 downto 0)) return BlockDisparityType is
332  begin
333  if (d = "10") then
334  return -2;
335  elsif (d = "11") then
336  return 0;
337  elsif (d = "00") then
338  return 2;
339  elsif (d = "01") then
340  return 4;
341  end if;
342  return 0;

346  -- Given an running disparity and a selected code disparity,
347  -- determine whether the selected code needs to be complimented, and what the out disparity is
348  -- Should maybe implement DisparityType as a constrained integer and just use math here
349  -- instead of this state machine. Not sure which is better.
350  procedure disparityControl (
351  prevDisp : in slv(1 downto 0);
352  blockDisp : in BlockDisparityType;
353  compliment : inout sl) is
354  variable dispInt : BlockDisparityType;
355  begin
356  compliment := '0';
357  dispInt := toBlockDisparityType(prevDisp);
358 
359  case prevDisp is
360  when "10" => -- -2
361  if (blockDisp = -2 or blockDisp = -4) then
362  compliment := '1';
363  end if;
364  when "11" => -- 0
365  if (blockDisp = -4) then
366  compliment := '1';
367  end if;
368  when "00" => -- 2
369  if (blockDisp = 2 or blockDisp = 4) then
370  compliment := '1';
371  end if;
372  when "01" => -- 4
373  if (blockDisp = 2 or blockDisp = 4) then
374  compliment := '1';
375  end if;
376  when others =>
377  null;
378  end case;
379 
380  end procedure;

394  else
395  ret(i).alt8b := ret(i).out8b;

397  end loop;

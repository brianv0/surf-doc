72  function GetFifoType (d_width : in integer; a_width : in integer) return string is
73  begin
74  if ((d_width >= 37) and (d_width <= 72) and (a_width = 9)) then
75  return "36Kb";
76  elsif ((d_width >= 19) and (d_width <= 36) and (a_width = 10)) then
77  return "36Kb";
78  elsif ((d_width >= 19) and (d_width <= 36) and (a_width = 9)) then
79  return "18Kb";
80  elsif ((d_width >= 10) and (d_width <= 18) and (a_width = 11)) then
81  return "36Kb";
82  elsif ((d_width >= 10) and (d_width <= 18) and (a_width = 10)) then
83  return "18Kb";
84  elsif ((d_width >= 5) and (d_width <= 9) and (a_width = 12)) then
85  return "36Kb";
86  elsif ((d_width >= 5) and (d_width <= 9) and (a_width = 11)) then
87  return "18Kb";
88  elsif ((d_width >= 1) and (d_width <= 4) and (a_width = 13)) then
89  return "36Kb";
90  elsif ((d_width >= 1) and (d_width <= 4) and (a_width = 12)) then
91  return "18Kb";
92  else
93  return "???Kb"; -- Generate error in Xilinx marco
94  end if;
95  end;

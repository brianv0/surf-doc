52  type StateType is (

68  arpSel => '0',

70  dly => AXI_STREAM_MASTER_INIT_C,
71  ibArpMaster => AXI_STREAM_MASTER_INIT_C,

79 begin
80 
81  comb : process (ibArpSlave, ibIpv4Slave, localMac, obMacMaster, r, rst) is
82  variable v : RegType;
83  begin
84  -- Latch the current value
85  v := r;
86 
87  -- Reset strobing signals
88  v.obMacSlave.tReady := '0';
89  if ibArpSlave.tReady = '1' then
90  v.ibArpMaster.tValid := '0';
91  end if;
92  if ibIpv4Slave.tReady = '1' then
93  v.ibIpv4Master.tValid := '0';
94  end if;
95 
96  -- Check if there is data to move
97  if (obMacMaster.tValid = '1') and (v.ibArpMaster.tValid = '0') and (v.ibIpv4Master.tValid = '0') then
98  ----------------------------------------------------------------------
99  -- Checking for non-VLAN
100  ----------------------------------------------------------------------
101  if (VLAN_G = false) then
102  -- Accept for data
103  v.obMacSlave.tReady := '1';
104  -- Check for SOF and not EOF
105  if (ssiGetUserSof(EMAC_AXIS_CONFIG_C, obMacMaster) = '1') and (obMacMaster.tLast = '0') then
106  -- Reset the flags
107  v.arpSel := '0';
108  v.ipv4Sel := '0';
109  -- Check for a valid ARP EtherType
110  if (obMacMaster.tData(111 downto 96) = ARP_TYPE_C) then
111  -- Check the destination MAC address
112  if(obMacMaster.tData(47 downto 0) = BROADCAST_MAC_C) or (obMacMaster.tData(47 downto 0) = localMac) then
113  v.arpSel := '1';
114  v.ibArpMaster := obMacMaster;
115  end if;
116  -- Check for a valid IPV4 EtherType and (IPVersion + Header length)
117  elsif (obMacMaster.tData(111 downto 96) = IPV4_TYPE_C) and (obMacMaster.tData(119 downto 112) = x"45") then
118  -- Check the destination MAC address
119  if(obMacMaster.tData(47 downto 0) = BROADCAST_MAC_C) or (obMacMaster.tData(47 downto 0) = localMac) then
120  v.ipv4Sel := '1';
121  v.ibIpv4Master := obMacMaster;
122  end if;
123  end if;
124  elsif r.arpSel = '1' then
125  v.ibArpMaster := obMacMaster;
126  elsif r.ipv4Sel = '1' then
127  v.ibIpv4Master := obMacMaster;
128  end if;
129  if obMacMaster.tLast = '1' then
130  -- Reset the flags
131  v.arpSel := '0';
132  v.ipv4Sel := '0';
133  end if;
134  ----------------------------------------------------------------------
135  -- Checking for VLAN
136  ----------------------------------------------------------------------
137  else
138  -- State Machine
139  case r.state is
140  ----------------------------------------------------------------------
141  when IDLE_S =>
142  -- Accept for data
143  v.obMacSlave.tReady := '1';
144  -- Check for SOF and not EOF
145  if (ssiGetUserSof(EMAC_AXIS_CONFIG_C, obMacMaster) = '1') and (obMacMaster.tLast = '0') then
146  -- Check for a valid VLAN EtherType
147  if (obMacMaster.tData(111 downto 96) = VLAN_TYPE_C) then
148  -- Reset the flags
149  v.arpSel := '0';
150  v.ipv4Sel := '0';
151  -- Latch the data bus
152  v.dly := obMacMaster;
153  -- Next state
154  v.state := CHECK_S;
155  end if;
156  end if;
157  ----------------------------------------------------------------------
158  when CHECK_S =>
159  -- Check for a valid ARP EtherType
160  if (obMacMaster.tData(15 downto 0) = ARP_TYPE_C) then
161  -- Check the destination MAC address
162  if(r.dly.tData(47 downto 0) = BROADCAST_MAC_C) or (r.dly.tData(47 downto 0) = localMac) then
163  v.arpSel := '1';
164  v.ibArpMaster := r.dly;
165  end if;
166  -- Check for a valid IPV4 EtherType and (IPVersion + Header length)
167  elsif (obMacMaster.tData(15 downto 0) = IPV4_TYPE_C) and (obMacMaster.tData(23 downto 16) = x"45") then
168  -- Check the destination MAC address
169  if(r.dly.tData(47 downto 0) = BROADCAST_MAC_C) or (r.dly.tData(47 downto 0) = localMac) then
170  v.ipv4Sel := '1';
171  v.ibIpv4Master := r.dly;
172  end if;
173  end if;
174  -- Next state
175  v.state := MOVE_S;
176  ----------------------------------------------------------------------
177  when MOVE_S =>
178  -- Accept for data
179  v.obMacSlave.tReady := '1';
180  if r.arpSel = '1' then
181  v.ibArpMaster := obMacMaster;
182  elsif r.ipv4Sel = '1' then
183  v.ibIpv4Master := obMacMaster;
184  end if;
185  if obMacMaster.tLast = '1' then
186  -- Next state
187  v.state := IDLE_S;
188  end if;
189  ----------------------------------------------------------------------
190  end case;
191  end if;
192  end if;
193 
194  -- Reset
195  if (rst = '1') then
196  v := REG_INIT_C;
197  end if;
198 
199  -- Register the variable for next clock cycle
200  rin <= v;
201 
202  -- Outputs
203  obMacSlave <= v.obMacSlave;
204  ibArpMaster <= r.ibArpMaster;
205  ibIpv4Master <= r.ibIpv4Master;
206 
207  end process comb;
208 
209  seq : process (clk) is
210  begin
211  if (rising_edge(clk)) then
212  r <= rin after TPD_G;
213  end if;
214  end process seq;
215 
216 end rtl;

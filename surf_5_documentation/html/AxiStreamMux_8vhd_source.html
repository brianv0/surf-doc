52 architecture structure of AxiStreamMux is
53 
54  constant DEST_SIZE_C : integer := bitSize(NUM_SLAVES_G-1);
55  constant ARB_BITS_C : integer := 2**DEST_SIZE_C;
56  constant ACNT_SIZE_G : integer := bitSize(ILEAVE_REARB_G);
57 
58  type StateType is (
59  IDLE_S,
60  MOVE_S);

62  type RegType is record

72  constant REG_INIT_C : RegType := (
73  state => IDLE_S,
74  acks => (others => '0'),
75  ackNum => toSlv(NUM_SLAVES_G-1, DEST_SIZE_C),
76  valid => '0',
77  arbCnt => (others=>'0'),

88 begin
89 
90  assert (MODE_G /= "INDEXED" or (7 - TDEST_LOW_G + 1 >= log2(NUM_SLAVES_G)))
91  report "In INDEXED mode, TDest range 7 downto " & integer'image(TDEST_LOW_G) &
92  " is too small for NUM_SLAVES_G=" & integer'image(NUM_SLAVES_G)
93  severity error;
94 
95  assert (MODE_G /= "ROUTED" or (TDEST_ROUTES_G'length = NUM_SLAVES_G))
96  report "In ROUTED mode, length of TDEST_ROUTES_G: " & integer'image(TDEST_ROUTES_G'length) &
97  " must equal NUM_SLAVES_G: " & integer'image(NUM_SLAVES_G)
98  severity error;
99 
100  -- Override tdests according to the routing table
101  TDEST_REMAP : process (sAxisMasters) is
102  variable tmp : AxiStreamMasterArray(NUM_SLAVES_G-1 downto 0);
103  variable i : natural;
104  variable j : natural;
105  begin
106  tmp := sAxisMasters;
107  if MODE_G = "ROUTED" then
108  for i in NUM_SLAVES_G-1 downto 0 loop
109  for j in 7 downto 0 loop
110  if (TDEST_ROUTES_G(i)(j) = '1') then
111  tmp(i).tDest(j) := '1';
112  elsif(TDEST_ROUTES_G(i)(j) = '0') then
113  tmp(i).tDest(j) := '0';
114  else
115  tmp(i).tDest(j) := sAxisMasters(i).tDest(j);
116  end if;
117  end loop;
118  end loop;
119  end if;
120  sAxisMastersTmp <= tmp;
121  end process;
122 
123  comb : process (axisRst, disableSel, pipeAxisSlave, r, sAxisMastersTmp) is
124  variable v : RegType;
125  variable requests : slv(ARB_BITS_C-1 downto 0);
126  variable selData : AxiStreamMasterType;
127  variable i : natural;
128  begin
129  -- Latch the current value
130  v := r;
131 
132  -- Reset the flags
133  for i in 0 to (NUM_SLAVES_G-1) loop
134  v.slaves(i).tReady := '0';
135  end loop;
136  if pipeAxisSlave.tReady = '1' then
137  v.master.tValid := '0';
138  end if;
139 
140  -- Select source
141  if NUM_SLAVES_G = 1 then
142  selData := sAxisMastersTmp(0);
143  else
144  selData := sAxisMastersTmp(conv_integer(r.ackNum));
145  end if;
146 
147  if MODE_G = "INDEXED" then
148  selData.tDest(7 downto TDEST_LOW_G) := (others => '0');
149  selData.tDest(DEST_SIZE_C+TDEST_LOW_G-1 downto TDEST_LOW_G) := r.ackNum;
150  end if;
151 
152  -- Format requests
153  requests := (others => '0');
154  for i in 0 to (NUM_SLAVES_G-1) loop
155  requests(i) := sAxisMastersTmp(i).tValid and not disableSel(i);
156  end loop;
157 
158  -- State machine
159  case r.state is
160  ----------------------------------------------------------------------
161  when IDLE_S =>
162  -- Arbitrate between requesters
163  if r.valid = '0' then
164  arbitrate(requests, r.ackNum, v.ackNum, v.valid, v.acks);
165  else
166  -- Reset the Arbitration flag
167  v.valid := '0';
168  -- Check if need to move data
169  if (v.master.tValid = '0') and (selData.tValid = '1') then
170  -- Accept the data
171  v.slaves(conv_integer(r.ackNum)).tReady := '1';
172  -- Move the AXIS data
173  v.master := selData;
174  -- Check for no-tLast
175  if selData.tLast = '0' then
176  -- Next state
177  v.state := MOVE_S;
178  end if;
179  else
180  -- Next state
181  v.state := MOVE_S;
182  end if;
183  end if;
184  v.arbCnt := (others=>'0');
185  ----------------------------------------------------------------------
186  when MOVE_S =>
187  -- Check if need to move data
188  if (v.master.tValid = '0') and (selData.tValid = '1') then
189  -- Accept the data
190  v.slaves(conv_integer(r.ackNum)).tReady := '1';
191  -- Move the AXIS data
192  v.master := selData;
193  v.arbCnt := r.arbCnt + 1;
194  -- Check for tLast
195  if selData.tLast = '1' then
196  -- Next state
197  v.state := IDLE_S;
198 
199  -- Rearbitrate after n transactions if enabled
200  elsif (ILEAVE_EN_G = true) and (ILEAVE_REARB_G /= 0) and (r.arbCnt = (ILEAVE_REARB_G-1)) then
201  v.state := IDLE_S;
202  end if;
203 
204  -- RE-arbitrate on gaps if interleaving frames
205  elsif (v.master.tValid = '0') and (selData.tValid = '0') and (ILEAVE_EN_G = true) then
206  v.state := IDLE_S;
207  end if;
208  ----------------------------------------------------------------------
209  end case;
210 
211  -- Reset
212  if (axisRst = '1') then
213  v := REG_INIT_C;
214  end if;
215 
216  -- Register the variable for next clock cycle
217  rin <= v;
218 
219  -- Outputs
220  sAxisSlaves <= v.slaves;
221  pipeAxisMaster <= r.master;
222 
223  end process comb;

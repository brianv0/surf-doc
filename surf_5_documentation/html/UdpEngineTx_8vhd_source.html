112  v.ibSlaves := (others => AXI_STREAM_SLAVE_INIT_C);
113  if (txSlave.tReady = '1') then
114  v.txMaster.tValid := '0';
115  v.txMaster.tLast := '0';

120  -- State Machine
121  case r.state is
122  ----------------------------------------------------------------------
123  when IDLE_S =>
124  -- Check for roll over
125  if (r.index = (SIZE_G-1)) then
126  -- Reset the counter
127  v.index := 0;
128  else
129  -- Increment the counter
130  v.index := r.index + 1;
131  end if;
132  -- Check for DHCP data and remote MAC is non-zero
133  if (obDhcpMaster.tValid = '1') and (v.txMaster.tValid = '0') then
134  -- Check for SOF
135  if (ssiGetUserSof(EMAC_AXIS_CONFIG_C, obDhcpMaster) = '1') then
136  -- Write the first header
137  v.txMaster.tValid := '1';
138  v.txMaster.tData(47 downto 0) := (others => '1'); -- Destination MAC address
139  v.txMaster.tData(63 downto 48) := x"0000"; -- All 0s
140  v.txMaster.tData(95 downto 64) := (others => '0'); -- Source IP address
141  v.txMaster.tData(127 downto 96) := (others => '1'); -- Destination IP address
142  ssiSetUserSof(EMAC_AXIS_CONFIG_C, v.txMaster, '1');
143  -- Next state
144  v.state := DHCP_HDR_S;
145  else
146  -- Blow off the data
147  v.obDhcpSlave.tReady := '1';
148  end if;
149  -- Check for data and remote MAC is non-zero
150  elsif (ibMasters(r.index).tValid = '1') and (v.txMaster.tValid = '0')and (remoteMac(r.index) /= 0) then
151  -- Check for SOF
152  if (ssiGetUserSof(EMAC_AXIS_CONFIG_C, ibMasters(r.index)) = '1') then
153  -- Latch the index
154  v.chPntr := r.index;
155  -- Write the first header
156  v.txMaster.tValid := '1';
157  v.txMaster.tData(47 downto 0) := remoteMac(r.index); -- Destination MAC address
158  v.txMaster.tData(63 downto 48) := x"0000"; -- All 0s
159  v.txMaster.tData(95 downto 64) := localIp; -- Source IP address
160  v.txMaster.tData(127 downto 96) := remoteIp(r.index); -- Destination IP address
161  ssiSetUserSof(EMAC_AXIS_CONFIG_C, v.txMaster, '1');
162  -- Next state
163  v.state := HDR_S;
164  else
165  -- Blow off the data
166  v.ibSlaves(r.index).tReady := '1';
167  end if;
168  end if;
169  ------------------------------------------------
170  -- Notes: Non-Standard IPv4 Pseudo Header Format
171  ------------------------------------------------
172  -- tData[0][47:0] = DST MAC Address
173  -- tData[0][63:48] = zeros
174  -- tData[0][95:64] = SRC IP Address
175  -- tData[0][127:96] = DST IP address
176  -- tData[1][7:0] = zeros
177  -- tData[1][15:8] = Protocol Type = UDP
178  -- tData[1][31:16] = IPv4 Pseudo header length
179  -- tData[1][47:32] = SRC Port
180  -- tData[1][63:48] = DST Port
181  -- tData[1][79:64] = UDP Length
182  -- tData[1][95:80] = UDP Checksum
183  -- tData[1][127:96] = UDP Datagram
184  ------------------------------------------------
185  ----------------------------------------------------------------------
186  when DHCP_HDR_S =>
187  -- Check if ready to move data
188  if (obDhcpMaster.tValid = '1') and (v.txMaster.tValid = '0') then
189  -- Accept the data
190  v.obDhcpSlave.tReady := '1';
191  -- Write the Second header
192  v.txMaster.tValid := '1';
193  v.txMaster.tData(7 downto 0) := x"00"; -- All 0s
194  v.txMaster.tData(15 downto 8) := UDP_C; -- Protocol Type = UDP
195  v.txMaster.tData(31 downto 16) := x"0000"; -- IPv4 Pseudo header length = Calculated in EthMac core
196  v.txMaster.tData(47 downto 32) := DHCP_CPORT; -- Source port
197  v.txMaster.tData(63 downto 48) := DHCP_SPORT; -- Destination port
198  v.txMaster.tData(79 downto 64) := x"0000"; -- UDP length = Calculated in EthMac core
199  v.txMaster.tData(95 downto 80) := x"0000"; -- UDP checksum = Calculated in EthMac core
200  v.txMaster.tData(127 downto 96) := obDhcpMaster.tData(31 downto 0); -- UDP Datagram
201  v.txMaster.tKeep(11 downto 0) := x"FFF";
202  v.txMaster.tKeep(15 downto 12) := obDhcpMaster.tKeep(3 downto 0); -- UDP Datagram
203  -- Track the leftovers
204  v.tData(95 downto 0) := obDhcpMaster.tData(127 downto 32);
205  v.tData(127 downto 96) := (others => '0');
206  v.tKeep(11 downto 0) := obDhcpMaster.tKeep(15 downto 4);
207  v.tKeep(15 downto 12) := (others => '0');
208  v.tLast := obDhcpMaster.tLast;
209  v.eofe := ssiGetUserEofe(EMAC_AXIS_CONFIG_C, obDhcpMaster);
210  -- Check for tLast
211  if (v.tLast = '1') then
212  -- Check the leftover tKeep is not empty
213  if (v.tKeep /= 0) then
214  -- Next state
215  v.state := LAST_S;
216  else
217  -- Set EOF and EOFE
218  v.txMaster.tLast := '1';
219  ssiSetUserEofe(EMAC_AXIS_CONFIG_C, v.txMaster, v.eofe);
220  -- Next state
221  v.state := IDLE_S;
222  end if;
223  else
224  -- Next state
225  v.state := DHCP_BUFFER_S;
226  end if;
227  end if;
228  ----------------------------------------------------------------------
229  when HDR_S =>
230  -- Check if ready to move data
231  if (ibMasters(r.chPntr).tValid = '1') and (v.txMaster.tValid = '0') then
232  -- Accept the data
233  v.ibSlaves(r.chPntr).tReady := '1';
234  -- Write the Second header
235  v.txMaster.tValid := '1';
236  v.txMaster.tData(7 downto 0) := x"00"; -- All 0s

282  v.txMaster.tKeep(11 downto 0) := r.tKeep(11 downto 0);
283  v.txMaster.tKeep(15 downto 12) := obDhcpMaster.tKeep(3 downto 0);
284  -- Track the leftovers
285  v.tData(95 downto 0) := obDhcpMaster.tData(127 downto 32);
286  v.tKeep(11 downto 0) := obDhcpMaster.tKeep(15 downto 4);
287  v.tLast := obDhcpMaster.tLast;
288  v.eofe := ssiGetUserEofe(EMAC_AXIS_CONFIG_C, obDhcpMaster);
289  -- Check for tLast
290  if (v.tLast = '1') then
291  -- Check the leftover tKeep is not empty

293  -- Next state
294  v.state := LAST_S;

318  v.tData(95 downto 0) := ibMasters(r.chPntr).tData(127 downto 32);
319  v.tKeep(11 downto 0) := ibMasters(r.chPntr).tKeep(15 downto 4);
320  v.tLast := ibMasters(r.chPntr).tLast;
321  v.eofe := ssiGetUserEofe(EMAC_AXIS_CONFIG_C, ibMasters(r.chPntr));
322  -- Check for tLast
323  if (v.tLast = '1') then
324  -- Check the leftover tKeep is not empty
325  if (v.tKeep /= 0) then
326  -- Next state
327  v.state := LAST_S;
328  else
329  -- Set EOF and EOFE
330  v.txMaster.tLast := '1';
331  ssiSetUserEofe(EMAC_AXIS_CONFIG_C, v.txMaster, v.eofe);
332  -- Next state
333  v.state := IDLE_S;
334  end if;
335  end if;
336  end if;
337  ----------------------------------------------------------------------
338  when LAST_S =>
339  -- Check if ready to move data
340  if (v.txMaster.tValid = '0') then
341  -- Move the data
342  v.txMaster.tValid := '1';
343  v.txMaster.tData := r.tData;
344  v.txMaster.tKeep := r.tKeep;
345  v.txMaster.tLast := '1';
346  ssiSetUserEofe(EMAC_AXIS_CONFIG_C, v.txMaster, r.eofe);

351  end case;

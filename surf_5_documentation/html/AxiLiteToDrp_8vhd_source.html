134  comb : process (drpDo, drpGnt, drpRdy, drpRst, r, readMaster, writeMaster) is
135  variable v : RegType;
136  variable axiStatus : AxiLiteStatusType;
137  variable axiResp : slv(1 downto 0);
138  begin
139  -- Latch the current value
140  v := r;
141 
142  -- Reset the strobes
143  v.drpEn := '0';
144  v.drpWe := '0';
145  v.drpUsrRst := '0';
146 
147  -- Determine the transaction type
148  axiSlaveWaitTxn(writeMaster, readMaster, v.writeSlave, v.readSlave, axiStatus);
149 
150  -- State Machine
151  case (r.state) is
152  ----------------------------------------------------------------------
153  when IDLE_S =>
154  -- Reset the flags
155  v.drpReq := '0';
156  v.timer := 0;
157  -- Check for a write request
158  if (axiStatus.writeEnable = '1') then
159  -- Set the write address bus (32-bit access alignment)
160  v.drpAddr := writeMaster.awaddr(ADDR_WIDTH_G+1 downto 2);
161  -- Set the write data bus
162  v.drpDi := writeMaster.wdata(DATA_WIDTH_G-1 downto 0);
163  -- Check for DRP request/grant
164  if EN_ARBITRATION_G then
165  -- Next state
166  v.state := REQ_S;
167  else
168  -- Send a write command
169  v.drpEn := '1';
170  v.drpWe := '1';
171  -- Next state
172  v.state := ACK_S;
173  end if;
174  -- Check for a read request
175  elsif (axiStatus.readEnable = '1') then
176  -- Set the write address bus (32-bit access alignment)
177  v.drpAddr := readMaster.araddr(ADDR_WIDTH_G+1 downto 2);
178  -- Check for DRP request/grant
179  if EN_ARBITRATION_G then
180  -- Next state
181  v.state := REQ_S;
182  else
183  -- Send a read command
184  v.drpEn := '1';
185  v.drpWe := '0';
186  -- Next state
187  v.state := ACK_S;
188  end if;
189  end if;
190  ----------------------------------------------------------------------
191  when REQ_S =>
192  -- Request the DRP bus
193  v.drpReq := '1';
194  -- Check for DRP bus access granted
195  if (drpGnt = '1') or (r.timer = TIMEOUT_G) then
196  -- Check for a write request
197  if (axiStatus.writeEnable = '1') then
198  -- Check for non-timeout
199  if (drpGnt = '1') then
200  -- Reset the timer
201  v.timer := 0;
202  -- Send a write command
203  v.drpEn := '1';
204  v.drpWe := '1';
205  end if;
206  -- Next state
207  v.state := ACK_S;
208  -- Check for a read request
209  elsif (axiStatus.readEnable = '1') then
210  -- Check for non-timeout
211  if (drpGnt = '1') then
212  -- Reset the timer
213  v.timer := 0;
214  -- Send a read command
215  v.drpEn := '1';
216  v.drpWe := '0';
217  end if;
218  -- Next state
219  v.state := ACK_S;
220  else
221  -- Next state
222  v.state := IDLE_S;
223  end if;
224  else
225  -- Increment the timer
226  v.timer := r.timer + 1;
227  end if;
228  ----------------------------------------------------------------------
229  when ACK_S =>
230  -- Check for DRP acknowledgement of command
231  if (drpRdy = '1') or (r.timer = TIMEOUT_G) then
232  -- Check for non-timeout
233  if (drpRdy = '1') then
234  -- Return good transaction
235  axiResp := AXI_RESP_OK_C;
236  else
237  -- Return good transaction
238  axiResp := AXI_ERROR_RESP_G;
239  -- Attempt to re-initialize the DRP interface
240  v.drpUsrRst := '1';
241  end if;
242  -- Check for a write request
243  if (axiStatus.writeEnable = '1') then
244  -- Send AXI-Lite response
245  axiSlaveWriteResponse(v.writeSlave, axiResp);
246  -- Check for a read request
247  elsif (axiStatus.readEnable = '1') then
248  -- Set the read bus
249  v.readSlave.rdata(DATA_WIDTH_G-1 downto 0) := drpDo;
250  -- Send AXI-Lite Response
251  axiSlaveReadResponse(v.readSlave, axiResp);
252  end if;
253  -- Next state
254  v.state := IDLE_S;
255  else
256  -- Increment the timer
257  v.timer := r.timer + 1;
258  end if;
259  ----------------------------------------------------------------------
260  end case;
261 
262  -- Synchronous Reset
263  if (drpRst = '1') then
264  v := REG_INIT_C;
265  end if;
266 
267  -- Register the variable for next clock cycle
268  rin <= v;
269 
270  -- Outputs
271  readSlave <= r.readSlave;
272  writeSlave <= r.writeSlave;
273  drpReq <= r.drpReq;
274  drpEn <= r.drpEn;
275  drpWe <= r.drpWe;
276  drpAddr <= r.drpAddr;
277  drpDi <= r.drpDi;
278  drpUsrRst <= r.drpUsrRst;
279 
280  end process comb;
281

283  begin
284  if (rising_edge(drpClk)) then
285  r <= rin after TPD_G;
286  end if;
287  end process seq;

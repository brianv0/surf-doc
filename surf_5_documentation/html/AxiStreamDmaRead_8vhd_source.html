132  comb : process (axiReadSlave, axiRst, dmaReq, mMaster, mSlave, pause, r, sSlave, swCache) is
133  variable v : RegType;
134  variable readSize : integer;
135  variable reqLen : natural;
136  variable pending : boolean;
137  begin
138  -- Latch the current value
139  v := r;
140 
141  -- Set cache value if enabled in software
142  if SW_CACHE_EN_G then
143  v.rMaster.arcache := swCache;
144  end if;
145 
146  -- Reset strobing Signals
147  v.rMaster.rready := '0';
148  v.shiftEn := '0';
149  if (axiReadSlave.arready = '1') then
150  v.rMaster.arvalid := '0';
151  end if;
152  if (sSlave.tReady = '1') then
153  v.sMaster.tValid := '0';
154  v.sMaster.tLast := '0';
155  v.sMaster.tUser := (others => '0');
156  v.sMaster.tKeep := (others => '1');
157  v.sMaster.tStrb := (others => '1');
158  end if;
159 
160  -- Calculate the pending bytes
161  v.pendBytes := r.reqCnt - r.ackCnt;
162 
163  -- Update variables
164  reqLen := 0;
165  pending := true;
166  -- Check for the threshold = zero case
167  if (PEND_THRESH_G = 0) then
168  if (r.pendBytes = 0) then
169  pending := false;
170  end if;
171  else
172  if (r.pendBytes < PEND_THRESH_G) then
173  pending := false;
174  end if;
175  end if;
176 
177  -- Track read status
178  if (axiReadSlave.rvalid = '1') and (axiReadSlave.rresp /= 0) and (axiReadSlave.rlast = '1') then
179  -- Error Detected
180  v.dmaAck.readError := '1';
181  v.dmaAck.errorValue := axiReadSlave.rresp;
182  end if;
183 
184  -- Check for handshaking
185  if (dmaReq.request = '0') and (r.dmaAck.done = '1') then
186  -- Reset the flags
187  v.dmaAck.done := '0';
188  end if;
189 
190  -- Memory Request State machine
191  case r.reqState is
192  ----------------------------------------------------------------------
193  when IDLE_S =>
194  -- Update the variables
195  v.dmaReq := dmaReq;
196  -- Reset the counters
197  v.shift := (others => '0');
198  v.reqCnt := (others => '0');
199  v.ackCnt := (others => '0');
200  -- Align shift and address to transfer size
201  if (DATA_BYTES_C /= 1) then
202  v.dmaReq.address(ADDR_LSB_C-1 downto 0) := (others => '0');
203  v.shift(ADDR_LSB_C-1 downto 0) := dmaReq.address(ADDR_LSB_C-1 downto 0);
204  end if;
205  -- Check for DMA request
206  if (dmaReq.request = '1') then
207  -- Reset the flags and counters
208  v.dmaAck.readError := '0';
209  v.dmaAck.errorValue := (others => '0');
210  -- Set the flags
211  v.shiftEn := '1';
212  v.first := '1';
213  -- Latch the value
214  v.size := dmaReq.size;
215  v.reqSize := dmaReq.size;
216  v.sMaster.tDest := dmaReq.dest;
217  v.sMaster.tId := dmareq.id;
218  -- Next state
219  v.reqState := FIRST_S;
220  end if;
221  ----------------------------------------------------------------------
222  when FIRST_S =>
223  -- Check if ready to make memory request
224  if (r.rMaster.arvalid = '0') then
225  -- Set the memory address
226  v.rMaster.araddr(AXI_CONFIG_G.ADDR_WIDTH_C-1 downto 0) := r.dmaReq.address(AXI_CONFIG_G.ADDR_WIDTH_C-1 downto 0);
227  -- Determine transfer size to align address to 16-transfer boundaries
228  -- This initial alignment will ensure that we never cross a 4k boundary
229  if (ARLEN_C > 0) then
230  -- Set the burst length
231  v.rMaster.arlen := ARLEN_C - r.dmaReq.address(ADDR_LSB_C+AXI_CONFIG_G.LEN_BITS_C-1 downto ADDR_LSB_C);
232  -- Limit read burst size
233  if (r.reqSize(31 downto ADDR_LSB_C) < v.rMaster.arlen) then
234  v.rMaster.arlen := resize(r.reqSize(ADDR_LSB_C+AXI_CONFIG_G.LEN_BITS_C-1 downto ADDR_LSB_C)-1, 8);
235  end if;
236  end if;
237  -- There is enough room in the FIFO for a burst
238  if (pause = '0') then
239  -- Set the flag
240  v.rMaster.arvalid := '1';
241  -- Update the request size
242  reqLen := DATA_BYTES_C*(conv_integer(v.rMaster.arlen) + 1);
243  v.reqCnt := toSlv(reqLen, 32) - conv_integer(r.shift);
244  v.reqSize := r.reqSize - reqLen;
245  -- Update next address
246  v.dmaReq.address := r.dmaReq.address + reqLen;
247  v.dmaReq.address(ADDR_LSB_C-1 downto 0) := (others => '0');
248  -- Next state
249  v.reqState := NEXT_S;
250  v.state := MOVE_S;
251  end if;
252  end if;
253  ----------------------------------------------------------------------
254  when NEXT_S =>
255  -- Check if ready to make memory request
256  if (r.rMaster.arvalid = '0') then
257  -- Set the memory address
258  v.rMaster.araddr(AXI_CONFIG_G.ADDR_WIDTH_C-1 downto 0) := r.dmaReq.address(AXI_CONFIG_G.ADDR_WIDTH_C-1 downto 0);
259  -- Bursts after the FIRST are guaranteed to be aligned
260  v.rMaster.arlen := ARLEN_C;
261  -- Limit read burst size
262  if (r.reqSize(31 downto ADDR_LSB_C) < v.rMaster.arlen) then
263  v.rMaster.arlen := resize(r.reqSize(ADDR_LSB_C+AXI_CONFIG_G.LEN_BITS_C-1 downto ADDR_LSB_C)-1, 8);
264  end if;
265  -- Check for the following:
266  -- 1) There is enough room in the FIFO for a burst
267  -- 2) pending flag
268  -- 3) Last transaction already completed
269  if (pause = '0') and (pending = false) and (r.reqCnt < r.dmaReq.size) then
270  -- Set the flag
271  v.rMaster.arvalid := '1';
272  -- Update the request size
273  reqLen := DATA_BYTES_C*(conv_integer(v.rMaster.arlen) + 1);
274  v.reqCnt := r.reqCnt + reqLen;
275  v.reqSize := r.reqSize - reqLen;
276  -- Update next address
277  v.dmaReq.address := r.dmaReq.address + reqLen;
278  v.dmaReq.address(ADDR_LSB_C-1 downto 0) := (others => '0');
279  end if;
280  end if;
281  ----------------------------------------------------------------------
282  end case;
283 
284  -- Data Collection State machine
285  case r.state is
286  ----------------------------------------------------------------------
287  when IDLE_S =>
288  -- Blowoff any out-of-phase data (should never happen)
289  v.rMaster.rready := '1';
290  ----------------------------------------------------------------------
291  when MOVE_S =>
292  -- Check if ready to move data
293  if (v.sMaster.tValid = '0') and (axiReadSlave.rvalid = '1') then
294  -- Accept the data
295  v.rMaster.rready := '1';
296  -- Move the data
297  v.sMaster.tValid := '1';
298  v.sMaster.tData((DATA_BYTES_C*8)-1 downto 0) := axiReadSlave.rdata((DATA_BYTES_C*8)-1 downto 0);
299  -- Check the flag
300  if r.first = '1' then
301  -- Reset the flag
302  v.first := '0';
303  -- Set the tUser for the first byte transferred
304  axiStreamSetUserField(
305  AXIS_CONFIG_G,
306  v.sMaster,
307  r.dmaReq.
308  firstUser(AXIS_CONFIG_G.TUSER_BITS_C-1 downto 0),
309  conv_integer(r.shift));
310  end if;
311  -- Calculate the read size
312  readSize := DATA_BYTES_C - ite(r.first = '1', conv_integer(r.shift), 0);
313  -- Check the read size
314  if (readSize > r.size) then
315  -- Bottom out at 0
316  v.size := (others => '0');
317  -- Top out at dma.size
318  v.ackCnt := r.dmaReq.size;
319  else
320  -- Decrement the counter
321  v.size := r.size - readSize;
322  -- Increment the counter
323  v.ackCnt := r.ackCnt + readSize;
324  end if;
325  -- Check for completion
326  if (v.size = 0) then
327  -- Terminate the frame
328  v.sMaster.tLast := '1';
329  v.sMaster.tKeep := genTKeep(conv_integer(r.size(4 downto 0)));
330  v.sMaster.tStrb := genTKeep(conv_integer(r.size(4 downto 0)));
331  -- Check for first transfer
332  if (r.first = '1') then
333  -- Compensate the tKeep and tStrb via shift module
334  v.sMaster.tKeep := shl(v.sMaster.tKeep, r.shift);
335  v.sMaster.tStrb := shl(v.sMaster.tStrb, r.shift);
336  end if;
337  -- Set the flags
338  v.dmaAck.done := '1';
339  v.leftovers := not(axiReadSlave.rlast);
340  -- Next state
341  v.state := LAST_S;
342  end if;
343  end if;
344  ----------------------------------------------------------------------
345  when LAST_S =>
346  -- Hold until last transfer (required for Remap_mAxisMaster process)
347  if (mMaster.tValid = '1') and (mMaster.tLast = '1') and (mSlave.tReady = '1') then
348  -- Next state
349  v.state := DONE_S;
350  end if;
351  ----------------------------------------------------------------------
352  when DONE_S =>
353  -- Check for ACK completion
354  if (r.dmaAck.done = '0')then
355  -- Reset the flag
356  v.leftovers := '0';
357  -- Check if no leftover memory request data
358  if r.leftovers = '0' then
359  -- Next states
360  v.reqState := IDLE_S;
361  v.state := IDLE_S;
362  else
363  -- Next state
364  v.state := BLOWOFF_S;
365  end if;
366  end if;
367  ----------------------------------------------------------------------
368  when BLOWOFF_S =>
369  -- Blowoff the data
370  v.rMaster.rready := '1';
371  -- Check for last transfer
372  if (axiReadSlave.rvalid = '1') and (axiReadSlave.rlast = '1') then
373  -- Next states
374  v.reqState := IDLE_S;
375  v.state := IDLE_S;
376  end if;
377  ----------------------------------------------------------------------
378  end case;
379 
380  -- Forward the state of the state machine
381  if v.state = IDLE_S then
382  -- Set the flag
383  v.dmaAck.idle := '1';
384  else
385  -- Reset the flag
386  v.dmaAck.idle := '0';
387  end if;
388 
389  -- Reset
390  if (axiRst = '1') then
391  v := REG_INIT_C;
392  end if;
393 
394  -- Register the variable for next clock cycle
395  rin <= v;
396 
397  -- Outputs
398  dmaAck <= r.dmaAck;
399  axiReadMaster <= r.rMaster;
400  axiReadMaster.rready <= v.rMaster.rready;
401 
402  end process comb;
403 
404  seq : process (axiClk) is
405  begin
406  if (rising_edge(axiClk)) then
407  r <= rin after TPD_G;
408  end if;
409  end process seq;

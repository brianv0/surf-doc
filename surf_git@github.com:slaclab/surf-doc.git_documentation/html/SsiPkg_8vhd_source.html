147  function ssiAxiStreamConfig (
148  dataBytes : natural;
149  tKeepMode : TKeepModeType := TKEEP_COMP_C;
150  tUserMode : TUserModeType := TUSER_FIRST_LAST_C;
151  tDestBits : integer range 0 to 8 := 4;
152  tUserBits : integer range 2 to 8 := 2)
153  return AxiStreamConfigType is
154  variable ret : AxiStreamConfigType;
155  begin
156  ret.TDATA_BYTES_C := dataBytes; -- Configurable data size
157  ret.TUSER_BITS_C := tUserBits; -- 2 TUSER: EOFE, SOF
158  ret.TDEST_BITS_C := tDestBits; -- 4 TDEST bits for VC
159  ret.TID_BITS_C := SSI_TID_BITS_C; -- TID not used
160  ret.TKEEP_MODE_C := tKeepMode; --
161  ret.TSTRB_EN_C := SSI_TSTRB_EN_C; -- No TSTRB support in SSI
162  ret.TUSER_MODE_C := tUserMode; -- User field valid on last only
163  return ret;
164  end function ssiAxiStreamConfig;

176  -------------------------------------------------------------------------------------------------
177  function ssi2AxisMaster (
178  axisConfig : AxiStreamConfigType;
179  ssiMaster : SsiMasterType)
180  return AxiStreamMasterType
181  is
182  variable ret : AxiStreamMasterType;
183  begin
184  ret := AXI_STREAM_MASTER_INIT_C;
185  ret.tValid := ssiMaster.valid;
186  ret.tData := ssiMaster.data;
187  ret.tLast := ssiMaster.eof;
188  ret.tStrb := ssiMaster.strb;
189  ret.tKeep := ssiMaster.keep;
190  ret.tDest(SSI_TDEST_BITS_C-1 downto 0) := ssiMaster.dest;
191  ssiSetUserSof(axisConfig, ret, ssiMaster.sof);
192  ssiSetUserEofe(axisConfig, ret, ssiMaster.eofe);
193  return ret;
194  end function ssi2AxisMaster;

217  function axis2SsiMaster (
218  axisConfig : AxiStreamConfigType;
219  axisMaster : AxiStreamMasterType)
220  return SsiMasterType
221  is
222  variable ret : SsiMasterType;
223  begin
224  ret.valid := axisMaster.tValid;
225  ret.data := axisMaster.tData;
226  ret.strb := axisMaster.tStrb;
227  ret.keep := axisMaster.tKeep;
228  ret.packed := toSl(axiStreamPacked(axisConfig, axisMaster));
229  ret.dest := axisMaster.tDest(SSI_TDEST_BITS_C-1 downto 0);
230  ret.sof := ssiGetUserSof(axisConfig, axisMaster);
231  ret.eof := axisMaster.tLast;
232  ret.eofe := ssiGetUserEofe(axisConfig, axisMaster);
233  return ret;
234  end function axis2SsiMaster;

328  function getAxiLen (
329  axiConfig : AxiConfigType;
330  burstBytes : integer range 1 to 4096 := 4096)
331  return slv is
332  begin
333  -- burstBytes / data bytes width is number of txns required.
334  -- Subtract by 1 for A*LEN value for even divides.
335  -- Convert to SLV and truncate to size of A*LEN port for this AXI bus
336  -- This limits number of txns approraiately based on size of len port
337  -- Then resize to 8 bits because our records define A*LEN as 8 bits always.
338  return resize(toSlv(wordCount(burstBytes,axiConfig.DATA_BYTES_C)-1, axiConfig.LEN_BITS_C), 8);
339  end function getAxiLen;
340 
341  -- Calculate number of txns in a burst based upon burst size, total remaining bytes,
342  -- current address and bus configuration.
343  -- Address is used to set a transaction size aligned to 4k boundaries
344  -- Returned value is number of txns-1, so can be assigned to AWLEN/ARLEN
345  function getAxiLen (
346  axiConfig : AxiConfigType;
347  burstBytes : integer range 1 to 4096 := 4096;
348  totalBytes : slv;
349  address : slv)
350  return slv is
351  variable max : natural;
352  variable req : natural;
353  variable maxLen : slv(7 downto 0);
354  begin
355 
356  max := 4096 - conv_integer(address(11 downto 0));
357  req := minimum(conv_integer(totalBytes),burstBytes);
358 
359  return getAxiLen(axiConfig,minimum(req,max));
360 
361  end function getAxiLen;
362 
363  -- Calculate the byte count for a read request
364  function getAxiReadBytes (
365  axiConfig : AxiConfigType;
366  axiRead : AxiReadMasterType)
367  return slv is
368  constant addrLsb : natural := bitSize(AxiConfig.DATA_BYTES_C-1);
369  variable tempSlv : slv(AxiConfig.LEN_BITS_C+addrLsb downto 0);
370  begin
371  tempSlv := (others=>'0');
372 
373  tempSlv(AxiConfig.LEN_BITS_C+addrLsb downto addrLsb)
374  := axiRead.arlen(AxiConfig.LEN_BITS_C-1 downto 0) + toSlv(1,AxiConfig.LEN_BITS_C+1);
375 
376  tempSlv := tempSlv - axiRead.araddr(addrLsb-1 downto 0);
377 
378  return(tempSlv);
379  end function getAxiReadBytes;

172  comb : process (axilReadMaster, axilWriteMaster, r, axiRst_i, s_statusTxArr, s_RdAddr, s_WrAddr, s_statusCnt) is
173  variable v : RegType;
174  variable axilStatus : AxiLiteStatusType;
175  variable axilWriteResp : slv(1 downto 0);
176  variable axilReadResp : slv(1 downto 0);
177  begin
178  -- Latch the current value
179  v := r;
180 
181  ----------------------------------------------------------------------------------------------
182  -- Axi-Lite interface
183  ----------------------------------------------------------------------------------------------
184  axiSlaveWaitTxn(axilWriteMaster, axilReadMaster, v.axilWriteSlave, v.axilReadSlave, axilStatus);
185 
186  if (axilStatus.writeEnable = '1') then
187  axilWriteResp := ite(axilWriteMaster.awaddr(1 downto 0) = "00", AXI_RESP_OK_C, AXI_ERROR_RESP_G);
188  case (s_WrAddr) is
189  when 16#00# => -- ADDR (0x0)
190  v.enableTx := axilWriteMaster.wdata(L_G-1 downto 0);
191  when 16#01# => -- ADDR (0x4)
192  v.sysrefDlyTx := axilWriteMaster.wdata(SYSRF_DLY_WIDTH_C-1 downto 0);
193  when 16#02# => -- ADDR (0x8)
194  v.txPolarity := axilWriteMaster.wdata(L_G-1 downto 0);
195  when 16#03# => -- ADDR (0xC)
196  v.loopback := axilWriteMaster.wdata(L_G-1 downto 0);
197  when 16#04# => -- ADDR (0x10)
198  v.commonCtrl := axilWriteMaster.wdata(6 downto 0);
199  when 16#05# => -- ADDR (0x14)
200  v.periodStep := axilWriteMaster.wdata;
201  when 16#06# => -- ADDR (0x18)
202  v.negAmplitude := axilWriteMaster.wdata(F_G*8-1 downto 0);
203  when 16#07# => -- ADDR (0x1C)
204  v.posAmplitude := axilWriteMaster.wdata(F_G*8-1 downto 0);
205  when 16#08# => -- ADDR (0x20)
206  v.invertData := axilWriteMaster.wdata(L_G-1 downto 0);
207  when 16#20# to 16#2F# =>
208  for I in (L_G-1) downto 0 loop
209  if (axilWriteMaster.awaddr(5 downto 2) = I) then
210  v.signalSelectArr(I) := axilWriteMaster.wdata(7 downto 0);
211  end if;
212  end loop;
213  when 16#80# to 16#9F# =>
214  for I in (L_G-1) downto 0 loop
215  if (axilWriteMaster.awaddr(6 downto 2) = I) then
216  v.txDiffCtrl(I) := axilWriteMaster.wdata(7 downto 0);
217  v.txPostCursor(I) := axilWriteMaster.wdata(15 downto 8);
218  v.txPreCursor(I) := axilWriteMaster.wdata(23 downto 16);
219  end if;
220  end loop;
221  when others =>
222  axilWriteResp := AXI_ERROR_RESP_G;
223  end case;
224  axiSlaveWriteResponse(v.axilWriteSlave);
225  end if;
226 
227  if (axilStatus.readEnable = '1') then
228  axilReadResp := ite(axilReadMaster.araddr(1 downto 0) = "00", AXI_RESP_OK_C, AXI_ERROR_RESP_G);
229  v.axilReadSlave.rdata := (others => '0');
230  case (s_RdAddr) is
231  when 16#00# => -- ADDR (0x0)
232  v.axilReadSlave.rdata(L_G-1 downto 0) := r.enableTx;
233  when 16#01# => -- ADDR (0x4)
234  v.axilReadSlave.rdata(SYSRF_DLY_WIDTH_C-1 downto 0) := r.sysrefDlyTx;
235  when 16#02# => -- ADDR (0x8)
236  v.axilReadSlave.rdata(L_G-1 downto 0) := r.txPolarity;
237  when 16#03# => -- ADDR (0xC)
238  v.axilReadSlave.rdata(L_G-1 downto 0) := r.loopback;
239  when 16#04# => -- ADDR (0x10)
240  v.axilReadSlave.rdata(6 downto 0) := r.commonCtrl;
241  when 16#05# => -- ADDR (0x14)
242  v.axilReadSlave.rdata := r.periodStep;
243  when 16#06# => -- ADDR (0x18)
244  v.axilReadSlave.rdata(F_G*8-1 downto 0) := r.negAmplitude;
245  when 16#07# => -- ADDR (0x1C)
246  v.axilReadSlave.rdata(F_G*8-1 downto 0) := r.posAmplitude;
247  when 16#08# => -- ADDR (0x20)
248  v.axilReadSlave.rdata(L_G-1 downto 0) := r.invertData;
249  when 16#10# to 16#1F# =>
250  for I in (L_G-1) downto 0 loop
251  if (axilReadMaster.araddr(5 downto 2) = I) then
252  v.axilReadSlave.rdata(TX_STAT_WIDTH_C-1 downto 0) := s_statusTxArr(I);
253  end if;
254  end loop;
255  when 16#20# to 16#2F# =>
256  for I in (L_G-1) downto 0 loop
257  if (axilReadMaster.araddr(5 downto 2) = I) then
258  v.axilReadSlave.rdata(7 downto 0) := r.signalSelectArr(I);
259  end if;
260  end loop;
261 
262  when 16#40# to 16#4F# =>
263  for I in (L_G-1) downto 0 loop
264  if (axilReadMaster.araddr(5 downto 2) = I) then
265  for J in 31 downto 0 loop
266  v.axilReadSlave.rdata(J) := s_statusCnt(I,J);
267  end loop;
268  end if;
269  end loop;
270  when 16#80# to 16#9F# =>
271  for I in (L_G-1) downto 0 loop
272  if (axilReadMaster.araddr(6 downto 2) = I) then
273  v.axilReadSlave.rdata(7 downto 0) := r.txDiffCtrl(I);
274  v.axilReadSlave.rdata(15 downto 8) := r.txPostCursor(I);
275  v.axilReadSlave.rdata(23 downto 16) := r.txPreCursor(I);
276  end if;
277  end loop;
278  when others =>
279  axilReadResp := AXI_ERROR_RESP_G;
280  end case;
281  axiSlaveReadResponse(v.axilReadSlave);
282  end if;
283 
284  -- Reset
285  if (axiRst_i = '1') then
286  v := REG_INIT_C;
287  end if;
288 
289  -- Register the variable for next clock cycle
290  rin <= v;
291 
292  -- Outputs
293  axilReadSlave <= r.axilReadSlave;
294  axilWriteSlave <= r.axilWriteSlave;
295  txDiffCtrl <= r.txDiffCtrl;
296  txPostCursor <= r.txPostCursor;
297  txPreCursor <= r.txPreCursor;
298  txPolarity <= r.txPolarity;
299  loopback <= r.loopback;
300 
301  end process comb;

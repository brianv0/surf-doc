7 -- Description:
8 -- Block for Register protocol.
9 -- Packet is a minimum of 4 x 32-bits
10 --
11 -- Incoming Request:
12 -- Word 0 Data[31:0] = TID[31:0] (echoed)
13 --
14 -- Word 1 Data[29:0] = Address[31:2] (only 26-bits if EN_32BIT_G = false)
15 -- Word 1 Data[31:30] = Opcode, 0x0=Read, 0x1=Write, 0x2=Set, 0x3=Clear
16 -- (bit set and bit clear not supported)
17 -- Word 2 Data[31:0] = WriteData[31:0] or ReadCount[8:0]
18 --
19 -- Word N-1 Data[31:0] = WriteData[31:0]
20 --
21 -- Word N Data[31:2] = Don't Care
22 --
23 -- Outgoing Response:
24 -- Word 0 Data[1:0] = VC (legacy, echoed)
25 -- Word 0 Data[7:2] = Dest_ID (legacy, echoed)
26 -- Word 0 Data[31:8] = TID[31:0] (legacy, echoed)
27 --
28 -- Word 1 Data[29:0] = Address[31:2]
29 -- Word 1 Data[31:30] = OpCode, 0x0=Read, 0x1=Write, 0x2=Set, 0x3=Clear
30 --
31 -- Word 2 Data[31:0] = ReadData[31:0]/WriteData[31:0]
32 --
33 -- Word N-1 Data[31:0] = ReadData[31:0]/WriteData[31:0]
34 --
35 -- Word N Data[31:18] = Don't Care
36 -- Word N Data[17] = Timeout Flag (response data)
37 -- Word N Data[16] = Fail Flag (response data)
38 -- Word N Data[15:00] = Don't Care

200  comb : process (axiLiteRst, mAxiLiteReadSlave, mAxiLiteWriteSlave, mFifoAxisCtrl, r, sFifoAxisMaster) is
201  variable v : RegType;
202  begin
203  v := r;
204 
205  -- Init
206  v.mFifoAxisMaster := sFifoAxisMaster;
207  v.mFifoAxisMaster.tUser := (others => '0');
208  v.mFifoAxisMaster.tKeep := (others => '1');
209  v.mFifoAxisMaster.tValid := '0';
210  v.mFifoAxisMaster.tLast := '0';
211 
212  v.sFifoAxisSlave.tReady := '0';
213 
214 
215  -- State machine
216  case r.state is
217 
218  -- Idle
219  when S_IDLE_C =>
220  v.mAxiLiteWriteMaster := AXI_LITE_WRITE_MASTER_INIT_C;
221  v.mAxiLiteReadMaster := AXI_LITE_READ_MASTER_INIT_C;
222  v.address := (others => '0');
223  v.rdSize := (others => '0');
224  v.rdCount := (others => '0');
225  v.timeout := '0';
226  v.fail := '0';
227 
228  -- Frame is starting
229  if sFifoAxisMaster.tValid = '1' and mFifoAxisCtrl.pause = '0' then
230  v.sFifoAxisSlave.tReady := '1';
231 
232  -- Bad frame
233  if sFifoAxisMaster.tLast = '0' then
234  v.mFifoAxisMaster.tValid := '1'; -- Echo word 0
235  v.mFifoAxisMaster.tUser := sFifoAxisMaster.tUser;
236  v.mFifoAxisMaster.tData := sFifoAxisMaster.tData;
237  v.state := S_ADDR_C;
238  end if;
239  end if;
240 
241  -- Address Field
242  when S_ADDR_C =>
243  v.sFifoAxisSlave.tReady := '1';
244 
245  if sFifoAxisMaster.tValid = '1' then
246 
247  if EN_32BIT_ADDR_G = true then
248  v.address(31 downto 26) := sFifoAxisMaster.tData(29 downto 24);
249  end if;
250 
251  v.address(25 downto 2) := sFifoAxisMaster.tData(23 downto 0);
252  v.mFifoAxisMaster.tValid := '1'; -- Echo word 1
253 
254  -- Short frame, return error
255  if sFifoAxisMaster.tLast = '1' then
256  v.fail := '1';
257  v.state := S_STATUS_C;
258 
259  -- Read
260  elsif sFifoAxisMaster.tData(31 downto 30) = "00" then
261  v.state := S_READ_SIZE_C;
262 
263  -- Write
264  elsif sFifoAxisMaster.tData(31 downto 30) = "01" then
265  v.state := S_WRITE_C;
266 
267  -- Not supported
268  else
269  v.fail := '1';
270  v.state := S_DUMP_C;
271  end if;
272  end if;
273 
274  -- Prepare Write Transaction
275  when S_WRITE_C =>
276  v.mAxiLiteWriteMaster.awaddr := r.address;
277  v.mAxiLiteWriteMaster.awprot := (others => '0');
278  v.mAxiLiteWriteMaster.wstrb := (others => '1');
279  v.mAxiLiteWriteMaster.wdata := sFifoAxisMaster.tData(31 downto 0);
280  v.sFifoAxisSlave.tReady := '1';
281  v.timer := (others => '1');
282 
283  if sFifoAxisMaster.tValid = '1' then
284  if sFifoAxisMaster.tLast = '1' then
285  -- check tkeep here
286  if (not axiStreamPacked(SLAVE_FIFO_SSI_CONFIG_C, sFifoAxisMaster)) then
287  v.fail := '1';
288  end if;
289  v.state := S_STATUS_C;
290  else
291  v.mFifoAxisMaster.tValid := '1'; -- Echo write data
292  v.mAxiLiteWriteMaster.awvalid := '1';
293  v.mAxiLiteWriteMaster.wvalid := '1';
294  v.mAxiLiteWriteMaster.bready := '1';
295  v.state := S_WRITE_AXI_C;
296  end if;
297  end if;
298 
299  -- Write Transaction, AXI
300  when S_WRITE_AXI_C =>
301  v.timer := r.timer - 1;
302 
303  -- Clear control signals on ack
304  if mAxiLiteWriteSlave.awready = '1' then
305  v.mAxiLiteWriteMaster.awvalid := '0';
306  end if;
307  if mAxiLiteWriteSlave.wready = '1' then
308  v.mAxiLiteWriteMaster.wvalid := '0';
309  end if;
310  if mAxiLiteWriteSlave.bvalid = '1' then
311  v.mAxiLiteWriteMaster.bready := '0';
312 
313  if mAxiLiteWriteSlave.bresp /= AXI_RESP_OK_C then
314  v.fail := '1';
315  end if;
316  end if;
317 
318  -- End transaction on timeout
319  if r.timer = 0 then
320  v.mAxiLiteWriteMaster.awvalid := '0';
321  v.mAxiLiteWriteMaster.wvalid := '0';
322  v.mAxiLiteWriteMaster.bready := '0';
323  v.timeout := '1';
324  end if;
325 
326  -- Transaction is done
327  if v.mAxiLiteWriteMaster.awvalid = '0' and
328  v.mAxiLiteWriteMaster.wvalid = '0' and
329  v.mAxiLiteWriteMaster.bready = '0' then
330 
331  v.address := r.address + 4;
332  v.state := S_WRITE_C;
333  end if;
334 
335  -- Read size
336  when S_READ_SIZE_C =>
337  v.rdCount := (others => '0');
338  v.rdSize := sFifoAxisMaster.tData(8 downto 0);
339 
340  -- Don't read if EOF (need for dump later)
341  if sFifoAxisMaster.tValid = '1' then
342  v.sFifoAxisSlave.tReady := not sFifoAxisMaster.tLast;
343  v.state := S_READ_C;
344  end if;
345 
346  -- Read transaction
347  when S_READ_C =>
348  v.mAxiLiteReadMaster.araddr := r.address;
349  v.mAxiLiteReadMaster.arprot := (others => '0');
350  v.timer := (others => '1');
351 
352  -- Start AXI transaction
353  v.mAxiLiteReadMaster.arvalid := '1';
354  v.mAxiLiteReadMaster.rready := '1';
355  v.state := S_READ_AXI_C;
356 
357  -- Read AXI
358  when S_READ_AXI_C =>
359  v.timer := r.timer - 1;
360 
361  -- Clear control signals on ack
362  if mAxiLiteReadSlave.arready = '1' then
363  v.mAxiLiteReadMaster.arvalid := '0';
364  end if;
365  if mAxiLiteReadSlave.rvalid = '1' then
366  v.mAxiLiteReadMaster.rready := '0';
367  v.mFifoAxisMaster.tData(31 downto 0) := mAxiLiteReadSlave.rdata;
368 
369  if mAxiLiteReadSlave.rresp /= AXI_RESP_OK_C then
370  v.fail := '1';
371  end if;
372  end if;
373 
374  -- End transaction on timeout
375  if r.timer = 0 then
376  v.mAxiLiteReadMaster.arvalid := '0';
377  v.mAxiLiteReadMaster.rready := '0';
378  v.timeout := '1';
379  end if;
380 
381  -- Transaction is done
382  if v.mAxiLiteReadMaster.arvalid = '0' and v.mAxiLiteReadMaster.rready = '0' then
383  v.mFifoAxisMaster.tValid := '1';
384  v.address := r.address + 4;
385  v.rdCount := r.rdCount + 1;
386 
387  if r.rdCount = r.rdSize then
388  v.state := S_DUMP_C;
389  else
390  v.state := S_READ_C;
391  end if;
392  end if;
393 
394  -- Dump until EOF
395  when S_DUMP_C =>
396  v.sFifoAxisSlave.tReady := '1';
397 
398  if sFifoAxisMaster.tValid = '1' and sFifoAxisMaster.tLast = '1' then
399  -- Check tKeep here
400  if (not axiStreamPacked(SLAVE_FIFO_SSI_CONFIG_C, sFifoAxisMaster)) then
401  v.fail := '1';
402  end if;
403  v.state := S_STATUS_C;
404  end if;
405 
406  -- Send Status
407  when S_STATUS_C =>
408  v.mFifoAxisMaster.tValid := '1';
409  v.mFifoAxisMaster.tLast := '1';
410  v.mFifoAxisMaster.tData(63 downto 0) := (others => '0');
411  v.mFifoAxisMaster.tData(17) := r.timeout;
412  v.mFifoAxisMaster.tData(16) := r.fail;
413  v.state := S_IDLE_C;
414 
415  when others =>
416  v.state := S_IDLE_C;
417 
418  end case;
419 
420  if (axiLiteRst = '1') then
421  v := REG_INIT_C;
422  end if;
423 
424  rin <= v;
425 
426  mAxiLiteWriteMaster <= r.mAxiLiteWriteMaster;
427  mAxiLiteReadMaster <= r.mAxiLiteReadMaster;
428  sFifoAxisSlave <= v.sFifoAxisSlave;
429  mFifoAxisMaster <= r.mFifoAxisMaster;
430 
431  end process comb;
432

434  begin
435  if (rising_edge(axiLiteClk)) then
436  r <= rin after TPD_G;
437  end if;
438  end process seq;

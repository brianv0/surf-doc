103  comb : process (r, rst_i, enable_i, init_i, data_i, strobe_i, length_i, s_dataWordSum) is
104  variable v : RegType;
105  begin
106  v := r;
107 
108  -- Cumulative sum of the data_i while enabled
109  if ( enable_i = '0') then
110  v.sum := ("00000" & init_i);
111  v.lenCnt := 0;
112  v.valid := '0';
113  elsif ( r.lenCnt >= length_i) then
114  v.sum := r.sum;
115  v.lenCnt := r.lenCnt;
116  v.valid := '1';
117  elsif ( strobe_i = '1') then
118  -- Add new word sum
119  v.sum := r.sum + s_dataWordSum;
120  v.lenCnt := r.lenCnt +1;
121  v.valid := '0';
122  else
123  v.sum := r.sum;
124  v.lenCnt := r.lenCnt;
125  v.valid := '0';
126  end if;
127 
128  -- Add the sum carry bits
129  v.chksum := '0' & r.sum(CSUM_WIDTH_G-1 downto 0) + r.sum(CSUM_WIDTH_G+4 downto CSUM_WIDTH_G);
130  -- Add the checksum carry bit
131  v.chksum(CSUM_WIDTH_G-1 downto 0) := v.chksum(CSUM_WIDTH_G-1 downto 0) + v.chksum(CSUM_WIDTH_G);
132 
133  -- Checksum output (calculated with 2 c-c delay towards data)
134  -- Ones complement
135  chksum_o <= not r.chksum(CSUM_WIDTH_G-1 downto 0);
136 
137  if (rst_i = '1') then
138  v := REG_INIT_C;
139  end if;
140 
141  rin <= v;
142  -----------------------------------------------------------
143  end process comb;
144 
145  seq : process (clk_i) is
146  begin
147  if (rising_edge(clk_i)) then
148  r <= rin after TPD_G;
149  end if;
150  end process seq;
151  ---------------------------------------------------------------------
152  -- Output assignment
153  valid_o <= r.valid;
154  check_o <= '1' when (not r.chksum(CSUM_WIDTH_G-1 downto 0)) = (r.chksum'range => '0') else '0';
155  ---------------------------------------------------------------------
156 end architecture rtl;

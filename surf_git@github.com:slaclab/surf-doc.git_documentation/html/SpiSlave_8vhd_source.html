114  seq : process (clk) is

123  comb : process (mosiSync, r, rdData, rdStb, rst, sclkSync, selLSync) is
124  variable v : RegType;
125 
126  impure function isLeadingEdge
127  return boolean is
128  begin
129  return selLSync = '0' and r.sclkLast = CPOL_G and sclkSync = not CPOL_G;
130  end function;
131 
132  impure function isTrailingEdge
133  return boolean is
134  begin
135  return selLSync = '0' and r.sclkLast = not CPOL_G and sclkSync = CPOL_G;
136  end function;
137 
138  -- Shift a new bit out
139  procedure shift is
140  begin
141  v.shiftReg := r.shiftReg(WORD_SIZE_G-1 downto 0) & '0';
142  end procedure;
143 
144  -- Clock in the current mosi bit and increment counter
145  procedure sample is
146  begin
147  v.shiftReg(0) := mosiSync;
148 
149  v.shiftCnt := r.shiftCnt + 1;
150  if (r.shiftCnt = MAX_COUNT_C) then
151  v.shiftCnt := (others => '0');
152  end if;
153  end procedure;
154 
155  begin
156  v := r;
157 
158  v.sclkLast := sclkSync;
159  v.mosiLast := mosiSync;
160  v.selLLast := selLSync;
161 
162  -- CPHA_G = 0 is a special case
163  -- Do a shift on falling edge of selL
164  if (CPHA_G = '0' and r.selLLast = '1' and selLSync = '0') then
165  shift;
166  end if;
167 
168  if (isLeadingEdge) then
169  v.gotSclk := '1';
170  if (CPHA_G = '1') then
171  shift;
172  elsif (CPHA_G = '0') then
173  sample;
174  end if;
175  end if;
176 
177  if (isTrailingEdge) then
178  if (CPHA_G = '1') then
179  sample;
180  elsif (CPHA_G = '0' and r.shiftCnt /= MAX_COUNT_C) then
181  shift;
182  end if;
183  end if;
184 
185  -- Assert wrStb when max count reached
186  if (r.shiftCnt = MAX_COUNT_C and isTrailingEdge and r.gotSclk = '1') then
187  v.wrStb := '1';
188  end if;
189 
190  -- Read strobe only allowed when Write strobe is high
191  if (r.wrStb = '1' and (rdStb = '1' or isLeadingEdge)) then
192  v.wrStb := '0';
193  v.shiftReg(WORD_SIZE_G-1 downto 0) := rdData;
194  end if;
195 
196  if (rst = '1') then
197  v := REG_INIT_C;
198  end if;
199 
200  rin <= v;
201 
202  -- Outputs
203  miso <= 'Z';
204  if (selLSync = '0') then
205  miso <= r.shiftReg(WORD_SIZE_G);
206  end if;
207 
208  wrData <= r.shiftReg(WORD_SIZE_G-1 downto 0);
209  wrStb <= r.wrStb;
210 
211  end process;
212

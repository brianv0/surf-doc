7 -- Description:
8 -- Block for Command protocol over the VC.
9 -- The receive packet is 4 x 32-bits.
10 -- Word 0 Data[1:0] = VC (unused, legacy)
11 -- Word 0 Data[7:2] = Dest_ID (unused, legacy)
12 -- Word 0 Data[31:8] = CmdCtx[31:0] (unused, legacy)
13 -- Word 1 Data[7:0] = OpCode[7:0]
14 -- Word 1 Data[31:8] = Don't Care
15 -- Word 2 = Don't Care
16 -- Word 3 = Don't Care

135  comb : process (cmdRst, fifoAxisMaster, r) is
136  variable v : RegType;
137  begin
138  v := r;
139 
140  -- Init, always read
141  v.cmdMaster.valid := '0';
142 
143 
144  if (fifoAxisMaster.tValid = '1') then
145  v.txnNumber := r.txnNumber + 1;
146 
147  case r.txnNumber is
148  when "000" =>
149  v.cmdMaster.context := fifoAxisMaster.tData(31 downto 8);
150  when "001" =>
151  v.cmdMaster.opCode := fifoAxisMaster.tData(7 downto 0);
152  when "011" =>
153  v.cmdMaster.valid := fifoAxisMaster.tLast and not ssiGetUserEofe(INT_CONFIG_C, fifoAxisMaster);
154  when "100" =>
155  -- Too many txns in frame, freeze counting
156  -- Will auto reset txnNumber to 0 on tLast
157  v.txnNumber := r.txnNumber;
158  when others => null;
159  end case;
160 
161 
162  --if (not axiStreamPacked(INT_CONFIG_C, fifoAxisMaster)) then
163  -- -- Fail frame if any txn is not packed
164  -- v.txnNumber := "100";
165  -- v.cmdMaster.valid := '0';
166  --end if;
167 
168  if (fifoAxisMaster.tLast = '1') then
169  -- Reset frame on tLast or EOFE
170  v.txnNumber := "000";
171  end if;
172 
173  end if;
174 
175  if (cmdRst = '1') then
176  v := REG_INIT_C;
177  end if;
178 
179  rin <= v;
180 
181  cmdMaster <= r.cmdMaster;
182 
183  end process comb;
184 
185  seq : process (cmdClk) is
186  begin

188  r <= rin after TPD_G;
189  end if;
190  end process seq;

21 -- File: i2cslv.vhd
22 -- Author: Jan Andersson - Gaisler Research
23 -- jan@gaisler.com
24 --
25 -- Documentation of generics:
26 --
27 -- [TENBIT_G]
28 -- Support for ten bit addresses.
29 --
30 -- [I2C_ADDR_G]
31 -- The slave's i2c address.
32 --
33 -- [OUTPUT_EN_POLARITY_G]
34 -- Output enable polarity
35 --
36 -- [FILTER_G]
37 -- Length of filters used on SCL and SDA.
38 -- This generic should specify, in number of system clock cycles plus one,
39 -- the time of the shortest pulse on the I2C bus to be registered as a valid
40 -- value. For instance, to disregard any pulse that is 50 ns or shorter in
41 -- a system with a system frequency of 54 MHz this generic should be set to:
42 -- ((pulse time) / (clock period)) + 1 = (50 ns) / ((1/(54 MHz)) + 1 = 3.7
43 -- The value from this calculation should always be rounded up.
44 -- In other words an appropriate filter length for a 54 MHz system is 4.
45 --
46 -- The slave has four different modes operation. The mode is defined by the
47 -- value of the bits RMODE and TMODE.
48 -- RMODE TMODE I2CSLAVE Mode
49 -- 0 0 0
50 -- 0 1 1
51 -- 1 0 2
52 -- 1 1 3
53 --
54 -- RMODE_G 0:
55 -- The slave accepts one byte and NAKs all other transfers until software has
56 -- acknowledged the received byte.
57 -- RMODE_G 1:
58 -- The slave accepts one byte and keeps SCL low until software has acknowledged
59 -- the received byte
60 -- TMODE_G 0:
61 -- The slave transmits the same byte to all if the master requests more than
62 -- one byte in the transfer. The slave then NAKs all read requests unless the
63 -- Transmit Always Valid (TAV) bit in the control register is set.
64 -- TMODE_G 1:
65 -- The slave transmits one byte and then keeps SCL low until software has
66 -- acknowledged that the byte has been transmitted.

104 architecture rtl of I2cSlave is
105  -----------------------------------------------------------------------------
106  -- Constants
107  -----------------------------------------------------------------------------
108  -- Core configuration
109  constant I2C_ADDR_LEN_C : integer := 7 + TENBIT_G*3;
110  constant I2C_SLAVE_ADDR_C : std_logic_vector((I2C_ADDR_LEN_C-1) downto 0) :=
111  conv_std_logic_vector(I2C_ADDR_G, I2C_ADDR_LEN_C);
112 
113  -- Misc constants
114  constant I2C_READ_C : std_ulogic := '1'; -- R/Wn bit
115  constant I2C_WRITE_C : std_ulogic := '0';
116 
117  constant OEPOL_LEVEL_C : std_ulogic := conv_std_logic(OUTPUT_EN_POLARITY_G = 1);
118 
119  constant I2C_LOW_C : std_ulogic := OEPOL_LEVEL_C; -- OE
120  constant I2C_HIZ_C : std_ulogic := not OEPOL_LEVEL_C;
121 
122  constant I2C_ACK_C : std_ulogic := '0';
123 
124  constant TENBIT_ADDR_START_C : std_logic_vector(4 downto 0) := "11110";
125 
126  -----------------------------------------------------------------------------
127  -- Types
128  -----------------------------------------------------------------------------
129  type i2c_in_array is array (FILTER_G downto 0) of i2c_in_type;
130 
131  type slv_state_type is (idle, checkaddr, check10bitaddr, sclhold,
132  movebyte, handshake);

134  type i2cslv_reg_type is record
135  slvstate : slv_state_type;
136  -- Transfer phase
137  active : boolean;
138  addr : boolean;
139 -- transmit : boolean;
140 -- receive : boolean;
141  -- Shift register
142  sreg : std_logic_vector(7 downto 0);
143  cnt : std_logic_vector(2 downto 0);
144  -- Synchronizers for inputs SCL and SDA
145  scl : std_ulogic;
146  sda : std_ulogic;
147  i2ci : i2c_in_array;
148  -- Output enables

155  constant REG_INIT_C : i2cslv_reg_type := (
156  slvstate => idle,
157  active => false,
158  addr => false,
159  sreg => (others => '0'),
160  cnt => (others => '0'),
161  scl => '0',
162  sda => '0',
163  i2ci => (others => (scl => '0', sda => '0')),
164  scloen => I2C_HIZ_C,
165  sdaoen => I2C_HIZ_C,
166  o => I2C_SLAVE_OUT_INIT_C);

168  -----------------------------------------------------------------------------
169  -- Subprograms
170  -----------------------------------------------------------------------------
171  -- purpose: Compares the first byte of a received address with the slave's
172  -- address. The tba input determines if the slave is using a ten bit address.
173  function compaddr1stb (
174  ibyte : std_logic_vector(7 downto 0)) -- I2C byte
175  return boolean
176  is
177  variable correct : std_logic_vector(7 downto 1);
178  begin -- compaddr1stb
179  if TENBIT_G = 1 then
180  correct(7 downto 3) := TENBIT_ADDR_START_C;
181  correct(2 downto 1) := I2C_SLAVE_ADDR_C((I2C_ADDR_LEN_C-1) downto (I2C_ADDR_LEN_C-2));
182  else
183  correct(7 downto 1) := I2C_SLAVE_ADDR_C(6 downto 0);
184  end if;
185  return ibyte(7 downto 1) = correct(7 downto 1);
186  end compaddr1stb;
187 
188  -- purpose: Compares the 2nd byte of a ten bit address with the slave address
189  function compaddr2ndb (
190  ibyte : std_logic_vector(7 downto 0)) -- I2C byte
191  return boolean is
192  begin -- compaddr2ndb
193  return ibyte((I2C_ADDR_LEN_C-3) downto 0) = I2C_SLAVE_ADDR_C((I2C_ADDR_LEN_C-3) downto 0);
194  end compaddr2ndb;
195 
196  -----------------------------------------------------------------------------
197  -- Signals
198  -----------------------------------------------------------------------------
199 
200  -- Register interface
201  signal r : i2cslv_reg_type := REG_INIT_C;
202  signal rin : i2cslv_reg_type;
203 
204 begin
205 
206  comb : process (r, sRst, i2ci, i2cSlaveIn)
207  variable v : i2cslv_reg_type;
208  variable sclfilt : std_logic_vector(FILTER_G-1 downto 0);
209  variable sdafilt : std_logic_vector(FILTER_G-1 downto 0);
210  begin -- process comb
211  v := r;
212 
213  v.i2ci(0) := i2ci; v.i2ci(FILTER_G downto 1) := r.i2ci(FILTER_G-1 downto 0);
214 
215  ----------------------------------------------------------------------------
216  -- Bus filtering
217  ----------------------------------------------------------------------------
218  for i in 0 to FILTER_G-1 loop
219  sclfilt(i) := r.i2ci(i+1).scl; sdafilt(i) := r.i2ci(i+1).sda;
220  end loop; -- i
221  if andv(sclfilt) = '1' then v.scl := '1'; end if;
222  if orv(sclfilt) = '0' then v.scl := '0'; end if;
223  if andv(sdafilt) = '1' then v.sda := '1'; end if;
224  if orv(sdafilt) = '0' then v.sda := '0'; end if;
225 
226  -- txAck pulsed for 1 clock only when set by state machine below.
227  v.o.txAck := '0';
228 
229  -- Reset rxValid when ack'd from IO
230  if (r.o.rxValid = '1' and i2cSlaveIn.rxAck = '1') then
231  v.o.rxValid := '0';
232  end if;
233 
234  ---------------------------------------------------------------------------
235  -- I2C slave control FSM
236  ---------------------------------------------------------------------------
237  case r.slvstate is
238  when idle =>
239  -- Release bus
240  if (r.scl and not v.scl) = '1' then
241  v.sdaoen := I2C_HIZ_C;
242  end if;
243 
244  when checkaddr =>
245  if compaddr1stb(r.sreg) then
246  if r.sreg(0) = I2C_READ_C then
247  if (TENBIT_G = 0 or (TENBIT_G = 1 and r.active)) then
248  if i2cSlaveIn.txValid = '1' then
249  -- Transmit data
250  v.o.txActive := '1';
251  v.slvstate := handshake;
252  else
253  -- No data to transmit, NAK
254  v.o.nack := '1';
255  v.slvstate := idle;
256  end if;
257  else
258  -- Ten bit address with R/Wn = 1 and slave not previously
259  -- addressed.
260  v.slvstate := idle;
261  end if;
262  else
263  v.o.rxActive := toSl(TENBIT_G = 0);
264  v.slvstate := handshake;
265  end if;
266  else
267  -- Slave address did not match
268  v.active := false;
269  v.slvstate := idle;
270  end if;
271  v.sreg := i2cSlaveIn.txData;
272 
273  when check10bitaddr =>
274  if compaddr2ndb(r.sreg) then
275  -- Slave has been addressed with a matching 10 bit address
276  -- If we receive a repeated start condition, matching address
277  -- and R/Wn = 1 we will transmit data. Without start condition we
278  -- will receive data.
279  v.addr := true;
280  v.active := true;
281  v.o.rxActive := '1';
282  v.slvstate := handshake;
283  else
284  v.slvstate := idle;
285  end if;
286 
287  when sclhold =>
288  -- This state is used when the device has been addressed to see if SCL
289  -- should be kept low until the receive register is free or the
290  -- transmit register is filled. It is also used when a data byte has
291  -- been transmitted or received to SCL low until software acknowledges
292  -- the transfer.
293  if (r.scl and not v.scl) = '1' then
294  v.scloen := I2C_LOW_C;
295  v.sdaoen := I2C_HIZ_C;
296  end if;
297  -- Ack has happened and rxValid set back to '0'
298  if ((r.o.rxActive = '1' and (r.o.rxValid = '0' or RMODE_G = 0)) or
299  (r.o.txActive = '1' and (i2cSlaveIn.txValid = '1' or TMODE_G = 0))) then
300  v.slvstate := movebyte;
301  v.scloen := I2C_HIZ_C;
302  -- Falling edge that should be detected in movebyte may have passed
303  if r.o.txActive = '1' and v.scl = '0' then
304  v.sdaoen := r.sreg(7) xor OEPOL_LEVEL_C;
305  end if;
306  end if;
307  v.sreg := i2cSlaveIn.txData;
308 
309  when movebyte =>
310  if (r.scl and not v.scl) = '1' then
311  if r.o.txActive = '1' then
312  v.sdaoen := r.sreg(7) xor OEPOL_LEVEL_C;
313  else
314  v.sdaoen := I2C_HIZ_C;
315  end if;
316  end if;
317  if (not r.scl and v.scl) = '1' then
318  v.sreg := r.sreg(6 downto 0) & r.sda;
319  if r.cnt = "111" then
320  if r.addr then
321  v.slvstate := checkaddr;
322  elsif r.o.rxActive = '1' nor r.o.txActive = '1' then
323  v.slvstate := check10bitaddr;
324  else
325  v.slvstate := handshake;
326  end if;
327  v.cnt := (others => '0');
328  else
329  v.cnt := r.cnt + 1;
330  end if;
331  end if;
332 
333  when handshake =>
334  -- Falling edge
335  if (r.scl and not v.scl) = '1' then
336  if r.addr then
337  v.sdaoen := I2C_LOW_C;
338  elsif r.o.rxActive = '1' then
339  -- Receive, send ACK/NAK
340  -- Acknowledge byte if core has room in receive register
341  -- This code assumes that the core's receive register is free if we are
342  -- in RMODE 1. This should always be the case unless software has
343  -- reconfigured the core during operation.
344  if r.o.rxValid = '0' then
345  v.sdaoen := I2C_LOW_C;
346  v.o.rxData := r.sreg;
347  v.o.rxValid := '1';
348  else
349  -- NAK the byte, the master must abort the transfer
350  v.sdaoen := I2C_HIZ_C;
351  v.slvstate := idle;
352  end if;
353  else
354  -- Transmit, release bus
355  v.sdaoen := I2C_HIZ_C;
356  -- Byte transmitted, ack it.
357  v.o.txAck := '1';
358  end if;
359  if not r.addr and r.o.rxActive = '1' and v.sdaoen = I2C_HIZ_C then
360  v.o.nack := '1';
361  end if;
362  end if;
363  -- Risinge edge
364  if (not r.scl and v.scl) = '1' then
365  if r.addr then
366  v.slvstate := movebyte;
367  else
368  if r.o.rxActive = '1' then
369  -- RMODE 0: Be ready to accept one more byte which will be NAK'd if
370  -- software has not read the receive register
371  -- RMODE 1: Keep SCL low until software has acknowledged received byte
372  if RMODE_G = 0 then
373  v.slvstate := movebyte;
374  else
375  v.slvstate := sclhold;
376  end if;
377  else
378  -- Transmit, check ACK/NAK from master
379  -- If the master NAKs the transmitted byte the transfer has ended and
380  -- we should wait for the master's next action. If the master ACKs the
381  -- byte the core will act depending on tmode:
382  -- TMODE 0:
383  -- If the master ACKs the byte we must continue to transmit and will
384  -- transmit the same byte on all requests.
385  -- TMODE 1:
386  -- IF the master ACKs the byte we will keep SCL low until software has
387  -- put new transmit data into the transmit register.
388  if r.sda = I2C_ACK_C then
389  if TMODE_G = 0 then
390  v.slvstate := movebyte;
391  else
392  v.slvstate := sclhold;
393  end if;
394  else
395  v.slvstate := idle;
396  end if;
397  end if;
398  end if;
399  v.addr := false;
400  v.sreg := i2cSlaveIn.txData;
401  end if;
402  end case;
403 
404  if i2cSlaveIn.enable = '1' then
405  -- STOP condition
406  if (r.scl and v.scl and not r.sda and v.sda) = '1' then
407  v.active := false;
408  v.slvstate := idle;
409  v.o.txActive := '0';
410  v.o.rxActive := '0';
411  end if;
412 
413  -- START or repeated START condition
414  if (r.scl and v.scl and r.sda and not v.sda) = '1' then
415  v.slvstate := movebyte;
416  v.cnt := (others => '0');
417  v.addr := true;
418  v.o.txActive := '0';
419  v.o.rxActive := '0';
420  end if;
421  end if;
422 
423  ----------------------------------------------------------------------------
424  -- Reset and idle operation
425  ----------------------------------------------------------------------------
426 
427  if (sRst = '1') then
428  v.slvstate := idle;
429  v.scl := '0';
430  v.active := false;
431  v.scloen := I2C_HIZ_C;
432  v.sdaoen := I2C_HIZ_C;
433  v.sreg := (others => '0');
434  v.cnt := (others => '0');
435  v.o.rxActive := '0';
436  v.o.rxValid := '0';
437  v.o.rxData := (others => '0');
438  v.o.txActive := '0';
439  v.o.txAck := '0';
440  v.o.nack := '0';
441 
442  end if;
443 
444  ----------------------------------------------------------------------------
445  -- Signal assignments
446  ----------------------------------------------------------------------------
447 
448  -- Update registers
449  rin <= v;
450 
451  -- Update outputs
452  i2cSlaveOut <= r.o;
453  i2co.scl <= '0';
454  i2co.scloen <= r.scloen;
455  i2co.sda <= '0';
456  i2co.sdaoen <= r.sdaoen;
457  i2co.enable <= i2cSlaveIn.enable;
458  end process comb;
459 
460  reg : process (clk, aRst)
461  begin -- process reg
462  if (aRst = '1') then
463  r.slvstate <= idle after TPD_G;
464  r.scl <= '0' after TPD_G;
465  r.active <= false after TPD_G;
466  r.scloen <= I2C_HIZ_C after TPD_G;
467  r.sdaoen <= I2C_HIZ_C after TPD_G;
468  r.sreg <= (others => '0') after TPD_G;
469  r.cnt <= (others => '0') after TPD_G;
470  r.o.rxActive <= '0' after TPD_G;
471  r.o.rxValid <= '0' after TPD_G;
472  r.o.rxData <= (others => '0') after TPD_G;
473  r.o.txActive <= '0' after TPD_G;
474  r.o.txAck <= '0' after TPD_G;
475  r.o.nack <= '0' after TPD_G;
476  elsif rising_edge(clk) then
477  r <= rin after TPD_G;
478  end if;
479  end process reg;

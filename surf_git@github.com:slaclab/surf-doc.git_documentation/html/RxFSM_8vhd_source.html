7 -- Description: Receiver FSM
8 -- Receiver has the following functionality:
9 -- Transport side FSM. Receive check and save segments to RX buffer.
10 -- - WAIT_SOF Waits for Transport side SOF,
11 -- - CHECK Determines the segment type and checks:
12 -- ACK, NULL, DATA, or RST segment
13 -- 1. Validates checksum (when valid),
14 -- 2. Header length (number of bytes),
15 -- 3. Sequence number (Only current seqN or lastSeqN+1 allowed)
16 -- 4. Acknowledgment number (Valid range is lastAckN to lastAckN + txWindowSize)
17 -- - CHECK_SYN Toggles through SYN header addresses and saves the RSSI parameters
18 -- Checks the following:
19 -- 1. Validates checksum (when valid),
20 -- 2. Validates Ack number if the ack is sent with the SYN segment
21 -- - DATA Receives the payload part of the DATA segment
22 -- - VALID Checks if next valid SEQn is received. If yes:
23 -- 1. increment the in order SEQn
24 -- 2. save seqN, type, and occupied to the window buffer at current rxBufferAddr
25 -- 3. increment rxBufferAddr
26 -- - DROP Just report dropped packet and got back to WAIT_SOF
27 -- Receiver side FSM. Send data to App side.
28 -- - CHECK_BUFFER and DATA Send the data frame to the Application
29 -- when the data at the next txSegmentAddr is ready.
30 -- - SENT Release the windowbuffer at txBufferAddr.
31 -- Increment txBufferAddr.
32 -- Register the received SeqN for acknowledgment.

261  signal r : RegType := REG_INIT_C;
262  signal rin : RegType;
263  signal s_chksumOk : sl;

265 begin
266 
267  -- Override checksum check if checksum disabled
268  s_chksumOk <= ite(HEADER_CHKSUM_EN_G, chksumOk_i, '1');

270  -----------------------------------------------------------------------------------------------
271  comb : process (r, rst_i, chksumValid_i, s_chksumOk, rxWindowSize_i, lastAckN_i, rxBufferSize_i,
272  txWindowSize_i, tspSsiMaster_i, connActive_i, rdBuffData_i, appSsiSlave_i) is
273 
274  variable v : RegType;
275 
276  begin
277  v := r;
278 
279  ------------------------------------------------------------
280  -- RX Transport side FSM:
281  -- Receive the segment from the peer
282  -- Check the segment:
283  -- - register the parameters from SYN header
284  -- - seqN, ackN
285  -- - check header checksum
286  -- - increment in order received SeqN
287  ------------------------------------------------------------
288 
289  -- Pipeline the transport master
290  v.tspSsiMaster := tspSsiMaster_i;
291  v.headerData := endianSwap64(tspSsiMaster_i.data(63 downto 0));
292  case r.tspState is
293  ----------------------------------------------------------------------
294  when WAIT_SOF_S =>
295 
296  -- Counters to 0
297  v.rxHeaderAddr := (others => '0');
298  v.rxSegmentAddr := (others => '1'); -- "-1" so the first address after increment to be 0
299  v.segmentWe := '0';
300 
301  -- Ready until SOF received
302  -- Also flush any dropped or non SOF segments
303  v.tspSsiSlave := SSI_SLAVE_RDY_C;
304 
305  -- Checksum commands
306  v.chkEn := '0';
307  v.chkStb := '0';
308  v.chkLen := 1;
309  --
310  v.segValid := '0';
311  v.segDrop := '0';
312 
313  -- Next state condition
314  if (tspSsiMaster_i.sof = '1' and tspSsiMaster_i.valid = '1') then
315  v.chkEn := '1';
316  v.chkStb := '1';
317 
318  -- When SOF has been received dessert ready until package is checked
319  v.tspSsiSlave := SSI_SLAVE_RDY_C;
320 
321  -- If the packet is longer than one set the data flag
322  if (tspSsiMaster_i.eof = '1') then
323  v.rxF.data := '0';
324  else
325  v.rxF.data := '1';
326  end if;
327  --
328  v.tspState := CHECK_S;
329  --
330  end if;
331  ----------------------------------------------------------------------
332  when CHECK_S =>
333  --
334  v.segValid := '0';
335  v.segDrop := '0';
336  v.rxSegmentAddr := (others => '1');
337 
338  -- Hold incoming AXI stream
339  v.tspSsiSlave := SSI_SLAVE_NOTRDY_C;
340 
341  if (r.tspSsiMaster.valid = '1' and r.tspSsiMaster.sof = '1') then
342  -- Register flags, header length and SEQn
343  v.rxF.syn := r.headerData (63);
344  v.rxF.ack := r.headerData (62);
345  v.rxF.eack := r.headerData (61);
346  v.rxF.rst := r.headerData (60);
347  v.rxF.nul := r.headerData (59);
348  v.rxF.busy := r.headerData (56);
349 
350  v.rxHeadLen := r.headerData (55 downto 48);
351  v.rxSeqN := r.headerData (47 downto 40);
352  v.rxAckN := r.headerData (39 downto 32);
353  end if;
354 
355  -- Checksum commands
356  v.chkEn := '1';
357  v.chkStb := '0';
358 
359  -- Syn header received (header is 3 c-c long)
360  if (v.rxF.syn = '1') then
361 
362  -- Register SYN header word 0 parameters
363  v.chkLen := 3; -- TODO make generic
364  v.rxParam.version := r.headerData (31 downto 28);
365  v.rxParam.chksumEn := r.headerData (26 downto 26);
366  v.rxParam.maxOutsSeg := r.headerData (23 downto 16);
367  v.rxParam.maxSegSize := r.headerData (15 downto 0);
368 
369  --
370  if (v.rxF.ack = '1' and v.rxAckN /= lastAckN_i) then
371  -- Acknowledgment not valid
372  v.tspState := DROP_S;
373  elsif (v.rxF.eack = '1' or v.rxF.rst = '1' or v.rxF.busy = '1') then
374  -- Wrong flags
375  v.tspState := DROP_S;
376  else
377  -- Go to SYN_CHECK_S
378  v.tspState := SYN_CHECK_S;
379  end if;
380 
381  -- Segment is ACK, DATA, RST, or NULL
382  elsif (v.rxF.syn = '0' and v.rxF.eack = '0' and chksumValid_i = '1') then --
383 
384  -- Check header
385  if (
386  -- Checksum
387  s_chksumOk = '1' and
388  -- Check length
389  r.rxHeadLen = toSlv(8, 8) and
390  -- Check SeqN range
391  (r.rxSeqN - r.inOrderSeqN) <= 1 and
392  -- Check AckN range
393  (r.rxAckN - lastAckN_i) <= txWindowSize_i
394  ) then
395  -- Valid data segment
396  if (r.rxF.data = '1' and v.rxF.nul = '0' and v.rxF.rst = '0') then
397  -- Wait if the buffer full
398  -- Note: Deadlock possibility! If the peer is not accepting data!
399  if (r.windowArray(conv_integer(r.rxBufferAddr)).occupied = '0') then
400  -- Go to data segment
401  v.tspState := DATA_S;
402  else
403  -- Buffer is full -> drop segment
404  v.tspState := DROP_S;
405  end if;
406  elsif (r.rxF.data = '0') then
407  -- Valid non data segment
408  v.tspState := VALID_S;
409  else
410  -- Error: Data is attached to NUL or RST segment
411  v.tspState := DROP_S;
412  end if;
413  else
414  -- Header not valid
415  v.tspState := DROP_S;
416  end if;
417  end if;
418 
419  ----------------------------------------------------------------------
420  when SYN_CHECK_S =>
421  --
422  v.segValid := '0';
423  v.segDrop := '0';
424  v.rxSegmentAddr := (others => '1');
425 
426  -- Ready to receive further header data
427  v.tspSsiSlave := SSI_SLAVE_RDY_C;
428 
429  -- Get the rest of the SYN header
430  if (tspSsiMaster_i.valid = '1') then
431  v.chkStb := '1';
432  v.rxHeaderAddr := r.rxHeaderAddr + 1;
433  else
434  v.chkStb := '0';
435  v.rxHeaderAddr := r.rxHeaderAddr;
436  end if;
437 
438  -- Register SYN header word 1 parameters
439  if (r.rxHeaderAddr = x"01" and r.tspSsiMaster.valid = '1') then
440  -- Syn parameters
441  v.rxParam.retransTout := r.headerData (63 downto 48);
442  v.rxParam.cumulAckTout:= r.headerData (47 downto 32);
443  v.rxParam.nullSegTout := r.headerData (31 downto 16);
444  v.rxParam.maxRetrans := r.headerData (15 downto 8);
445  v.rxParam.maxCumAck := r.headerData ( 7 downto 0);
446  --
447  end if;
448 
449  -- Register SYN header word 2 parameters
450  if (r.rxHeaderAddr = x"02" ) then
451  v.chkStb := '0';
452  v.rxHeaderAddr := r.rxHeaderAddr;
453  v.tspSsiSlave := r.tspSsiSlave;
454 
455  if (r.tspSsiMaster.valid = '1') then
456 
457  -- Syn parameters
458  v.rxParam.maxOutofseq := r.headerData (63 downto 56);
459  v.rxParam.timeoutUnit := r.headerData (55 downto 48);
460  v.rxParam.connectionId(31 downto 0):= r.headerData (47 downto 16);
461 
462  -- Tsp parameters
463  v.tspSsiSlave := SSI_SLAVE_NOTRDY_C;
464  end if;
465 
466  -- Wait for checksum
467  if ( chksumValid_i = '1') then
468  -- Check received data header
469 
470  if (
471  -- Checksum
472  s_chksumOk = '1' and
473  -- Check length
474  r.rxHeadLen = toSlv(24, 8)
475  ) then
476  -- Header is valid
477  v.tspState := VALID_S;
478  else
479  -- Header not valid
480  v.tspState := DROP_S;
481  end if;
482  end if;
483  end if;
484  ----------------------------------------------------------------------
485  when DATA_S =>
486  --
487  v.segValid := '0';
488  v.segDrop := '0';
489  --
490  v.chkEn := '0';
491  v.chkStb := '0';
492 
493  -- Ready to receive further header data
494  v.tspSsiSlave := SSI_SLAVE_RDY_C;
495 
496  -- Write enable and segment address
497  if (tspSsiMaster_i.valid = '1') then
498  v.rxSegmentAddr := r.rxSegmentAddr + 1;
499  v.segmentWe := '1';
500  else
501  v.rxSegmentAddr := r.rxSegmentAddr;
502  v.segmentWe := '0';
503  end if;
504 
505  -- Wait until receiving EOF
506  if (tspSsiMaster_i.eof = '1' and tspSsiMaster_i.valid = '1') then
507 
508  -- Save tKeep of the last packet
509  v.windowArray(conv_integer(r.rxBufferAddr)).keep := tspSsiMaster_i.keep;
510 
511  -- Save packet length (+1 because it has not incremented for EOF yet)
512  v.windowArray(conv_integer(r.rxBufferAddr)).segSize := conv_integer(r.rxSegmentAddr(SEGMENT_ADDR_SIZE_G-1 downto 0))+1;
513 
514  -- Check EOF Error
515  if (tspSsiMaster_i.eofe = '0') then
516  v.tspState := VALID_S;
517  else
518  v.tspState := DROP_S;
519  end if;
520  elsif (r.tspSsiSlave.ready = '1' and r.rxSegmentAddr > rxBufferSize_i ) then
521  v.tspState := DROP_S;
522  elsif (r.tspSsiSlave.ready = '1' and r.rxSegmentAddr(SEGMENT_ADDR_SIZE_G) = '1' ) then
523  v.tspState := DROP_S;
524  end if;
525  ----------------------------------------------------------------------
526  when VALID_S =>
527  --
528  v.segValid := '1';
529  v.segDrop := '0';
530  --
531  v.chkEn := '0';
532  v.chkStb := '0';
533  v.segmentWe:= '0';
534  --
535  v.tspSsiSlave := SSI_SLAVE_NOTRDY_C;
536 
537  -- Initialize when valid SYN segment received
538  -- 1. Set the initial SeqN
539  -- 2. Initialize the buffer address
540  -- 3. Initialize window
541  if (connActive_i = '0' and r.rxF.syn = '1') then
542  v.rxF.ack := r.rxF.ack;
543  v.inOrderSeqN := r.rxSeqN;
544  v.rxBufferAddr := (others => '0');
545  v.windowArray := REG_INIT_C.windowArray;
546 
547  -- Check if next valid SEQn is received. If yes:
548  -- 1. increment the in order SEQn
549  -- 2. save seqN, type, and occupied to the current buffer address
550  -- 3. increase buffer
551  elsif ( (r.rxF.data = '1' or r.rxF.nul = '1' or r.rxF.rst = '1' ) and
552  -- Next seqN absolute difference is one
553  r.rxSeqN - r.inOrderSeqN = 1
554  ) then
555  --
556  v.windowArray(conv_integer(r.rxBufferAddr)).seqN := r.rxSeqN;
557  v.windowArray(conv_integer(r.rxBufferAddr)).segType(0) := r.rxF.data;
558  v.windowArray(conv_integer(r.rxBufferAddr)).segType(1) := r.rxF.nul;
559  v.windowArray(conv_integer(r.rxBufferAddr)).segType(2) := r.rxF.rst;
560  v.windowArray(conv_integer(r.rxBufferAddr)).occupied := '1';
561  --
562  v.inOrderSeqN := r.rxSeqN;
563  --
564  if r.rxBufferAddr < (rxWindowSize_i-1) then
565  v.rxBufferAddr := r.rxBufferAddr +1;
566  else
567  v.rxBufferAddr := (others => '0');
568  end if;
569  --
570  else
571  v.rxBufferAddr := r.rxBufferAddr;
572  v.inOrderSeqN := r.inOrderSeqN;
573  end if;
574 
575  -- Get ready to receive new packet
576  v.tspState := WAIT_SOF_S;
577 
578  ----------------------------------------------------------------------
579  when DROP_S =>
580  --
581  v.segValid := '0';
582  v.segDrop := '1';
583  --
584  v.chkEn := '0';
585  v.chkStb := '0';
586  v.segmentWe:= '0';
587  --
588  v.tspSsiSlave := SSI_SLAVE_NOTRDY_C;
589 
590  -- Get ready to receive new packet
591  v.tspState := WAIT_SOF_S;
592 
593  ----------------------------------------------------------------------
594  when others =>
595  --
596  v := REG_INIT_C;
597 
598  ----------------------------------------------------------------------
599  end case;
600 
601  ------------------------------------------------------------
602  -- TX Application side FSM:
603  -- Transmit the segments in correct order
604  -- Check the buffer if the next slot is available and send the buffer to APP
605  ------------------------------------------------------------
606 
607  -- Reset flags
608  -- These flags will hold if not overridden
609  v.appSsiMaster:= SSI_MASTER_INIT_C;
610 
611  -- Pipeline incomming slave
612  v.appSsiSlave:= appSsiSlave_i;
613 
614  case r.appState is
615  ----------------------------------------------------------------------
616  when CHECK_BUFFER_S =>
617 
618  -- Counters to 0
619  v.txSegmentAddr := (others => '0');
620  v.rxLastSeqN := r.rxLastSeqN;
621 
622  --
623  if connActive_i = '0' then
624  v.txBufferAddr := (others => '0');
625  v.rxLastSeqN := r.inOrderSeqN;
626  -- Data segment in buffer only one word long take TKEEP and apply EOF
627  elsif (r.windowArray(conv_integer(r.txBufferAddr)).occupied = '1' and
628  r.windowArray(conv_integer(r.txBufferAddr)).segType = "001" and -- Data segment type
629  r.windowArray(conv_integer(r.txBufferAddr)).segSize = 0
630  ) then
631  --
632  v.txBufferAddr := r.txBufferAddr;
633 
634  if (appSsiSlave_i.pause = '0') then
635 
636  v.appSsiMaster.sof := '1';
637  v.appSsiMaster.valid := '1';
638  v.appSsiMaster.strb := (others => '1');
639  v.appSsiMaster.dest := (others => '0');

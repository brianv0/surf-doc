81  -- State Machine
82  case r.state is
83  ----------------------------------------------------------------------
84  when IDLE_S =>
85  -- Check for data
86  if (sAxisMaster.tValid = '1') then
87  -- Drop frames when pause is asserted to avoid downstream errors
88  if (mAxisCtrl.pause = '1') and (dropOnPause = '1') then
89  -- Drop the packet

91  -- Check for no EOF

93  -- Next State
94  v.state := DROP_S;
95  end if;
96  -- Local match, broadcast or multicast
97  elsif (filtEnable = '0') or
98  (sAxisMaster.tData(47 downto 0) = macAddress) or -- Local
99  (sAxisMaster.tData(0) = '1') or -- Multicast

101  -- Check for no EOF
102  if (sAxisMaster.tLast = '0') then
103  -- Next State
104  v.state := PASS_S;
105  end if;
106  -- Drop frame
107  else
108  -- Drop the packet
109  v.mAxisMaster.tValid := '0';
110  -- Check for no EOF
111  if (sAxisMaster.tLast = '0') then
112  -- Next State
113  v.state := DROP_S;
114  end if;
115  end if;
116  end if;
117  ----------------------------------------------------------------------
118  when DROP_S =>
119  -- Drop the packet
120  v.mAxisMaster.tValid := '0';
121  -- Check for a valid EOF
122  if (sAxisMaster.tValid = '1') and (sAxisMaster.tLast = '1') then
123  -- Next State
124  v.state := IDLE_S;
125  end if;
126  ----------------------------------------------------------------------
127  when PASS_S =>
128  -- Check for a valid EOF
129  if (sAxisMaster.tValid = '1') and (sAxisMaster.tLast = '1') then
130  -- Next State
131  v.state := IDLE_S;
132  end if;
133  ----------------------------------------------------------------------
134  end case;

139  end if;

151  comb : process (axiReadMaster, axiRst, axiWriteMaster, cntOut, localMac, r, statusOut) is
152  variable v : RegType;
153  variable regCon : AxiLiteEndPointType;
154  variable rdPntr : natural;
155  begin
156  -- Latch the current value
157  v := r;
158 
159  -- Determine the transaction type
160  axiSlaveWaitTxn(regCon, axiWriteMaster, axiReadMaster, v.axiWriteSlave, v.axiReadSlave);
161 
162  -- Reset strobe signals
163  v.cntRst := '0';
164  v.config.softRst := '0';
165  v.hardRst := '0';
166 
167  -- Calculate the read pointer
168  rdPntr := conv_integer(axiReadMaster.araddr(9 downto 2));
169 
170  -- Register Mapping
171  axiSlaveRegisterR(regCon, "0000--------", 0, muxSlVectorArray(cntOut, rdPntr));
172  axiSlaveRegisterR(regCon, x"100", 0, statusOut);
173  --axiSlaveRegisterR(regCon, x"104", 0, status.macStatus.rxPauseValue);
174 
175  axiSlaveRegister(regCon, x"200", 0, v.config.macConfig.macAddress(31 downto 0));
176  axiSlaveRegister(regCon, x"204", 0, v.config.macConfig.macAddress(47 downto 32));
177  --axiSlaveRegister(regCon, x"208", 0, v.config.macConfig.byteSwap);
178 
179  --axiSlaveRegister(regCon, x"210", 0, v.config.macConfig.txShift);
180  --axiSlaveRegister(regCon, x"214", 0, v.config.macConfig.txShiftEn);
181  --axiSlaveRegister(regCon, x"218", 0, v.config.macConfig.interFrameGap);
182  axiSlaveRegister(regCon, x"21C", 0, v.config.macConfig.pauseTime);
183 
184  --axiSlaveRegister(regCon, x"220", 0, v.config.macConfig.rxShift);
185  --axiSlaveRegister(regCon, x"224", 0, v.config.macConfig.rxShiftEn);
186  axiSlaveRegister(regCon, x"228", 0, v.config.macConfig.filtEnable);
187  axiSlaveRegister(regCon, x"22C", 0, v.config.macConfig.pauseEnable);
188 
189  axiSlaveRegister(regCon, x"230", 0, v.config.configVector);
190 
191  axiSlaveRegister(regCon, x"F00", 0, v.rollOverEn);
192  axiSlaveRegister(regCon, x"FF4", 0, v.cntRst);
193  axiSlaveRegister(regCon, x"FF8", 0, v.config.softRst);
194  axiSlaveRegister(regCon, x"FFC", 0, v.hardRst);
195 
196  -- Closeout the transaction
197  axiSlaveDefault(regCon, v.axiWriteSlave, v.axiReadSlave, AXI_ERROR_RESP_G);
198 
199  -- Synchronous Reset
200  if (axiRst = '1') or (v.hardRst = '1') then
201  v.cntRst := '1';
202  v.rollOverEn := (others => '0');
203  v.config := XAUI_CONFIG_INIT_C;
204  if (axiRst = '1') then
205  v.axiReadSlave := AXI_LITE_READ_SLAVE_INIT_C;
206  v.axiWriteSlave := AXI_LITE_WRITE_SLAVE_INIT_C;
207  end if;
208  end if;
209 
210  -- Update the MAC address
211  v.config.macConfig.macAddress := localMac;
212 
213  -- Register the variable for next clock cycle
214  rin <= v;
215 
216  -- Outputs
217  axiReadSlave <= r.axiReadSlave;
218  axiWriteSlave <= r.axiWriteSlave;
219 
220  end process comb;

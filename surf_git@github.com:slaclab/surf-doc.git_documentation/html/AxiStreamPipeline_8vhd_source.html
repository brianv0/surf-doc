68  comb : process (axisRst, mAxisSlave, r, sAxisMaster) is
69  variable v : RegType;
70  variable i : natural;
71  begin
72  -- Latch the current value
73  v := r;
74 
75  -- Check if we need to shift register
76  if (r.mAxisMaster(PIPE_STAGES_C).tValid = '0') or (mAxisSlave.tReady = '1') then
77  -- Shift the data up the pipeline
78  for i in PIPE_STAGES_C downto 2 loop
79  v.mAxisMaster(i) := r.mAxisMaster(i-1);
80  end loop;
81  -- Check if the lowest cell is empty
82  if r.mAxisMaster(0).tValid = '0' then
83  -- Set the ready bit
84  v.sAxisSlave.tReady := '1';
85  -- Check if we were pulling the FIFO last clock cycle
86  if r.sAxisSlave.tReady = '1' then
87  -- Shift the FIFO data
88  v.mAxisMaster(1) := sAxisMaster;
89  else
90  -- Clear valid in stage 1
91  v.mAxisMaster(1).tValid := '0';
92  end if;
93  else
94  -- Shift the lowest cell
95  v.mAxisMaster(1) := r.mAxisMaster(0);
96  -- Check if we were pulling the FIFO last clock cycle
97  if r.sAxisSlave.tReady = '1' then
98  -- Reset the ready bit
99  v.sAxisSlave.tReady := '0';
100  -- Fill the lowest cell
101  v.mAxisMaster(0) := sAxisMaster;
102  else
103  -- Set the ready bit
104  v.sAxisSlave.tReady := '1';
105  -- Reset the lowest cell tValid
106  v.mAxisMaster(0).tValid := '0';
107  end if;
108  end if;
109  else
110  -- Reset the ready bit
111  v.sAxisSlave.tReady := '0';
112  -- Check if we were pulling the FIFO last clock cycle
113  if r.sAxisSlave.tReady = '1' then
114  -- Fill the lowest cell
115  v.mAxisMaster(0) := sAxisMaster;
116  elsif r.mAxisMaster(0).tValid = '0' then
117  -- Set the ready bit
118  v.sAxisSlave.tReady := '1';
119  end if;
120  -- Check if we need to internally shift the data to remove gaps
121  for i in PIPE_STAGES_C-1 downto 1 loop
122  -- Check for empty cell ahead of a filled cell
123  if (r.mAxisMaster(i).tValid = '0') and (r.mAxisMaster(i-1).tValid = '1') then
124  -- Shift the lowest cell
125  v.mAxisMaster(i) := r.mAxisMaster(i-1);
126  -- Reset the flag
127  v.mAxisMaster(i-1).tValid := '0';
128  end if;
129  end loop;
130  end if;
131 
132  -- Synchronous Reset
133  if axisRst = '1' then
134  v := REG_INIT_C;
135  end if;
136 
137  -- Register the variable for next clock cycle
138  rin <= v;
139 
140  -- Outputs
141  sAxisSlave <= r.sAxisSlave;
142  mAxisMaster <= r.mAxisMaster(PIPE_STAGES_C);
143 
144  end process comb;
145 
146  seq : process (axisClk) is
147  begin
148  if rising_edge(axisClk) then
149  r <= rin after TPD_G;
150  end if;
151  end process seq;
152 
153  end generate;
154 
155 end rtl;

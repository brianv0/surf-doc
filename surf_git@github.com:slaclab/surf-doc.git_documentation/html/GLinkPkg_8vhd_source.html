139  variable retVar : slv(19 downto 0) := (others => '0');

212  -----------------------------------------------------------------------------
213  -- Test if a word is valid.
214  -- Detectable error states listed on page 23 of HDMP-1032A/1034A Data Sheet
215  -----------------------------------------------------------------------------
216  function isValidWord (
217  word : GLinkWordType)
218  return boolean
219  is
220  variable validVar : boolean := true;
221  begin
222  if (std_match(word.c, "-00-") or
223  std_match(word.c, "-11-") or
224  std_match(word.c, "0101") or
225  std_match(word.c, "1010")) then
226  validVar := false;
227  elsif word.c = "1100" then --check for invalid Filled frames
228  if (std_match(word.w, "-------0--------") or
229  std_match(word.w, "-------11-------")) then
230  validVar := false;
231  end if;
232  end if;
233  return validVar;
234  end function isValidWord;

250  function isIdleWord (
251  word : GLinkWordType)
252  return boolean
253  is
254  variable retVar : boolean := false;
255  begin
256  if (word.c = GLINK_CONTROL_WORD_C) then
257  for i in GLINK_VALID_IDLE_WORDS_C'range loop
258  if (word.w = GLINK_VALID_IDLE_WORDS_C(i)) then
259  retVar := true;
260  end if;
261  end loop; -- i
262  end if;
263  return retVar;
264  end function;

295  function getControlPayload (
296  word : GLinkWordType)
297  return slv
298  is
299  variable retVar : slv(15 downto 0);
300  begin
301  retVar(6 downto 0) := bitReverse(word.w(0 to 6));
302  retVar(13 downto 7) := bitReverse(word.w(9 to 15));
303  retVar(15 downto 14) := word.w(7 to 8); -- actually don't care
304  return retVar;
305  end function;
306 
307  function getDataPayload(
308  word : GLinkWordType)
309  return slv
310  is
311  begin
312  return bitReverse(word.w);
313  end function;
314 
315  function getFlag(
316  word : GLinkWordType)
317  return sl
318  is
319  begin
320  if (word.c = GLINK_DATA_WORD_FLAG_HIGH_C or
321  word.c = GLINK_DATA_WORD_INV_FLAG_HIGH_C) then
322  return '1';
323  else
324  return '0';
325  end if;
326  end function;
